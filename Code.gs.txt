// S04.F03
function ensureSheet_(ss, name, headers, color) {
  var sheet = ss.getSheetByName(name) || ss.insertSheet(name);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(headers);
    var hdr = sheet.getRange(1, 1, 1, headers.length);
    hdr.setFontWeight('bold')
       .setBackground(color || '#2c3e50')
       .setFontColor('#ffffff')
       .setHorizontalAlignment('center');
    sheet.setFrozenRows(1);
    sheet.setRowHeight(1, 32);
  }
  return sheet;
}

// S04.F04
function applyStageDropdown_(sheet) {
  if (!sheet) return;
  try {
    var rule = SpreadsheetApp.newDataValidation()

      .build();

  } catch (e) {}
}

// S05.F02
function installTrigger_(handlerFn, type, value) {
  deleteTriggerByHandler_(handlerFn);
  var t = ScriptApp.newTrigger(handlerFn).timeBased();
  if (type === 'everyMinutes') t.everyMinutes(value).create();
  else if (type === 'atHour')  t.atHour(value).everyDays(1).create();

  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === name) ScriptApp.deleteTrigger(t);
  });
}

// S06.F01 — entry point, every 15 min trigger
function runPipeline() {
  // S06.F01-A — pause guard
  if (isPipelinePaused_()) {
    Logger.log('S06: Pipeline paused — skipping run.');
    return;
  }

  // S06.F01-B — single-instance lock (30s wait)
  var lock = LockService.getScriptLock();
  if (!lock.tryLock(30000)) {
    appendLog_({ status: 'LOCK_SKIP', notes: 'Another run already in progress.' });
    return;
  }

  try {
    var startMs = Date.now();
    _RUN_STATE   = { emails: {}, mobiles: {} };   // fresh dedup state per run

    // S06.F01-C — IST quota reset + cap check
    resetQuotaIfNewDay_();
    var qs = getQuotaState_();
    if (qs.cvTotalRemaining <= 0) {
      appendLog_({ status: 'QUOTA_CAP_HIT', notes: 'Daily total cap reached.' });
      return;
    }

    // S06.F01-D — drain pending emails from previous quota-full runs
    flushPendingEmails_();

    // S06.F01-E — decide FRESH vs BACKLOG
    var mode   = decideRunMode_();
    var maxRun = (mode === 'FRESH') ? CONFIG_QUOTA.perRunFresh : CONFIG_QUOTA.perRunBacklog;

    var label = GmailApp.getUserLabelByName(CONFIG.labels.input);
    if (!label) {
      Logger.log('S06: Label ' + CONFIG.labels.input + ' not found — run setupPipeline().');
      return;
    }

    var allThreads = label.getThreads(0, maxRun + 5);
    if (!allThreads.length) {
      Logger.log('S06: No threads to process.');
      return;
    }

    // S06.F01-F — filter by mode
    var todayIst = istDateString_();
    var threads  = [];
    for (var i = 0; i < allThreads.length; i++) {
      var msg    = allThreads[i].getMessages()[allThreads[i].getMessageCount() - 1];
      var isFresh= (istDateString_(msg.getDate()) === todayIst);
      if (mode === 'FRESH'   && isFresh)  threads.push(allThreads[i]);
      if (mode === 'BACKLOG' && !isFresh) threads.push(allThreads[i]);
      if (threads.length >= maxRun) break;
    }
    if (!threads.length) {
      Logger.log('S06: No ' + mode + ' threads found.');
      return;
    }

    Logger.log('S06: Mode=' + mode + ' | Processing ' + threads.length + ' thread(s).');

    // S06.F01-G — process each thread
    var runRejected = 0;
    var circuit = { consecutive429: 0, broken: false };

    for (var j = 0; j < threads.length; j++) {
      if (Date.now() - startMs > CONFIG.maxExecutionMs) {
        Logger.log('S06: Execution time limit — stopping this run.');
        break;
      }
      var qs2 = getQuotaState_();
      if (qs2.cvTotalRemaining <= 0) { Logger.log('S06: Quota cap mid-run.'); break; }
      if (circuit.broken) {
        sendAlert_('Circuit breaker tripped', circuit.consecutive429 + ' consecutive 429s. Stopped.');
        break;
      }
      try {
        var result = processThread_(threads[j], circuit);
        if (result === 'REJECTED') runRejected++;
      } catch (e) {
        Logger.log('S06: Thread error — ' + e.message);
        try {
          logError_(threads[j], null, null, e.message, '');
          moveLabel_(threads[j], CONFIG.labels.input, CONFIG.labels.error);
        } catch (e2) {}
      }
      if (j < threads.length - 1) Utilities.sleep(CONFIG.geminiInterCallDelayMs);
    }

    logQuota_(runRejected);
    Logger.log('S06: Run complete.');

  } finally {
    try { lock.releaseLock(); } catch (e) {}
  }
}

// S07 · THREAD PROCESSOR
// ═══════════════════════════════════════════════════════════════════
// S07.F01 — full per-thread orchestration
function processThread_(thread, circuit) {
  var messages = thread.getMessages();
  var message  = messages[messages.length - 1];
  var subject  = message.getSubject() || '';
  var body     = message.getPlainBody() || '';
  var threadId = thread.getId();

  // S07.F01-A — Extract original application date (handles forwarded CVs)
  var appDateResult = extractOriginalApplicationDate_(message);
  var appDate = appDateResult.date;
  if (appDate < CONFIG.earliestMessageDate) {
    moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.processed);
    return 'SKIPPED_DATE';
  }

  // S07.F01-B — Identify the actual candidate sender (not the forwarder)
  var rawSender    = message.getFrom();
  var actualEmail  = extractForwardedSender_(body) || parseEmailFromHeader_(rawSender);
  var senderEmail  = normalizeEmail_(actualEmail);

  // S07.F01-C — Pick best CV attachment using weighted scoring
  var cvAttachment = pickBestAttachment_(message);
  var cvLink       = '';
  var geminiParts  = [];

  if (cvAttachment) {
    if (cvAttachment.getSize() > CONFIG.maxAttachmentBytes) {
      logError_(thread, rawSender, subject, 'Attachment too large: ' + cvAttachment.getSize() + ' bytes', '');
      moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.error);
      return 'ERROR_SIZE';
    }
    cvLink = saveCvToDrive_(cvAttachment, senderEmail);
    geminiParts = buildGeminiParts_(cvAttachment, body, senderEmail, subject);
  } else if (body && body.length >= 200) {
    // Body-as-CV path (no attachment)
    geminiParts = [
      { text: buildParsePrompt_(body, senderEmail, subject) },
      { text: 'CV content from email body:\n' + body.slice(0, 30000) }
    ];
  } else {
    logError_(thread, rawSender, subject, 'No valid CV attachment and body < 200 chars.', '');
    moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.error);
    return 'ERROR_NO_CV';
  }

  // S07.F01-D — Pre-parse dedup on sender email
  if (isDuplicate_(senderEmail, null)) {
    Logger.log('S07: Duplicate sender: ' + maskEmail_(senderEmail));
    moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.duplicate);
    return 'DUPLICATE';
  }

  // S07.F01-E — Parse CV with Gemini
  var parsed = parseCvWithGemini_(geminiParts, circuit);
  if (!parsed) {
    logError_(thread, rawSender, subject, 'Gemini returned null — check API key and quota.', '');
    moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.error);
    return 'ERROR_GEMINI';
  }

  // S07.F01-F — Normalise extracted fields
  normalizeCandidate_(parsed, senderEmail);
  var candEmail  = parsed.email  || senderEmail;
  var candMobile = parsed.mobile || '';

  // S07.F01-G — Post-parse dedup on extracted email/mobile
  if (isDuplicate_(candEmail, candMobile)) {
    Logger.log('S07: Candidate duplicate: ' + maskEmail_(candEmail));
    moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.duplicate);
    return 'DUPLICATE';
  }
  updateRunState_(candEmail, candMobile);

  // S07.F01-H — Load trade rules + taxonomy
  var rules    = loadRules_();
  var taxonomy = loadTaxonomy_();
  var taxResult = mapCandidateToTaxonomy_(parsed, taxonomy);
  if (taxResult.industry && !parsed.industry) parsed.industry = taxResult.industry;
  if (taxResult.trade    && !parsed.trade)    parsed.trade    = taxResult.trade;

  // S07.F01-I — Score the candidate
  var scored = scoreCandidate_(parsed, rules, taxResult);

  // S07.F01-J — Apply decision rules for flags (never changes REJECT verdict)
  applyDecisionRules_(scored, rules);

  // S07.F01-K — Route: REJECT (age only) → Rejected sheet
  if (scored.verdict === 'REJECT') {
    writeToRejected_(scored, cvLink, appDate);
    writeToMeta_(candEmail, candMobile, parsed.full_name, parsed.industry, threadId, message.getId());
    appendConsent_(parsed, appDate, threadId);
    incrementQuota_('cv');
    moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.processed);
    appendLog_({
      status: 'REJECTED_AGE',
      name: maskName_(parsed.full_name), email: maskEmail_(candEmail),
      mobile: maskPhone_(candMobile), trade: scored.trade,
      notes: 'Age=' + (scored.age || '?') + ' | ' + scored.notes
    });
    return 'REJECTED';
  }

  // S07.F01-L — Write to Candidates sheet
  writeToSheet_(scored, threadId, cvLink, appDate);
  writeToMeta_(candEmail, candMobile, parsed.full_name, parsed.industry, threadId, message.getId());
  appendConsent_(parsed, appDate, threadId);

  appendLog_({
    status: 'PROCESSED',
    name: maskName_(parsed.full_name), email: maskEmail_(candEmail),
    mobile: maskPhone_(candMobile), trade: scored.trade,
    notes: scored.verdict + ' | Score=' + scored.score + ' | Flag=' + scored.flag
  });

  incrementQuota_('cv');

  // S07.F01-M — Send follow-up email for NEEDS_CALL with missing fields
  if (scored.verdict === 'NEEDS_CALL' && candEmail && !isInternalDomain_(candEmail)) {
    maybeSendFollowUpEmail_(scored, candEmail);
  }

  moveLabel_(thread, CONFIG.labels.input, CONFIG.labels.processed);
  Logger.log('S07: PROCESSED ' + maskName_(parsed.full_name) + ' | ' + scored.verdict + ' | ' + scored.score);
  return 'PROCESSED';
}

// S08.F01 — pick best CV attachment using weighted scoring
function pickBestAttachment_(message) {
  var atts = message.getAttachments();
  if (!atts || !atts.length) return null;
  var best = null, bestScore = -1;
  for (var i = 0; i < atts.length; i++) {
    var ext = (atts[i].getName() || '').split('.').pop().toLowerCase();
    if (CONFIG.supportedExtensions.indexOf(ext) < 0) continue;
    var s = scoreAttachment_(atts[i]);
    if (s > bestScore) { bestScore = s; best = atts[i]; }
  }
  return best;
}

// S08.F02 — weighted attachment quality score
function scoreAttachment_(att) {
  var pts  = 10;  // base
  var name = (att.getName() || '').toLowerCase();
  var ext  = name.split('.').pop();
  var size = att.getSize();
  // Name keywords
  if (/cv|resume|biodata|profile|curriculum/.test(name)) pts += 25;
  // Extension
  if      (ext === 'pdf')             pts += 15;
  else if (ext === 'docx')            pts += 10;
  else if (ext === 'doc')             pts += 6;
  else if (/jpg|jpeg|png/.test(ext))  pts += 8;
  // Size bands
  if      (size >= 100000 && size <= 2097152)  pts += 10;  // 100KB–2MB ideal
  else if (size >  2097152 && size <= 5242880) pts += 5;   // 2–5MB OK
  else if (size < 51200 || size > 8388608)     pts -= 5;   // <50KB or >8MB suspicious
  return pts;
}

// S08.F03 — build Gemini multipart input from attachment + email context
function buildGeminiParts_(attachment, emailBody, senderEmail, subject) {
  var ext  = (attachment.getName() || '').split('.').pop().toLowerCase();
  var mime = mimeFromExt_(ext);
  var parts = [{ text: buildParsePrompt_(emailBody, senderEmail, subject) }];

  if (['pdf','jpg','jpeg','png'].indexOf(ext) >= 0) {
    // Multimodal: send as base64 inline_data
    parts.push({ inline_data: {
      mime_type: mime,
      data: Utilities.base64Encode(attachment.getBytes())
    }});
  } else if (ext === 'docx') {
    // Try fast Utilities.unzip path first, fallback to Drive convert
    var text = extractTextFromDocxSafe_(attachment);
    if (!text || text.length < 50) text = extractTextFromDocViaDrive_(attachment);
    if (text && text.length >= 50) {
      parts.push({ text: 'CV document text:\n' + text.slice(0, 30000) });
    } else {
      parts.push({ inline_data: { mime_type: mime, data: Utilities.base64Encode(attachment.getBytes()) }});
    }
  } else if (ext === 'doc') {
    var docText = extractTextFromDocViaDrive_(attachment);
    if (docText && docText.length >= 50) {
      parts.push({ text: 'CV document text:\n' + docText.slice(0, 30000) });
    } else {
      parts.push({ inline_data: { mime_type: mime, data: Utilities.base64Encode(attachment.getBytes()) }});
    }
  } else {
    if (emailBody) parts.push({ text: 'Email body as CV:\n' + emailBody.slice(0, 30000) });
  }
  return parts;
}

// S08.F04 — MIME type map
function mimeFromExt_(ext) {
  return ({
    pdf:  'application/pdf',
    jpg:  'image/jpeg', jpeg: 'image/jpeg', png: 'image/png',
    doc:  'application/msword',
    docx: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'
  })[ext] || 'application/octet-stream';
}

// S08.F05 — DOCX text extraction via Utilities.unzip (fast, no Drive quota)
function extractTextFromDocxSafe_(blob) {
  try {
    var safeBlob = blob.copyBlob().setContentType('application/zip');
    var files    = Utilities.unzip(safeBlob);
    for (var i = 0; i < files.length; i++) {
      if (files[i].getName() === 'word/document.xml') {
        var xml = files[i].getDataAsString('UTF-8');
        return decodeXmlEntities_(
          xml.replace(/<w:br[^\/]*\/>/g, '\n')
             .replace(/<w:tab\/>/g, '\t')
             .replace(/<[^>]+>/g, ' ')
        ).replace(/[ \t]+\n/g, '\n').replace(/\n[ \t]+/g, '\n').replace(/\n{3,}/g, '\n\n').trim();
      }
    }
    return '';
  } catch (e) { return ''; }
}

// S08.F06 — DOC/DOCX Drive convert fallback
function extractTextFromDocViaDrive_(blob) {
  var tempFile = null;
  try {
    tempFile = Drive.Files.insert(
      { title: 'kai_cv_tmp_' + Date.now(), mimeType: 'application/vnd.google-apps.document' },
      blob, { convert: true }
    );
    return DocumentApp.openById(tempFile.id).getBody().getText() || '';
  } catch (e) {
    try {
      var raw = blob.getDataAsString('windows-1252');
      return raw.replace(/[^\x09\x0A\x0D\x20-\x7E]/g, ' ').replace(/\s+/g, ' ').trim();
    } catch (e2) { return ''; }
  } finally {
    if (tempFile) { try { Drive.Files.remove(tempFile.id); } catch (e) {} }
  }
}

// S08.F07 — XML entity decoder
function decodeXmlEntities_(t) {
  return String(t||'').replace(/&amp;/g,'&').replace(/&lt;/g,'<')
    .replace(/&gt;/g,'>').replace(/&quot;/g,'"').replace(/&apos;/g,"'");
}

// S09.F01 — core Gemini call with retry + circuit breaker
function callGemini_(parts, circuit) {
  var apiKey = validateApiKey_();
  var url    = CONFIG.geminiEndpoint + CONFIG.defaultModel + ':generateContent?key=' + encodeURIComponent(apiKey);
  var payload = {
    contents: [{ role: 'user', parts: parts }],
    generationConfig: {
      responseMimeType: 'application/json',
      temperature: 0.1, topP: 0.1,
      maxOutputTokens: CONFIG.maxOutputTokens
    }
  };

  var lastErr = null;
  for (var attempt = 1; attempt <= CONFIG.geminiRetries; attempt++) {
    try {
      Utilities.sleep(attempt > 1 ? Math.min(Math.pow(2, attempt - 1) * 1500, 30000) : 0);
      var resp = UrlFetchApp.fetch(url, {
        method: 'post', contentType: 'application/json',
        payload: JSON.stringify(payload), muteHttpExceptions: true
      });
      var code = resp.getResponseCode();
      var text = resp.getContentText();
      if (code === 200) {
        if (circuit) circuit.consecutive429 = 0;
        return text;
      }
      if (code === 429) {
        if (circuit) {
          circuit.consecutive429++;
          if (circuit.consecutive429 >= CONFIG.geminiCircuitBreakerThreshold) {
            circuit.broken = true;
            throw new Error('CIRCUIT_BROKEN after ' + circuit.consecutive429 + ' 429s');
          }
        }
        var retryMs = parseInt((resp.getHeaders()['retry-after'] || '0'), 10) * 1000 || 15000;
        Utilities.sleep(Math.min(retryMs * attempt, 60000));
      }
      lastErr = new Error('Gemini HTTP ' + code + ': ' + text.slice(0, 300));
    } catch (e) {
      if (e.message && e.message.indexOf('CIRCUIT_BROKEN') === 0) throw e;
      lastErr = e;
    }
  }
  throw lastErr || new Error('Gemini failed after ' + CONFIG.geminiRetries + ' attempts');
}

// S09.F02 — parse CV with Gemini, returning raw JSON object
function parseCvWithGemini_(parts, circuit) {
  try {
    var raw  = callGemini_(parts, circuit);
    var text = extractGeminiText_(raw);
    if (!text || text.length < 10) return null;
    return parseGeminiJson_(cleanJsonResponse_(text));
  } catch (e) {
    Logger.log('S09: parseCvWithGemini_ error: ' + e.message);
    return null;
  }
}

// S09.F04 — strip markdown fences from JSON response
function cleanJsonResponse_(text) {
  var s = String(text || '').trim();
  s = s.replace(/^```json\s*/i, '').replace(/^```\s*/i, '').replace(/```\s*$/i, '').trim();
  return repairJsonNewlines_(s);
}

// S09.F05 — 4-level JSON repair chain
function parseGeminiJson_(text) {
  // Level 1 — direct parse
  try { var r = JSON.parse(text); if (r) return r; } catch (e1) {}
  // Level 2 — strip trailing commas before ] or }
  var s2 = text.replace(/,\s*([\}\]])/g, '$1');
  try { var r2 = JSON.parse(s2); if (r2) return r2; } catch (e2) {}
  // Level 3 — escape literal newlines inside strings
  var s3 = repairJsonNewlines_(s2);
  try { var r3 = JSON.parse(s3); if (r3) return r3; } catch (e3) {}
  // Level 4 — extract first complete {…} block
  var m = text.match(/(\{[\s\S]*\})/);
  if (m) { try { var r4 = JSON.parse(m[1]); if (r4) return r4; } catch (e4) {} }
  Logger.log('S09: JSON repair failed for: ' + text.slice(0, 200));
  return null;
}

// S09.F06 — Level 3 JSON repair: escape literal newlines inside string values
function repairJsonNewlines_(s) {
  try { JSON.parse(s); return s; } catch (e) {}
  try {
    var fixed = s.replace(/("(?:[^"\\]|\\.)*")/g, function(m) {
      return m.replace(/\n/g,'\\n').replace(/\r/g,'\\r').replace(/\t/g,'\\t');
    });
    JSON.parse(fixed);
    return fixed;
  } catch (e) {}
  return s;
}

// S10.F04 — phone normalisation with GCC country-code inference
function normalizePhone_(raw) {
  if (!raw) return '';
  var s = String(raw).replace(/\s+/g, '').replace(/[().\-]/g, '');
  if (!s) return '';
  if (s.charAt(0) === '+') return s;          // already has CC
  var plain = s.replace(/^0+/, '');
  if (/^[6-9]\d{9}$/.test(plain))   return '+91'  + plain;   // India
  if (/^5\d{8}$/.test(plain))       return '+966' + plain;   // Saudi
  if (/^5[0-9]\d{7}$/.test(plain))  return '+971' + plain;   // UAE
  if (/^[3-5]\d{7}$/.test(plain))   return '+974' + plain;   // Qatar
  if (/^[36]\d{7}$/.test(plain))    return '+973' + plain;   // Bahrain
  if (/^[79]\d{7}$/.test(plain))    return '+968' + plain;   // Oman
  if (/^[569]\d{7}$/.test(plain))   return '+965' + plain;   // Kuwait
  return s;
}

// S10.F05 — DOB to YYYY-MM-DD (7 format parsers)
function normalizeDob_(raw) {
  if (!raw) return '';
  var s = String(raw).trim();
  if (/^(n\/a|na|null|none|not specified)$/i.test(s)) return '';
  // ISO already
  if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s;
  // dd/mm/yyyy or dd-mm-yyyy or dd.mm.yyyy
  var m = s.match(/^(\d{1,2})[\/\-\.](\d{1,2})[\/\-\.](\d{4})$/);
  if (m) {
    var d=parseInt(m[1],10), mo=parseInt(m[2],10), y=parseInt(m[3],10);
    if (d<=31 && mo<=12) return y+'-'+pad2_(mo)+'-'+pad2_(d);
  }
  // dd MMM yyyy or dd MMMM yyyy
  var MONTHS={jan:1,feb:2,mar:3,apr:4,may:5,jun:6,jul:7,aug:8,sep:9,oct:10,nov:11,dec:12};
  var m2 = s.toLowerCase().match(/(\d{1,2})\s+([a-z]{3,})\s+(\d{4})/);
  if (m2) {
    var mk = m2[2].slice(0,3);
    if (MONTHS[mk]) return m2[3]+'-'+pad2_(MONTHS[mk])+'-'+pad2_(parseInt(m2[1],10));
  }
  // yyyy/mm/dd
  var m3 = s.match(/^(\d{4})[\/\-\.](\d{2})[\/\-\.](\d{2})$/);
  if (m3) return m3[1]+'-'+m3[2]+'-'+m3[3];
  // Native Date.parse fallback
  var parsed = new Date(s);
  if (!isNaN(parsed.getTime())) {
    return parsed.getFullYear()+'-'+pad2_(parsed.getMonth()+1)+'-'+pad2_(parsed.getDate());
  }
  return s;  // return raw if unparseable
}

// S10.F06 — compute age from DOB (handles Date, Excel serial, string)
function computeAge_(dob) {
  if (!dob && dob !== 0) return null;
  var d;
  if (typeof dob === 'number' && dob > 25000 && dob < 80000) {
    d = new Date(Math.round((dob - 25569) * 86400 * 1000));
  } else if (dob instanceof Date) {
    d = dob;
  } else {
    d = new Date(String(dob).replace(/^(\d{4}-\d{2}-\d{2}).*/, '$1'));
  }
  if (!d || isNaN(d.getTime())) return null;
  var now = new Date();
  var age = now.getFullYear() - d.getFullYear();
  if (now.getMonth() < d.getMonth() ||
     (now.getMonth() === d.getMonth() && now.getDate() < d.getDate())) age--;
  return (age >= 10 && age <= 90) ? age : null;
}

// S10.F07 — Gulf country normalisation
function normalizeGulfExperience_(raw) {
  if (!raw) return '';
  var s = String(raw);
  s = s.replace(/\b(dubai|sharjah|abu dhabi|ajman|fujairah|uae|united arab emirates)\b/gi, 'UAE');
  s = s.replace(/\b(saudi|ksa|riyadh|jeddah|dammam|saudi arabia)\b/gi,                'KSA');
  s = s.replace(/\b(qatar|doha|qatarenergy|qatar energy)\b/gi,                         'Qatar');
  s = s.replace(/\b(bahrain|manama|bapco)\b/gi,                                        'Bahrain');
  s = s.replace(/\b(oman|muscat|sohar|pdo)\b/gi,                                       'Oman');
  s = s.replace(/\b(kuwait|knpc|kuwait)\b/gi,                                          'Kuwait');
  return s.trim();
}

// S10.F08
function normalizeGulfCountry_(c) {
  if (!c) return '';
  var s = String(c).toLowerCase().trim();
  if (/uae|dubai|abu dhabi|sharjah|emirates/.test(s)) return 'UAE';
  if (/saudi|ksa|riyadh|jeddah|dammam/.test(s))       return 'KSA';
  if (/qatar|doha/.test(s))                            return 'Qatar';
  if (/bahrain|manama/.test(s))                        return 'Bahrain';
  if (/oman|muscat/.test(s))                           return 'Oman';
  if (/kuwait/.test(s))                                return 'Kuwait';
  return c.charAt(0).toUpperCase() + c.slice(1).toLowerCase();
}

// S10.F09 — KAI Assessment 4-line format validator + fallback builder
function normalizeKaiAssessment_(raw, candidate) {
  if (raw && String(raw).trim().length > 30) {
    var s = String(raw).trim();
    // Already has 4-line format
    if (/✓|⚠|★|→/.test(s)) return s;
  }
  // Build fallback
  var name  = (candidate && candidate.full_name)   || 'Candidate';
  var trade = (candidate && candidate.trade)        || 'general labour';
  var exp   = (candidate && candidate.total_exp_yrs)|| 0;
  var nat   = (candidate && candidate.nationality)  || '';
  var gulf  = (candidate && candidate.gulf_experience_detail) || '';
  return (
    '✓ Strengths: ' + exp + 'yr experience in ' + trade +
      (nat ? ', ' + nat + ' national' : '') + '.\n' +
    '⚠ Concerns: Automated extraction — manual verification of key details advised.\n' +
    '★ Best Fit: ' + trade + (gulf ? ' with Gulf (' + gulf + ') exposure.' : '.') + '\n' +
    '→ Action: Screen CV and verify DOB, certifications, and mobile.'
  );
}

// S10.F10 — zero-pad helper

// S11 · SCORING ENGINE
// ═══════════════════════════════════════════════════════════════════
// S11.F01 — master scoring function; returns scored object
function scoreCandidate_(parsed, rules, taxResult) {
  var score     = 0;
  var breakdown = [];
  var flags     = [];
  var notes     = [];
  var agePriority = '';

  // ── S11.1  Experience (max 30) ─────────────────────────────────
  var exp = parseFloat(parsed.total_exp_yrs) || 0;
  var expPts = 0;
  if      (exp >= 15) expPts = 30;
  else if (exp >= 10) expPts = 25;
  else if (exp >= 7)  expPts = 20;
  else if (exp >= 5)  expPts = 15;
  else if (exp >= 3)  expPts = 10;
  else if (exp >= 1)  expPts = 5;
  if (expPts) { score += expPts; breakdown.push('Exp:' + expPts); }

  // ── S11.2  Gulf Experience (max 20) ───────────────────────────
  var gulf = String(parsed.gulf_experience_detail || '').toLowerCase();
  var gulfPts = 0;
  var gulfYrsMch = gulf.match(/(\d+(?:\.\d+)?)\s*(?:yr|year)/i);
  var gulfYrs = gulfYrsMch ? parseFloat(gulfYrsMch[1]) : 0;
  if (gulf && gulf.length > 3) {
    if      (gulfYrs >= 5)  gulfPts = 20;
    else if (gulfYrs >= 3)  gulfPts = 15;
    else if (gulfYrs >= 1)  gulfPts = 10;
    else                    gulfPts = 8;   // mentioned but no numeric years
  }
  if (gulfPts) { score += gulfPts; breakdown.push('Gulf:' + gulfPts); }

  // ── S11.3  Education (max 15) ─────────────────────────────────
  var edu = String(parsed.education || '').toLowerCase();
  var eduPts = 0;
  if      (/phd|doctorate/.test(edu))                                     eduPts = 15;
  else if (/master|mba|msc|m\.sc|m\.eng/.test(edu))                       eduPts = 13;
  else if (/bachelor|b\.sc|b\.eng|b\.tech|b\.e\.|degree/.test(edu))       eduPts = 10;
  else if (/diploma|polytechnic|advanced cert/.test(edu))                 eduPts = 7;
  else if (/certificate|iti|vocational|matric|sslc|10th|12th/.test(edu))  eduPts = 4;
  if (eduPts) { score += eduPts; breakdown.push('Edu:' + eduPts); }

  // ── S11.4  Gulf Gov / Major Company (max 10) ──────────────────
  var allText = (String(parsed.government_companies || '') +
                ' ' + String(parsed.kai_assessment || '') +
                ' ' + gulf).toLowerCase();
  var govPts = 0;
  for (var gi = 0; gi < CONFIG.gulfGovCompanies.length; gi++) {
    if (allText.indexOf(CONFIG.gulfGovCompanies[gi]) >= 0) { govPts = 10; break; }
  }
  if (govPts) { score += govPts; breakdown.push('GovCo:' + govPts); }

  // ── S11.5  Trade Certifications (max 10) ──────────────────────
  var certText = (String(parsed.certifications || '') + ' ' + allText).toLowerCase();
  var certPts = 0;
  for (var ci = 0; ci < CONFIG.tradeCerts.length; ci++) {
    if (certText.indexOf(CONFIG.tradeCerts[ci]) >= 0) { certPts = 10; break; }
  }
  if (certPts) { score += certPts; breakdown.push('Cert:' + certPts); }

  // ── S11.6  GCC License / Visa Status (max 5) ──────────────────
  var licPts = 0;
  for (var li = 0; li < CONFIG.gccLicenses.length; li++) {
    if (certText.indexOf(CONFIG.gccLicenses[li]) >= 0) { licPts = 5; break; }
  }
  if (licPts) { score += licPts; breakdown.push('License:' + licPts); }

  // ── S11.7  Taxonomy bonus (max 5) ─────────────────────────────
  var taxPts = taxResult ? (taxResult.pts || 0) : 0;
  if (taxPts) { score += taxPts; breakdown.push('Tax:' + taxPts); }

  // ── S11.8  AGE BAND SCORING (max 10) ─────────────────────────
  //  CRITICAL: REJECT only on age < 21 or > 50.
  //  Score alone NEVER causes REJECT. Never.
  var age = computeAge_(parsed.date_of_birth);
  if (age !== null) {
    var agePts = 0;
    if      (age >= 26 && age <= 35) { agePts = 10; agePriority = 'P1'; }
    else if (age >= 36 && age <= 45) { agePts = 8;  agePriority = 'P1'; }
    else if (age >= 22 && age <= 25) { agePts = 5;  agePriority = 'P2'; }
    else if (age >= 46 && age <= 49) { agePts = 3;  agePriority = 'P2'; }
    else if (age === 21 || age === 50){ agePts = 2;  agePriority = 'P2'; }
    // age < 21 or > 50 → agePts=0 → AGE_OUT_OF_RANGE → REJECT

    if (agePts) { score += agePts; breakdown.push('Age:' + agePts); }
    parsed.age = age;

    if (age < 21 || age > 50) {
      flags.push('AGE_OUT_OF_RANGE');
      notes.push('Age ' + age + ' is outside 21–50 range — direct REJECT.');
    } else if (agePriority === 'P2') {
      notes.push('[AGE_P2] Age ' + age + ' in second-priority band (22-25 or 46-49).');
    }
  } else {
    parsed.age = '';
    notes.push('[NO_DOB] Age could not be determined — verification needed.');
    flags.push('AGE_UNKNOWN');
  }

  // ── S11.9  Trade rule check (flags only, NEVER REJECT) ────────
  var rule = findRule_(rules, parsed.trade);
  if (rule && exp < rule.minYears) {
    flags.push('BELOW_TRADE_MIN_EXP');
    notes.push('[TRADE_RULE] ' + parsed.trade + ' requires ' + rule.minYears + 'yr min; candidate has ' + exp + 'yr.');
  }

  // ── S11.10  Verdict assignment ────────────────────────────────
  var flag, verdict;
  if (flags.indexOf('AGE_OUT_OF_RANGE') >= 0) {
    flag = 'RED'; verdict = 'REJECT';
  } else if (score >= SCORING.greenMin) {
    flag    = agePriority === 'P2' ? 'YELLOW' : 'GREEN';
    verdict = agePriority === 'P2' ? 'NEEDS_REVIEW' : 'SHORTLISTED';
  } else if (score >= SCORING.yellowMin) {
    flag = 'YELLOW'; verdict = 'NEEDS_REVIEW';
  } else {
    flag = 'ORANGE'; verdict = 'NEEDS_CALL';  // low score + valid age = NEVER REJECT
  }

  return {
    full_name:       parsed.full_name       || '',
    email:           parsed.email           || '',
    mobile:          parsed.mobile          || '',
    nationality:     parsed.nationality     || '',
    dob:             parsed.date_of_birth   || '',
    age:             parsed.age             !== undefined ? parsed.age : '',
    education:       parsed.education       || '',
    positionApplied: parsed.current_position|| '',
    trade:           parsed.trade           || '',
    industry:        parsed.industry        || '',
    experience:      exp,
    gulfExperience:  parsed.gulf_experience_detail || '',
    score:           score,
    scoreBreakdown:  breakdown.join(' | '),
    flag:            flag,
    verdict:         verdict,
    notes:           notes.join(' | '),
    flags:           flags,
    agePriority:     agePriority,
    kaiAssessment:   parsed.kai_assessment   || '',
    recommendedRoles:parsed.recommended_roles|| '',
    recruiterAction: '',
    _rule:           rule
  };
}

// S12 · DECISION RULES ENGINE
// ═══════════════════════════════════════════════════════════════════
// S12.F01 — enrich flags and recruiterAction; never changes REJECT verdict
function applyDecisionRules_(scored, rules) {
  if (scored.verdict === 'REJECT') return;  // age-out: do not touch

  var actions = [];
  var f = scored.flags || [];

  // S12.1 — Trade not identified
  if (!scored.trade || scored.trade.length < 2) {
    if (f.indexOf('TRADE_UNCLEAR') < 0) f.push('TRADE_UNCLEAR');
    actions.push('Trade/skill unclear — verify with candidate.');
  }

  // S12.2 — Education missing
  if (!scored.education) {
    if (f.indexOf('EDUCATION_MISSING') < 0) f.push('EDUCATION_MISSING');
    actions.push('Education level not confirmed.');
  }

  // S12.3 — Gulf experience unverified (mentioned but no duration/countries)
  if (scored.gulfExperience && !scored.gulfExperience.match(/\d/)) {
    if (f.indexOf('GULF_UNVERIFIED') < 0) f.push('GULF_UNVERIFIED');
    actions.push('Gulf experience mentioned but duration not confirmed.');
  }

  // S12.4 — Low experience on NEEDS_REVIEW → downgrade to NEEDS_CALL
  if (scored.experience < SCORING.yellowFlagMinYears && scored.verdict === 'NEEDS_REVIEW') {
    scored.verdict = 'NEEDS_CALL';
    scored.flag    = 'ORANGE';
    actions.push('Experience < 2yr — downgraded to NEEDS_CALL.');
  }

  // S12.5 — Age P2 on SHORTLISTED → downgrade to NEEDS_REVIEW (per policy)
  if (scored.agePriority === 'P2' && scored.verdict === 'SHORTLISTED') {
    scored.verdict = 'NEEDS_REVIEW';
    scored.flag    = 'YELLOW';
    actions.push('[AGE_P2] Second priority band — NEEDS_REVIEW, not SHORTLISTED.');
  }

  // S12.6 — Mobile missing
  if (!scored.mobile) {
    if (f.indexOf('MOBILE_MISSING') < 0) f.push('MOBILE_MISSING');
    actions.push('No mobile number — follow-up required.');
  }

  // S12.7 — Below trade minimum years
  if (f.indexOf('BELOW_TRADE_MIN_EXP') >= 0 && scored.verdict === 'NEEDS_REVIEW') {
    scored.verdict = 'NEEDS_CALL';
    scored.flag    = 'ORANGE';
    actions.push('Below trade minimum experience threshold.');
  }

  scored.flags = f;
  if (actions.length) {
    scored.recruiterAction = actions.join(' | ');
  }
}

// S14.F01 — IST date string (yyyy-MM-dd)
function istDateString_(d) {
  return Utilities.formatDate(d || new Date(), 'Asia/Kolkata', 'yyyy-MM-dd');
}

// S14.F02 — current IST hour (0-23)
function istHour_() {
  return parseInt(Utilities.formatDate(new Date(), 'Asia/Kolkata', 'H'), 10);
}

// S14.F03 — FRESH if < 23:00 IST, BACKLOG after
function decideRunMode_() {
  return istHour_() >= CONFIG_QUOTA.backlogHourIst ? 'BACKLOG' : 'FRESH';
}

// S14.F04 — reset all quota buckets at IST midnight
function resetQuotaIfNewDay_() {
  var props = PropertiesService.getScriptProperties();
  var today = istDateString_();
  if (props.getProperty(CONFIG_QUOTA.lastResetKey) === today) return;
  var reset = {};
  var p = CONFIG_QUOTA.keyPrefix;
  reset[p + 'cv_fresh']  = '0';
  reset[p + 'cv_total']  = '0';
  reset[p + 'reply']     = '0';
  reset[p + 'email']     = '0';
  reset[CONFIG_QUOTA.lastResetKey] = today;
  props.setProperties(reset);
}

// S14.F05 — read all quota buckets
function getQuotaState_() {
  var props = PropertiesService.getScriptProperties();
  var p     = CONFIG_QUOTA.keyPrefix;
  var cvF   = parseInt(props.getProperty(p+'cv_fresh') || '0', 10);
  var cvT   = parseInt(props.getProperty(p+'cv_total') || '0', 10);
  var rep   = parseInt(props.getProperty(p+'reply')    || '0', 10);
  var em    = parseInt(props.getProperty(p+'email')    || '0', 10);
  return {
    cvFreshUsed: cvF,   cvFreshRemaining:  Math.max(0, CONFIG_QUOTA.capFreshCv    - cvF),
    cvTotalUsed: cvT,   cvTotalRemaining:  Math.max(0, CONFIG_QUOTA.capDailyTotal - cvT),
    replyUsed:   rep,   replyRemaining:    Math.max(0, CONFIG_QUOTA.capReplyParse - rep),
    emailUsed:   em,    emailRemaining:    Math.max(0, CONFIG_QUOTA.capEmail      - em)
  };
}

// S14.F06 — increment a quota bucket
function incrementQuota_(bucket) {
  var props = PropertiesService.getScriptProperties();
  var p     = CONFIG_QUOTA.keyPrefix;
  var MAP   = { cv: [p+'cv_total', p+'cv_fresh'], reply: [p+'reply'], email: [p+'email'] };
  (MAP[bucket] || []).forEach(function(k) {
    props.setProperty(k, String(parseInt(props.getProperty(k)||'0',10)+1));
  });
}

// S14.F07 — log daily quota summary to _Quota sheet
function logQuota_(runRejected) {
  var sheet = getMasterSS_().getSheetByName(CONFIG.quotaSheetName);
  if (!sheet) return;
  var qs    = getQuotaState_();
  var today = istDateString_();
  var data  = sheet.getLastRow() < 2 ? []
    : sheet.getRange(2, 1, sheet.getLastRow() - 1, 7).getValues();
  // Update existing today row
  for (var i = 0; i < data.length; i++) {
    var rowDate = data[i][0] instanceof Date
      ? istDateString_(data[i][0])
      : String(data[i][0]).slice(0, 10);
    if (rowDate === today) {
      sheet.getRange(i + 2, 1, 1, 7).setValues([[
        data[i][0], qs.cvTotalUsed, (runRejected || 0) + Number(data[i][2] || 0),
        qs.emailUsed, qs.replyUsed, Number(data[i][5] || 0), ''
      ]]);
      return;
    }
  }
  sheet.appendRow([new Date(), qs.cvTotalUsed, runRejected||0, qs.emailUsed, qs.replyUsed, 0, '']);
}

// S15.F01 — check run-state and meta sheet for duplicates
function isDuplicate_(email, mobile) {
  var eKey = normalizeEmailKey_(email);
  var mKey = normalizeMobileKey_(mobile);
  // Run-level
  if (eKey && _RUN_STATE.emails[eKey])   return true;
  if (mKey && _RUN_STATE.mobiles[mKey])  return true;
  // Meta sheet
  var sheet = getMasterSS_().getSheetByName(CONFIG.metaSheetName);
  if (!sheet || sheet.getLastRow() < 2)  return false;
  var data = sheet.getRange(2, 1, sheet.getLastRow()-1, 3).getValues();
  for (var i = 0; i < data.length; i++) {
    if (eKey && normalizeEmailKey_(data[i][1])  === eKey)  return true;
    if (mKey && normalizeMobileKey_(data[i][2]) === mKey)  return true;
  }
  return false;
}

// S15.F02 — register new candidate in run-level state
function updateRunState_(email, mobile) {
  var eKey = normalizeEmailKey_(email);
  var mKey = normalizeMobileKey_(mobile);
  if (eKey) _RUN_STATE.emails[eKey]   = true;
  if (mKey) _RUN_STATE.mobiles[mKey]  = true;
}

// S15.F03
function normalizeEmailKey_(e) {
  return e ? String(e).toLowerCase().trim().replace(/\s+/g,'') : '';
}

// S15.F04
function normalizeMobileKey_(m) {
  return m ? String(m).replace(/^'/,'').replace(/[^0-9]/g,'').replace(/^0+/,'').slice(-10) : '';
}

// S16.F01 — load _Rules sheet into array
function loadRules_() {
  var sheet = getMasterSS_().getSheetByName(CONFIG.rulesSheetName);
  if (!sheet || sheet.getLastRow() < 2) return [];
  return sheet.getRange(2, 1, sheet.getLastRow()-1, 4).getValues()
    .filter(function(r) { return r[0]; })
    .map(function(r) {
      return {
        pattern:  String(r[0]).toLowerCase().trim(),
        minYears: parseFloat(r[1]) || 0,
        minScore: parseFloat(r[2]) || 0,
        notes:    String(r[3] || '')
      };
    });
}

// S16.F02 — find matching rule for a trade
function findRule_(rules, trade) {
  if (!rules || !trade) return null;
  var t = String(trade).toLowerCase();
  for (var i = 0; i < rules.length; i++) {
    if (t.indexOf(rules[i].pattern) >= 0) return rules[i];
  }
  return null;
}

// S17.F01 — load Taxonomy sheet with 1800s CacheService
function loadTaxonomy_() {
  var cache   = CacheService.getScriptCache();
  var cached  = cache.get('kai_taxonomy_v1');
  if (cached) { try { return JSON.parse(cached); } catch (e) {} }
  var sheet = getMasterSS_().getSheetByName(CONFIG.taxonomySheetName);
  if (!sheet || sheet.getLastRow() < 2) return [];
  var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, 4).getValues()
    .filter(function(r) { return r[0] || r[2]; })
    .map(function(r) {
      return {
        industry:       String(r[0]||'').toLowerCase(),
        department:     String(r[1]||'').toLowerCase(),
        trade:          String(r[2]||'').toLowerCase(),
        specialization: String(r[3]||'').toLowerCase()
      };
    });
  try { cache.put('kai_taxonomy_v1', JSON.stringify(rows), 1800); } catch (e) {}
  return rows;
}

// S17.F02 — map candidate to taxonomy; returns { pts, industry, trade }
function mapCandidateToTaxonomy_(candidate, taxonomy) {
  if (!taxonomy || !taxonomy.length) return { pts: 0, industry: '', trade: '' };
  var trade    = String(candidate.trade    || '').toLowerCase();
  var industry = String(candidate.industry || '').toLowerCase();
  var pts = 0, matchedInd = '', matchedTrade = '';
  for (var i = 0; i < taxonomy.length; i++) {
    var row = taxonomy[i];
    if (trade && row.trade          && trade.indexOf(row.trade) >= 0)
      { pts += 3; matchedTrade = matchedTrade || row.trade; matchedInd = matchedInd || row.industry; }
    if (industry && row.industry    && industry.indexOf(row.industry) >= 0)
      { pts += 2; matchedInd = matchedInd || row.industry; }
    if (trade && row.specialization && trade.indexOf(row.specialization) >= 0)
      { pts += 1; }
    if (pts >= 6) break;
  }
  return { pts: Math.min(pts, 5), industry: matchedInd, trade: matchedTrade };
}

    FOLLOWUP_CONFIG.missingFieldChecks.forEach(function(f) {
      var v = scored[f.key];
      if (v === null || v === undefined || String(v).trim() === '') missing.push(f);
    });
    if (!missing.length) return;

    var qs = getQuotaState_();
    var token   = buildKarToken_();
    var subject = FOLLOWUP_CONFIG.subjectPrefix + token + FOLLOWUP_CONFIG.subjectSuffix;
    var body    = buildFollowUpEmailBody_(scored.full_name, missing, token);

    if (!isCandidateSafeEmailBody_(body)) {
      appendLog_({ status: 'FOLLOWUP_BLOCKED', notes: 'Body firewall triggered for ' + maskEmail_(toEmail) });
      return;
    }

    if (qs.emailRemaining <= 0) {
      queuePendingEmail_(toEmail, subject, body, token);
      appendLog_({ status: 'FOLLOWUP_QUEUED', email: maskEmail_(toEmail), notes: 'Email cap reached — queued.' });
      return;
    }

    GmailApp.sendEmail(toEmail, subject, body, { name: CONFIG.brand.senderDisplayName });
    incrementQuota_('email');

    // Store KAR token in scored.notes for reply matching
    var notesObj = {};
    try { notesObj = JSON.parse(scored.notes || '{}'); } catch (e) {}
    notesObj[FOLLOWUP_CONFIG.noteKey] = {
      token: token, sentAt: new Date().toISOString(),
      status: 'sent', missing: missing.map(function(m) { return m.key; })
    };
    scored.notes = JSON.stringify(notesObj);
    scored.recruiterAction = (scored.recruiterAction ? scored.recruiterAction + ' | ' : '') +
      'Auto email sent — awaiting reply (missing: ' + missing.map(function(m) { return m.label; }).join(', ') + ')';

    appendLog_({ status: 'FOLLOWUP_SENT', email: maskEmail_(toEmail),
                 trade: scored.trade, notes: 'KAR-' + token });
  } catch (e) {
    appendLog_({ status: 'FOLLOWUP_ERROR', notes: 'maybeSendFollowUpEmail_: ' + e.message });
  }
}

// S19.F03 — build candidate-facing email body
function buildFollowUpEmailBody_(name, missingFields, token) {
  var greeting = name ? 'Dear ' + name + ',' : 'Dear Applicant,';
  var list = missingFields.map(function(f, i) { return (i + 1) + '. ' + f.label; }).join('\n');
  return greeting + '\n\n' +
    'Thank you for your interest in a position with ' + CONFIG.brand.companyName + '.\n\n' +
    'To proceed with your application, we need the following information:\n\n' +
    list + '\n\n' +
    'Kindly reply to this email with the details above at your earliest convenience.\n\n' +
    'Warm regards,\n' + CONFIG.brand.signatureLine1 + '\n' + CONFIG.brand.signatureLine2 + '\n\n' +
    '---\n[Reference: KAR-' + token + ' — please keep this in your reply subject line]';
}

// S19.F04 — internal-term firewall: confirm no classified data in body
function isCandidateSafeEmailBody_(body) {
  var lower = String(body || '').toLowerCase();
  var blocked = [
    'shortlisted','rejected','needs_call','needs_review','verdict',
    'flags:','orange flag','red flag','kai assessment',
    'score breakdown','score:','trade test','gamca','svp',
    'manpower','deployment','recruiter action'
  ];
  for (var i = 0; i < blocked.length; i++) {
    if (lower.indexOf(blocked[i]) !== -1) return false;
  }
  return true;
}

// S19.F05 — generate KAR token
function buildKarToken_() {
  return Utilities.getUuid().replace(/-/g,'').slice(0, 8).toUpperCase();
}

// S19.F06 — queue email to _PendingEmails when cap is reached
function queuePendingEmail_(toEmail, subject, body, token) {
  var sheet = getMasterSS_().getSheetByName(CONFIG.pendingEmailsSheet);
  if (!sheet) return;
  sheet.appendRow([new Date(), toEmail, subject, body, token || '', 'QUEUED']);
}

// S19.F07 — flush queued emails when quota is available
function flushPendingEmails_() {
  var sheet = getMasterSS_().getSheetByName(CONFIG.pendingEmailsSheet);
  if (!sheet || sheet.getLastRow() < 2) return;
  var data = sheet.getRange(2, 1, sheet.getLastRow()-1, 6).getValues();
  for (var i = 0; i < data.length; i++) {
    if (data[i][5] !== 'QUEUED') continue;
    if (getQuotaState_().emailRemaining <= 0) break;
    try {
      GmailApp.sendEmail(String(data[i][1]), String(data[i][2]), String(data[i][3]),
                         { name: CONFIG.brand.senderDisplayName });
      incrementQuota_('email');
      sheet.getRange(i + 2, 6).setValue('SENT');
    } catch (e) {
      sheet.getRange(i + 2, 6).setValue('ERROR: ' + e.message.slice(0, 60));
    }
  }
}

// S20 · REPLY SCANNER  (10-min trigger)
// ═══════════════════════════════════════════════════════════════════
// S20.F01 — scan inbox for KAR-token replies
function processCandidateReplies() {
  if (isPipelinePaused_()) return;

  var lock = LockService.getScriptLock();
  if (!lock.tryLock(15000)) return;

  try {
    resetQuotaIfNewDay_();
    var qs = getQuotaState_();
    if (qs.replyRemaining <= 0) {
      appendLog_({ status: 'REPLY_QUOTA_CAP', notes: 'Reply parse cap reached.' });
      return;
    }

    var threads = GmailApp.search(FOLLOWUP_CONFIG.replySearchQuery, 0,
                                   FOLLOWUP_CONFIG.replyMaxThreadsPerRun);
    if (!threads || !threads.length) return;

    var index = buildFollowUpIndex_();

    for (var t = 0; t < threads.length; t++) {
      if (getQuotaState_().replyRemaining <= 0) break;
      try {
        var thread  = threads[t];
        var subject = thread.getFirstMessageSubject() || '';
        var m = subject.match(/Ref:\s*KAR-([A-Z0-9]+)/i);
        if (!m) { thread.markRead(); continue; }
        var token = m[1].toUpperCase();

        var hit = index[token];
        if (!hit) { thread.markRead(); continue; }

        var msgs     = thread.getMessages();
        var ownerLc  = getOwnerEmail_().toLowerCase();
        var replyMsg = null;
        for (var j = msgs.length - 1; j >= 0; j--) {
          if (String(msgs[j].getFrom() || '').toLowerCase().indexOf(ownerLc) === -1) {
            replyMsg = msgs[j]; break;
          }
        }
        if (!replyMsg) { thread.markRead(); continue; }

        var replyBody = replyMsg.getPlainBody() || '';
        if (replyBody.length < 5) { thread.markRead(); continue; }

        var parsed = callGeminiForReply_(replyBody);
        if (parsed) {
          applyFollowUpReplyToRow_(hit.sheet, hit.rowNumber, hit.notesObj, token, parsed);
          incrementQuota_('reply');
          appendLog_({ status: 'REPLY_PROCESSED', notes: 'KAR-' + token + ' row=' + hit.rowNumber });
        }
        thread.markRead();
      } catch (e) {
        appendLog_({ status: 'REPLY_ERROR', notes: e.message });
      }
    }

    // ── Phase 2: "Action Required" replies (email-matched, v289) ───────
    // Handles candidates who replied to "Action Required: Complete Your Application"
    // emails which have NO KAR token — matched by sender email address instead.
    if (typeof processActionRequiredRepliesV289_ === 'function') {
      processActionRequiredRepliesV289_();
    }

  } finally {
    try { lock.releaseLock(); } catch (e) {}
  }
}

// S20.F02 — build token→row index from Candidates sheet Notes column
function buildFollowUpIndex_() {
  var index = {};
  var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
  if (!sheet || sheet.getLastRow() < 2) return index;
  var notes = sheet.getRange(2, CONFIG.inputColumns.notes, sheet.getLastRow()-1, 1).getValues();
  for (var i = 0; i < notes.length; i++) {
    var raw = String(notes[i][0] || '').trim();
    if (!raw || raw.indexOf('followUp') < 0) continue;
    try {
      var obj = JSON.parse(raw);
      if (!obj.followUp || !obj.followUp.token) continue;
      if (obj.followUp.status === 'received')   continue;
      index[obj.followUp.token.toUpperCase()] = {
        sheet: sheet, rowNumber: i + 2, notesObj: obj
      };
    } catch (e) {}
  }
  return index;
}

// S20.F03 — Gemini lightweight reply parser
function callGeminiForReply_(replyBody) {
  try {
    var apiKey = validateApiKey_();
    var url    = CONFIG.geminiEndpoint + CONFIG.defaultModel +
                ':generateContent?key=' + encodeURIComponent(apiKey);
    var prompt =
      'Extract candidate information from this recruitment reply email.\n' +
      'Return ONLY valid JSON with ALL these fields:\n' +
      '{"date_of_birth":"","nationality":"","education":"","mobile":"","trade":"",' +
      '"employment_status":"","passport_type":"","current_location":"","notice_period_days":""}\n' +
      'Rules:\n' +
      '- passport_type: "ECNR" if ECNR or Non-ECR mentioned, "ECR" if ECR mentioned, "" otherwise\n' +
      '- employment_status: exactly one of: "Available Immediately" | "Working - Notice Period" | "Working" | ""\n' +
      '- notice_period_days: number only, e.g. "40" for "40 days notice period", "" if not mentioned\n' +
      '- current_location: city and/or country if stated, "" otherwise\n' +
      '- Use "" for any field not found. Never guess or invent data.\n\n' +
      'Reply email:\n' + replyBody.slice(0, 3000);
    var resp = UrlFetchApp.fetch(url, {
      method: 'post', contentType: 'application/json',
      payload: JSON.stringify({
        contents: [{ role: 'user', parts: [{ text: prompt }] }],
        generationConfig: { responseMimeType: 'application/json', temperature: 0.1, maxOutputTokens: 500 }
      }), muteHttpExceptions: true
    });
    if (resp.getResponseCode() !== 200) return null;
    return parseGeminiJson_(cleanJsonResponse_(extractGeminiText_(resp.getContentText())));
  } catch (e) { return null; }
}

// S20.F04 — apply parsed reply fields to candidate row + re-run decision rules
function applyFollowUpReplyToRow_(sheet, rowNum, notesObj, token, parsed) {
  var col = CONFIG.inputColumns;
  var ec  = CONFIG_V2.extCol;
  // Read enough columns to cover all extended fields (up to col 38)
  var readCols = Math.max(CONFIG.headers.length, ec.noticeDays || 32, 38);
  var row = sheet.getRange(rowNum, 1, 1, readCols).getValues()[0];
  var upd = {};

  function empty_(v) { return !v || String(v).trim() === ''; }

  // ── Original 5 fields ──────────────────────────────────────────────
  if (parsed.date_of_birth && empty_(row[col.dob - 1])) {
    upd[col.dob] = normalizeDob_(parsed.date_of_birth);
    var age = computeAge_(upd[col.dob]);
    if (age && empty_(row[col.age - 1])) upd[col.age] = age;
  }
  if (parsed.nationality && empty_(row[col.nationality - 1]))
    upd[col.nationality] = normalizeNationality_(parsed.nationality);
  if (parsed.education   && empty_(row[col.education - 1]))
    upd[col.education]   = parsed.education;
  if (parsed.mobile      && empty_(row[col.mobile - 1]))
    upd[col.mobile]      = normalizePhone_(parsed.mobile);
  if (parsed.trade       && empty_(row[col.trade - 1]))
    upd[col.trade]       = parsed.trade;

  // ── NEW: Extended fields from v289 expanded parser ─────────────────
  if (parsed.employment_status && empty_(row[ec.empStatus - 1]))
    upd[ec.empStatus]       = parsed.employment_status;
  if (parsed.passport_type && empty_(row[ec.ecrStatus - 1]))
    upd[ec.ecrStatus]       = parsed.passport_type;
  if (parsed.current_location && empty_(row[ec.currentLocation - 1]))
    upd[ec.currentLocation] = parsed.current_location;
  if (parsed.notice_period_days) {
    var nd = parseInt(parsed.notice_period_days) || 0;
    if (nd > 0 && empty_(row[ec.noticeDays - 1])) upd[ec.noticeDays] = nd;
  }

  notesObj.followUp.status     = 'received';
  notesObj.followUp.receivedAt = new Date().toISOString();
  upd[col.notes]           = JSON.stringify(notesObj);
  upd[col.recruiterAction] = 'Candidate replied — fields updated. Verify and re-screen.';

  Object.keys(upd).forEach(function(c) {
    sheet.getRange(rowNum, parseInt(c, 10)).setValue(upd[c]);
  });
}

// S21.F01 — send HTML summary to owner at 20:00
function sendDailyDigest() {
  try {
    var ss     = getMasterSS_();
    var sheet  = ss.getSheetByName(CONFIG.sheetName);
    var rSheet = ss.getSheetByName(CONFIG.rejectedSheetName);
    var tz     = 'Asia/Kolkata';
    var today  = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd');
    var ic     = CONFIG.inputColumns;

    var total=0, shortlisted=0, needsCall=0, needsReview=0, rejected=0;
    var trades = {};

    if (sheet && sheet.getLastRow() > 1) {
      sheet.getRange(2, 1, sheet.getLastRow()-1, CONFIG.headers.length)
           .getValues().forEach(function(row) {
        var ts  = row[ic.applicationDate - 1];
        var ds  = ts instanceof Date ? Utilities.formatDate(ts, tz, 'yyyy-MM-dd') : String(ts).slice(0,10);
        if (ds !== today) return;
        total++;
        var v = String(row[ic.verdict - 1] || '');
        if (v==='SHORTLISTED')  shortlisted++;
        else if (v==='NEEDS_CALL')  needsCall++;
        else if (v==='NEEDS_REVIEW') needsReview++;
        var t = String(row[ic.trade - 1] || '');
        if (t) trades[t] = (trades[t]||0) + 1;
      });
    }
    if (rSheet && rSheet.getLastRow() > 1) {
      rSheet.getRange(2, 1, rSheet.getLastRow()-1, 2).getValues().forEach(function(r) {
        var ts = r[0];
        var ds = ts instanceof Date ? Utilities.formatDate(ts, tz, 'yyyy-MM-dd') : String(ts).slice(0,10);
        if (ds === today) rejected++;
      });
    }

    var qs = getQuotaState_();
    var topTrades = Object.keys(trades)
      .sort(function(a,b){ return trades[b]-trades[a]; })
      .slice(0,5).map(function(t){ return t+'('+trades[t]+')'; }).join(', ');

    var html = digestHtml_(today, total, shortlisted, needsCall, needsReview, rejected, topTrades, qs);
    MailApp.sendEmail({
      to: getOwnerEmail_(),
      subject: 'KAI Digest — ' + today + ' | ' + total + ' CVs | ' + shortlisted + ' shortlisted',
      htmlBody: html
    });
    appendLog_({ status: 'DIGEST_SENT', notes: 'total='+total+' shortlisted='+shortlisted });
  } catch (e) {
    appendLog_({ status: 'DIGEST_FAILED', notes: e.message });
  }
}

// S22.F02 — deduplicate and archive stale rows
function cleanupCandidates() {
  if (isPipelinePaused_()) return;
  var lock = LockService.getScriptLock();
  if (!lock.tryLock(30000)) { appendLog_({ status: 'CLEANUP_LOCK_SKIP' }); return; }
  try {
    var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
    if (!sheet) return;
    var stats = cleanupSheet_(sheet);
    appendLog_({ status: 'CLEANUP_DONE',
      notes: 'scanned='+stats.scanned+' dupes='+stats.dupes+' empty='+stats.empty+' archived='+stats.archived });
    try {
      SpreadsheetApp.getUi().alert(
        'Cleanup Complete\n\nScanned: '+stats.scanned+'\nDuplicates: '+stats.dupes+
        '\nEmpty rows: '+stats.empty+'\nArchived: '+stats.archived
      );
    } catch (e) {}
  } finally { try { lock.releaseLock(); } catch (e) {} }
}

// S22.F03 — per-sheet cleanup
function cleanupSheet_(sheet) {
  var stats   = { scanned: 0, dupes: 0, empty: 0, archived: 0 };
  var lr      = sheet.getLastRow();
  if (lr < 2) return stats;
  var col     = CONFIG.inputColumns;
  var data    = sheet.getRange(2, 1, lr-1, CONFIG.headers.length).getValues();
  stats.scanned = data.length;
  var cutoff  = new Date(Date.now() - CONFIG.archiveAfterDays * 86400000);

  // Pass 1 — mark empty rows SUPERSEDED (handles blank _Active legacy rows too)
  for (var i = 0; i < data.length; i++) {
    var _avP1 = String(data[i][col.active-1] || '').toUpperCase().trim();
    if (_avP1 === 'SUPERSEDED' || _avP1 === 'ARCHIVED') continue;
    var nm = String(data[i][col.name-1]||'').trim();
    var em = String(data[i][col.email-1]||'').trim();
    var mo = String(data[i][col.mobile-1]||'').replace(/^'/,'').trim();
    if (!nm && !em && !mo) {
      sheet.getRange(i+2, col.active).setValue('SUPERSEDED');
      stats.empty++;
      data[i][col.active-1] = 'SUPERSEDED';
    }
  }

  // Pass 2 — archive SUPERSEDED rows older than archiveAfterDays
  for (var j = 0; j < data.length; j++) {
    if (String(data[j][col.active-1]) !== 'SUPERSEDED') continue;
    var dt = data[j][col.applicationDate-1];
    if (dt instanceof Date && dt < cutoff) {
      writeToArchive_(data[j], 'superseded+archived');
      sheet.getRange(j+2, col.active).setValue('ARCHIVED');
      stats.archived++;
    }
  }

  // Pass 3 — group active rows by email/mobile for dedup (handles blank _Active legacy rows)
  var groups  = {};
  for (var k = 0; k < data.length; k++) {
    var _avP3 = String(data[k][col.active-1] || '').toUpperCase().trim();
    if (_avP3 === 'SUPERSEDED' || _avP3 === 'ARCHIVED') continue;
    var eK = normalizeEmailKey_(data[k][col.email-1]);
    var mK = normalizeMobileKey_(data[k][col.mobile-1]);
    var rn = k + 2;
    var appDt = data[k][col.applicationDate-1];
    var entry = { rowNum: rn, dt: appDt instanceof Date ? appDt : new Date(appDt) };
    if (eK) { groups['e:'+eK] = groups['e:'+eK]||[]; groups['e:'+eK].push(entry); }
    if (mK) { groups['m:'+mK] = groups['m:'+mK]||[]; groups['m:'+mK].push(entry); }
  }

  var decided = {};
  Object.keys(groups).forEach(function(gk) {
    var grp = groups[gk];
    if (grp.length < 2) return;
    grp.sort(function(a,b){ return b.dt.getTime()-a.dt.getTime(); });
    if (!decided[grp[0].rowNum]) decided[grp[0].rowNum] = 'KEEP';
    for (var y = 1; y < grp.length; y++) {
      if (!decided[grp[y].rowNum]) decided[grp[y].rowNum] = grp[0].rowNum; // kept row ref
    }
  });

  Object.keys(decided).forEach(function(rn) {
    if (decided[rn] === 'KEEP') return;
    var rowInt = parseInt(rn, 10);
    sheet.getRange(rowInt, col.active).setValue('SUPERSEDED');
    var raw  = sheet.getRange(rowInt, col.notes).getValue();
    var nObj; try { nObj = JSON.parse(raw); } catch(e) { nObj = raw ? {freeText:raw} : {}; }
    nObj.superseded = { reason:'duplicate', keptRow: decided[rn], date: new Date().toISOString() };
    sheet.getRange(rowInt, col.notes).setValue(JSON.stringify(nObj));
    stats.dupes++;
  });

  return stats;
}

// S22.F04 — one-time backfill: stamp _Active=TRUE on all legacy rows that have content
// Run this ONCE from menu after pasting new script — fixes dashboard no-data on existing sheets
function backfillActiveColumn() {
  var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
  if (!sheet || sheet.getLastRow() < 2) {
    try { SpreadsheetApp.getUi().alert('Candidates sheet empty — nothing to backfill.'); } catch(e){}
    return;
  }
  var col   = CONFIG.inputColumns;
  var lr    = sheet.getLastRow();
  var range = sheet.getRange(2, 1, lr - 1, CONFIG.headers.length);
  var data  = range.getValues();
  var count = 0;
  for (var i = 0; i < data.length; i++) {
    var av = String(data[i][col.active - 1] || '').toUpperCase().trim();
    var nm = String(data[i][col.name  - 1] || '').trim();
    var em = String(data[i][col.email - 1] || '').trim();
    // Only set TRUE on rows that (a) have candidate data AND (b) have no _Active value yet
    if (!av && (nm || em)) {
      data[i][col.active - 1] = 'TRUE';
      count++;
    }
  }
  range.setValues(data);
  CacheService.getScriptCache().remove('kai_dashboard_v1');
  var msg = '✓ Backfill complete.\n\n' + count + ' legacy rows stamped _Active=TRUE.\n\nRefresh your dashboard — all candidates will now appear.';
  try { SpreadsheetApp.getUi().alert(msg); } catch(e){ Logger.log(msg); }
  Logger.log('S22.F04: backfillActiveColumn done — ' + count + ' rows updated.');
}

// S25.F01 — archive rows > 30 days to Drive CSV
function rotateLogs_() {
  var ss      = getMasterSS_();
  var cutoff  = new Date(Date.now() - 30 * 86400000);
  var targets = [
    { name: CONFIG.logSheetName,   headers: CONFIG.logHeaders },
    { name: CONFIG.errorSheetName, headers: CONFIG.errorHeaders }
  ];
  var folder = ensureLogArchiveFolder_();
  targets.forEach(function(sc) {
    var sheet = ss.getSheetByName(sc.name);
    if (!sheet || sheet.getLastRow() < 2) return;
    var data = sheet.getRange(2, 1, sheet.getLastRow()-1, sc.headers.length).getValues();
    var keep = [], archive = [];
    data.forEach(function(row) {
      if (row[0] instanceof Date && row[0] < cutoff) archive.push(row);
      else keep.push(row);
    });
    if (!archive.length) return;
    var csv = [sc.headers.join(',')].concat(archive.map(function(row) {
      return row.map(function(v){ return '"'+String(v||'').replace(/"/g,'""')+'"'; }).join(',');
    })).join('\n');
    folder.createFile(
      sc.name+'_'+Utilities.formatDate(new Date(),'Asia/Kolkata','yyyyMMdd_HHmmss')+'.csv',
      csv, 'text/csv'
    );
    sheet.getRange(2, 1, data.length, sc.headers.length).clearContent();
    if (keep.length > 0) sheet.getRange(2, 1, keep.length, sc.headers.length).setValues(keep);
  });
}

// S27.F02 — called from dashboard via google.script.run
function getDashboardData() {
  try {
    var cache  = CacheService.getScriptCache();
    var cached = cache.get('kai_dashboard_v1');
    if (cached) {
      try {
        var p = JSON.parse(cached);
        if (p && p.records && p.records.length > 0) return p;
      } catch(e) {}
    }

    var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
    if (!sheet) return { metrics:{}, records:[], version:CONFIG.version };
    var lr = sheet.getLastRow();
    if (lr < 2) return { metrics:{totalProcessed:0}, records:[], version:CONFIG.version };

    var data = sheet.getRange(2, 1, lr-1, CONFIG.headers.length).getValues();
    var ic   = CONFIG.inputColumns;
    var tz   = 'Asia/Kolkata';
    var now  = new Date();
    var today = Utilities.formatDate(now, tz, 'yyyy-MM-dd');

    var metrics = { totalProcessed:0, shortlisted:0, needsReview:0, needsCall:0,
                    todayCount:0, last7Count:0, last30Count:0 };
    var trades={}, verdicts={}, nationalities={};

    var records = [];
    data.forEach(function(row, idx) {
      // FIX S27.F02 — accept blank _Active (legacy rows) + boolean true + string 'TRUE'
      // Only skip rows explicitly marked SUPERSEDED or ARCHIVED
      var _av = String(row[ic.active-1] || '').toUpperCase().trim();
      if (_av === 'SUPERSEDED' || _av === 'ARCHIVED') return;
      // Skip completely empty rows (no name AND no email)
      if (!String(row[ic.name-1]||'').trim() && !String(row[ic.email-1]||'').trim()) return;
      metrics.totalProcessed++;
      var v   = String(row[ic.verdict-1]||'');
      var t   = String(row[ic.trade-1]||'');
      var nat = String(row[ic.nationality-1]||'');

      if (v==='SHORTLISTED')  metrics.shortlisted++;
      if (v==='NEEDS_REVIEW') metrics.needsReview++;
      if (v==='NEEDS_CALL')   metrics.needsCall++;

      // Robust date handling (Date object OR Excel serial OR string)
      var appDtRaw = row[ic.applicationDate-1];
      var appDt = null;
      if (appDtRaw instanceof Date) appDt = appDtRaw;
      else if (typeof appDtRaw==='number' && appDtRaw>25000 && appDtRaw<80000)
        appDt = new Date(Math.round((appDtRaw-25569)*86400000));
      else if (appDtRaw) { var tmp=new Date(String(appDtRaw)); if(!isNaN(tmp)) appDt=tmp; }
      var appDtStr = appDt ? Utilities.formatDate(appDt, tz, 'yyyy-MM-dd HH:mm') : '';

      if (appDt) {
        if (Utilities.formatDate(appDt,tz,'yyyy-MM-dd')===today) metrics.todayCount++;
        if (appDt >= new Date(now.getTime()-7*86400000)) metrics.last7Count++;
        if (appDt >= new Date(now.getTime()-30*86400000)) metrics.last30Count++;
      }

      if (t) trades[t]=true; if (v) verdicts[v]=true; if (nat) nationalities[nat]=true;

      records.push({
        rowIndex:         idx+2,
        stage:            String(row[ic.stage-1]||'Pending action'),
        applicationDate:  appDtStr,
        nationality:      nat,
        name:             String(row[ic.name-1]||''),
        mobile:           String(row[ic.mobile-1]||'').replace(/^'/,''),
        email:            String(row[ic.email-1]||''),
        education:        String(row[ic.education-1]||''),
        positionApplied:  String(row[ic.positionApplied-1]||''),
        trade:            t,
        industry:         String(row[ic.industry-1]||''),
        experience:       Number(row[ic.experience-1])||0,
        gulfExperience:   String(row[ic.gulf-1]||''),
        dob:              String(row[ic.dob-1]||''),
        age:              String(row[ic.age-1]||''),
        verdict:          v,
        flags:            String(row[ic.flags-1]||''),
        score:            Number(row[ic.score-1])||0,
        scoreBreakdown:   String(row[ic.scoreBreakdown-1]||''),
        recommendedRoles: String(row[ic.recommendedRoles-1]||''),
        kaiAssessment:    String(row[ic.kaiAssessment-1]||''),
        recruiterAction:  String(row[ic.recruiterAction-1]||''),
        cvLink:           String(row[ic.cvLink-1]||''),
        notes:            String(row[ic.notes-1]||''),
        whatsapp:         buildWhatsAppLink_(row[ic.mobile-1], row[ic.name-1], row[ic.trade-1])
      });
    });

    records.sort(function(a,b){ return String(b.applicationDate).localeCompare(String(a.applicationDate)); });
    var result = {
      metrics:       metrics,
      records:       records,
      trades:        Object.keys(trades).sort(),
      verdicts:      Object.keys(verdicts).sort(),
      nationalities: Object.keys(nationalities).sort(),
      quota:         getQuotaState_(),
      version:       CONFIG.version
    };
    try { cache.put('kai_dashboard_v1', JSON.stringify(result), 90); } catch(e) {}
    return result;
  } catch (err) {
    Logger.log('S27: getDashboardData error: ' + err.message);
    return { metrics:{}, records:[], error: err.message, version: CONFIG.version };
  }
}

// S28.F03 — extract actual candidate sender from forwarded email body
function extractForwardedSender_(body) {
  if (!body) return '';
  var searchIn = body;
  var markers  = ['Forwarded message', 'Begin forwarded', 'Original Message', 'Forwarded Message'];
  var idx = -1;
  markers.forEach(function(mk) {
    var i = body.indexOf(mk);
    if (i >= 0 && (idx < 0 || i < idx)) idx = i;
  });
  if (idx >= 0) searchIn = body.slice(idx, idx + 2000);
  else          searchIn = body.slice(0, 2000);
  var patterns = [
    /From:\s*[^<\n]*<([^\s>]+@[^\s>]+)>/i,
    /From:\s*([^\s<\n]+@[^\s<\n]+)/i,
    /Reply-To:\s*<?([^\s>]+@[^\s>]+)>?/i,
    /E-?mail:\s*<?([^\s>]+@[^\s>]+)>?/i,
    /Contact:\s*<?([^\s>]+@[^\s>]+)>?/i,
    /Sent by:\s*<?([^\s>]+@[^\s>]+)>?/i
  ];
  for (var i = 0; i < patterns.length; i++) {
    var m = searchIn.match(patterns[i]);
    if (m && m[1] && m[1].indexOf('@') >= 0) {
      var cand = m[1].trim().toLowerCase();
      if (!isInternalDomain_(cand)) return cand;
    }
  }
  return '';
}
// S28.F04 — extract original Date from forwarded email body
function extractOriginalApplicationDate_(message) {
  var fallback = message.getDate ? message.getDate() : new Date();
  try {
    var body = message.getPlainBody() || '';
    var idx  = -1;
    ['Forwarded message','Begin forwarded','Original Message'].forEach(function(mk) {
      var i = body.indexOf(mk);
      if (i >= 0 && (idx < 0 || i < idx)) idx = i;
    });
    var region  = idx >= 0 ? body.slice(idx, idx+3000) : body.slice(0,2000);
    var m       = region.match(/^[ \t]*Date:[ \t]+(.+)$/im);
    if (!m) return { date: fallback, fallback: true };
    var raw     = m[1].trim().replace(/\s*\([^)]+\)\s*$/,'');
    var parsed  = parseFlexibleDate_(raw);
    if (!parsed) return { date: fallback, fallback: true };
    var now     = new Date();
    var tenAgo  = new Date(now.getFullYear()-10, 0, 1);
    if (parsed > now || parsed < tenAgo) return { date: fallback, fallback: true };
    return { date: parsed, fallback: false };
  } catch (e) { return { date: fallback, fallback: true }; }
}
// S28.F05
function parseFlexibleDate_(raw) {
  if (!raw) return null;
  var ms = Date.parse(raw);
  if (!isNaN(ms)) return new Date(ms);
  var m = raw.match(/^(\d{1,2})[\/\-\.](\d{1,2})[\/\-\.](\d{4})/);
  if (m) {
    var d=parseInt(m[1],10),mo=parseInt(m[2],10),y=parseInt(m[3],10);
    if (d<=31&&mo<=12) return new Date(y,mo-1,d);
  }
  var MONTHS={jan:0,feb:1,mar:2,apr:3,may:4,jun:5,jul:6,aug:7,sep:8,oct:9,nov:10,dec:11};
  var m2=raw.toLowerCase().match(/(\d{1,2})\s+([a-z]{3,})[a-z]*\s+(\d{4})/);
  if (m2 && MONTHS.hasOwnProperty(m2[2].slice(0,3)))
    return new Date(parseInt(m2[3],10), MONTHS[m2[2].slice(0,3)], parseInt(m2[1],10));
  return null;
}

// S30 · LIVE DIAGNOSTICS  (run from editor to debug dashboard)
// ═══════════════════════════════════════════════════════════════════
// S30.F01 — run this from GAS editor to diagnose dashboard no-data
function diagnoseDashboard() {
  var results = [];
  var pass = 0, fail = 0;

  function chk(label, ok, detail) {
    results.push((ok ? '✓' : '✗') + ' ' + label + (detail ? ': ' + detail : ''));
    if (ok) pass++; else fail++;
  }

  // 1. Spreadsheet access
  var ssId = getSpreadsheetId_();
  chk('SpreadsheetID', !!ssId, ssId);
  var ss = null;
  try { ss = SpreadsheetApp.openById(ssId); chk('Sheet open OK', true, ss.getName()); }
  catch(e) { chk('Sheet open', false, e.message); }

  // 2. Candidates sheet
  var sheet = ss ? ss.getSheetByName(CONFIG.sheetName) : null;
  chk('Candidates sheet exists', !!sheet, CONFIG.sheetName);
  var lr = sheet ? sheet.getLastRow() : 0;
  chk('Candidates has data rows', lr > 1, lr + ' rows total');

  // 3. _Active column check
  if (sheet && lr > 1) {
    var sample = sheet.getRange(2, CONFIG.inputColumns.active, Math.min(lr-1, 20), 1).getValues();
    var trueCount=0, blankCount=0, otherCount=0;
    sample.forEach(function(r) {
      var v = String(r[0]||'').toUpperCase().trim();
      if (v==='TRUE') trueCount++;
      else if (!v)    blankCount++;
      else            otherCount++;
    });
    chk('_Active col (first 20 rows)', true,
      'TRUE='+trueCount+' BLANK='+blankCount+' OTHER='+otherCount);
    if (blankCount > 0) results.push('  → Run backfillActiveColumn() from menu to fix blanks');
  }

  // 4. getDashboardData returns records
  try {
    CacheService.getScriptCache().remove('kai_dashboard_v1');
    var d = getDashboardData();
    chk('getDashboardData() records', d.records && d.records.length > 0,
      (d.records ? d.records.length : 0) + ' records returned');
    if (d.error) results.push('  → Error in getDashboardData: ' + d.error);
  } catch(e) { chk('getDashboardData()', false, e.message); }

  // 5. Script version
  chk('Script version', true, 'KAI v' + CONFIG.version);

  // 6. Cache cleared
  CacheService.getScriptCache().remove('kai_dashboard_v1');
  chk('Cache cleared', true, 'kai_dashboard_v1 removed');

  var report = '═══ KAI DASHBOARD DIAGNOSTICS ═══\n\n' +
    results.join('\n') + '\n\n' +
    '─────────────────────────\n' +
    'PASSED: ' + pass + '  FAILED: ' + fail + '\n\n' +
    (fail > 0
      ? '⚠  ACTION NEEDED — see ✗ items above.\n'
      : '✓  All checks passed — redeploy the Web App then refresh.\n') +
    '\nIf records=0 after fix: run Karigar AI → 🔧 Fix Dashboard (Backfill)';

  Logger.log(report);
  try { SpreadsheetApp.getUi().alert(report); } catch(e) {}
  return report;
}

// ───────────────────────────────────────────────────────────────────
// S31.F02 — formatWorkbook()  ← menu entry point
// Iterates all sheets, routes to correct formatter, flushes once.
// ───────────────────────────────────────────────────────────────────
function formatWorkbook() {
  var ss;
  try { ss = getMasterSS_(); }
  catch(e) { Logger.log('S31: Cannot open spreadsheet: ' + e.message); return; }

  var startMs  = Date.now();
  var map      = getFmtSheetMap_();
  var sheets   = ss.getSheets();
  var done = 0, skipped = 0;

  sheets.forEach(function(sh) {
    // Bail out if approaching 5-minute Apps Script limit
    if (Date.now() - startMs > 280000) {
      Logger.log('S31: Time limit approaching — stopping at sheet ' + sh.getName());
      skipped++;
      return;
    }
    var type = map[sh.getName()] || 'compact';
    try {
      formatSheet(sh, type);
      done++;
    } catch(e) {
      Logger.log('S31: WARNING — skipped "' + sh.getName() + '": ' + e.message);
      skipped++;
    }
  });

  SpreadsheetApp.flush();
  var elapsed = Math.round((Date.now() - startMs) / 1000);
  Logger.log('S31: formatWorkbook done — ' + done + ' formatted, ' + skipped + ' skipped, ' + elapsed + 's');
  try {
    SpreadsheetApp.getUi().alert(
      '✓ Workbook formatted.\n' + done + ' sheets styled in ' + elapsed + 's.'
    );
  } catch(e) {}
}

// ───────────────────────────────────────────────────────────────────
// S31.F03 — formatSheet(sheet, type)   ← public dispatcher
// type values: 'candidates' | 'rejected' | 'logs' | 'errors' |
//              'standard' | 'compact'
// ───────────────────────────────────────────────────────────────────
function formatSheet(sheet, type) {
  if (!sheet) return;
  if (!type) {
    var map = getFmtSheetMap_();
    type = map[sheet.getName()] || 'compact';
  }
  switch (type) {
    case 'candidates': formatCandidatesSheet_(sheet); break;
    case 'rejected':   formatRejectedSheet_(sheet);   break;
    case 'logs':       formatLogsSheet_(sheet);       break;
    case 'errors':     formatErrorsSheet_(sheet);     break;
    case 'standard':   formatStandardSheet_(sheet, false); break;
    case 'compact':
    default:           formatStandardSheet_(sheet, true);  break;
  }
}

// ───────────────────────────────────────────────────────────────────
// S31.F04 — formatCandidatesSheet_()
// Full professional styling for the primary 24-column Candidates sheet.
// ───────────────────────────────────────────────────────────────────
function formatCandidatesSheet_(sheet) {
  var lr  = sheet.getLastRow();
  var lc  = CONFIG.headers.length;   // 24
  var ic  = CONFIG.inputColumns;
  var dr  = Math.max(0, lr - 1);     // data rows

  // 1. Header
  applyHeaderStyle(sheet, lc, FMT_THEME.hdrBg, FMT_THEME.hdrFg);

  // 2. Freeze + heights
  sheet.setFrozenRows(1);
  sheet.setFrozenColumns(1);
  sheet.setRowHeight(1, FMT_THEME.hdrHeight);
  if (dr > 0) sheet.setRowHeightsForced(2, dr, FMT_THEME.dataHeight);

  // 3. Font for entire sheet
  standardizeFonts(sheet);

  // 4. Column widths (exact pixel map)
  FMT_THEME.candidateColW.forEach(function(w, i) {
    sheet.setColumnWidth(i + 1, w);
  });

  // 5. Alignment
  normalizeAlignment(sheet);
  if (dr > 0) {
    // Center numeric/status columns
    [ic.age, ic.experience, ic.score, ic.applicationDate, ic.active]
      .forEach(function(c) {
        sheet.getRange(2, c, dr, 1).setHorizontalAlignment('center');
      });
    // Wrap only KAI Assessment + Notes
    sheet.getRange(1, 1, lr, lc).setWrap(false);
    sheet.getRange(2, ic.kaiAssessment, dr, 1).setWrap(true);
    sheet.getRange(2, ic.notes,         dr, 1).setWrap(true);
    // Name column bold
    sheet.getRange(2, ic.name, dr, 1).setFontWeight('bold');
    // Score column bold + centered
    sheet.getRange(2, ic.score, dr, 1).setFontWeight('bold').setHorizontalAlignment('center');
  }

  // 6. Borders
  applyBorders(sheet);

  // 7. Conditional formatting (alternating + verdict + score + flags)
  applyAlternatingRows(sheet);   // sets base alternating rule
  fmtCandidateConditional_(sheet, lc, lr);
}

// ───────────────────────────────────────────────────────────────────
// S31.F05 — fmtCandidateConditional_()
// Verdict / Score / FLAGS / _Active conditional format rules.
// Collected into one array → applied in single setConditionalFormatRules call.
// ───────────────────────────────────────────────────────────────────
function fmtCandidateConditional_(sheet, lc, lr) {
  if (lr < 2) return;
  var dr  = Math.min(lr - 1, 1000);
  var ic  = CONFIG.inputColumns;
  var rules = [];

  // ── Alternating rows (base — lowest priority, added first) ───
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=MOD(ROW(),2)=0')
    .setBackground(FMT_THEME.rowOdd)
    .setRanges([sheet.getRange(2, 1, dr, lc)])
    .build());

  // ── Verdict (col 15) ─────────────────────────────────────────
  var vR = sheet.getRange(2, ic.verdict, dr, 1);
  [
    ['SHORTLISTED',  FMT_THEME.vGreenBg,  FMT_THEME.vGreenFg ],
    ['NEEDS_REVIEW', FMT_THEME.vYellowBg, FMT_THEME.vYellowFg],
    ['NEEDS_CALL',   FMT_THEME.vOrangeBg, FMT_THEME.vOrangeFg],
    ['REJECT',       FMT_THEME.vRedBg,    FMT_THEME.vRedFg   ]
  ].forEach(function(v) {
    rules.push(SpreadsheetApp.newConditionalFormatRule()
      .whenTextEqualTo(v[0])
      .setBackground(v[1]).setFontColor(v[2])
      .setRanges([vR]).build());
  });

  // ── FLAGS (col 16) ───────────────────────────────────────────
  var fR = sheet.getRange(2, ic.flags, dr, 1);
  [
    ['GREEN',  FMT_THEME.fGreen ],
    ['YELLOW', FMT_THEME.fYellow],
    ['ORANGE', FMT_THEME.fOrange],
    ['RED',    FMT_THEME.fRed   ]
  ].forEach(function(f) {
    rules.push(SpreadsheetApp.newConditionalFormatRule()
      .whenTextContains(f[0])
      .setBackground(f[1])
      .setRanges([fR]).build());
  });

  // ── Score (col 17) ───────────────────────────────────────────
  var sR = sheet.getRange(2, ic.score, dr, 1);
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenNumberGreaterThanOrEqualTo(70)
    .setBackground(FMT_THEME.sHigh).setRanges([sR]).build());
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenNumberBetween(40, 69)
    .setBackground(FMT_THEME.sMid).setRanges([sR]).build());
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenNumberLessThan(40)
    .setBackground(FMT_THEME.sLow).setRanges([sR]).build());

  // ── _Active (col 24) — grey-out superseded/archived ──────────
  var aR = sheet.getRange(2, ic.active, dr, 1);
  ['SUPERSEDED','ARCHIVED'].forEach(function(s) {
    rules.push(SpreadsheetApp.newConditionalFormatRule()
      .whenTextContains(s)
      .setBackground('#f1f5f9').setFontColor('#94a3b8')
      .setRanges([aR]).build());
  });

  sheet.setConditionalFormatRules(rules);
}

function applyHeaderStyle(sheet, numCols, bgColor, fgColor) {
  if (!sheet || !numCols) return;

  hdr.setBackground(bgColor   || FMT_THEME.hdrBg)
     .setFontColor(fgColor    || FMT_THEME.hdrFg)
     .setFontFamily(FMT_THEME.dataFont)

     .setVerticalAlignment('middle')

// ───────────────────────────────────────────────────────────────────
// S31.F11 — autoResizeColumns(sheet)
// Resize all columns, then enforce min 80 / max 300 caps.
// ───────────────────────────────────────────────────────────────────
function autoResizeColumns(sheet) {
  if (!sheet) return;
  var lc = sheet.getLastColumn();
  if (lc < 1) return;
  try {
    sheet.autoResizeColumns(1, lc);
    for (var c = 1; c <= lc; c++) {
      var w = sheet.getColumnWidth(c);
      if (w < 80)  sheet.setColumnWidth(c, 80);
      if (w > 300) sheet.setColumnWidth(c, 300);
    }
  } catch(e) { Logger.log('S31: autoResizeColumns warning: ' + e.message); }
}

// ───────────────────────────────────────────────────────────────────
// S31.F12 — cleanEmptyRows(sheet)
// Removes trailing blank rows (checks up to last 50 rows).
// Safe: never removes row 1 (header).
// ───────────────────────────────────────────────────────────────────
function cleanEmptyRows(sheet) {
  if (!sheet) return;
  var lr = sheet.getLastRow();
  var lc = sheet.getLastColumn();
  if (lr < 2 || lc < 1) return;
  var checkFrom = Math.max(2, lr - 49);
  var numCheck  = lr - checkFrom + 1;
  var data = sheet.getRange(checkFrom, 1, numCheck, lc).getValues();
  var delCount = 0;
  for (var i = data.length - 1; i >= 0; i--) {
    var empty = data[i].every(function(c) { return String(c).trim() === ''; });
    if (empty) delCount++;
    else break;
  }
  if (delCount > 0) {
    sheet.deleteRows(lr - delCount + 1, delCount);
    Logger.log('S31: cleanEmptyRows removed ' + delCount + ' row(s) from ' + sheet.getName());
  }
}

// ───────────────────────────────────────────────────────────────────
// S31.F13 — applyBorders(sheet)
// Outer + inner borders on header + all data rows (capped at 500 rows
// to stay within quota limits on large sheets).
// ───────────────────────────────────────────────────────────────────
function applyBorders(sheet) {
  if (!sheet) return;
  var lr = Math.min(sheet.getLastRow(), 500);
  var lc = sheet.getLastColumn();
  if (lr < 1 || lc < 1) return;
  try {
    sheet.getRange(1, 1, lr, lc).setBorder(
      true, true, true, true, true, true,
      FMT_THEME.borderColor,
      SpreadsheetApp.BorderStyle.SOLID
    );
  } catch(e) { Logger.log('S31: applyBorders warning: ' + e.message); }
}

// ───────────────────────────────────────────────────────────────────
// S31.F14 — applyAlternatingRows(sheet)
// Applies MOD(ROW(),2)=0 conditional rule for zebra striping.
// Merges with any existing rules on the sheet.
// ───────────────────────────────────────────────────────────────────
function applyAlternatingRows(sheet) {
  if (!sheet) return;
  var lr = sheet.getLastRow();
  var lc = sheet.getLastColumn();
  if (lr < 2 || lc < 1) return;
  var dr = Math.min(lr - 1, 1000);
  // Preserve existing rules, then add/replace alternating rule at index 0
  var existing = sheet.getConditionalFormatRules().filter(function(r) {
    // Remove any existing MOD(ROW() rules to avoid duplicates
    try {
      var spec = r.getBooleanCondition();
      return !(spec && String(spec.getCriteriaValues()[0]).indexOf('MOD(ROW()') >= 0);
    } catch(e) { return true; }
  });
  var altRule = SpreadsheetApp.newConditionalFormatRule()
    .whenFormulaSatisfied('=MOD(ROW(),2)=0')
    .setBackground(FMT_THEME.rowOdd)
    .setRanges([sheet.getRange(2, 1, dr, lc)])
    .build();
  sheet.setConditionalFormatRules([altRule].concat(existing));
}

// ───────────────────────────────────────────────────────────────────
// S31.F15 — standardizeFonts(sheet)
// Applies uniform font family + size to the entire sheet.
// Header row keeps bold weight; data rows normalised.
// ───────────────────────────────────────────────────────────────────
function standardizeFonts(sheet) {
  if (!sheet) return;
  var lr = sheet.getLastRow();
  var lc = sheet.getLastColumn();
  if (lr < 1 || lc < 1) return;
  // Entire sheet: uniform font family
  sheet.getRange(1, 1, lr, lc)
    .setFontFamily(FMT_THEME.dataFont)
    .setFontSize(FMT_THEME.dataSize);
  // Header row: bold, slightly larger
  sheet.getRange(1, 1, 1, lc)
    .setFontSize(FMT_THEME.hdrSize)
    .setFontWeight('bold');
  // Data rows: normal weight
  if (lr > 1) {
    sheet.getRange(2, 1, lr - 1, lc).setFontWeight('normal');
  }
}

// ───────────────────────────────────────────────────────────────────
// S31.F16 — normalizeAlignment(sheet)
// Header: centre. Data rows: left + middle vertical.
// ───────────────────────────────────────────────────────────────────
function normalizeAlignment(sheet) {
  if (!sheet) return;
  var lr = sheet.getLastRow();
  var lc = sheet.getLastColumn();
  if (lr < 1 || lc < 1) return;
  sheet.getRange(1, 1, 1, lc)
    .setHorizontalAlignment('center')
    .setVerticalAlignment('middle');
  if (lr > 1) {
    sheet.getRange(2, 1, lr - 1, lc)
      .setHorizontalAlignment('left')
      .setVerticalAlignment('middle');
  }
}
