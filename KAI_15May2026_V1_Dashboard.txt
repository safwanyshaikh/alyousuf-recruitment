<script>
var ALL = [];
var STAGES = ['Pending action','Screened','Shortlisted','Client Sent','Selected','Offer','Visa Processing','Deployed'];

function loadData() {
  document.getElementById('btnRefresh').disabled = true;
  document.getElementById('ts').textContent = 'Refreshing…';
  document.getElementById('tCon').innerHTML = '<div class="loading"><div class="spin"></div><p>Loading…</p></div>';
  hidErr();
  google.script.run
    .withSuccessHandler(gotData)
    .withFailureHandler(gotErr)
    .getDashboardData();
}

function gotData(d) {
  document.getElementById('btnRefresh').disabled = false;
  document.getElementById('ts').textContent = 'Updated ' + new Date().toLocaleTimeString();

  if (!d) { gotErr('Server returned nothing'); return; }
  if (d.error) { showErr('Server error: ' + d.error); }

  // Metrics
  var m = d.metrics || {};
  set('mTotal',  m.totalProcessed || 0);
  set('mShort',  m.shortlisted    || 0);
  set('mReview', m.needsReview    || 0);
  set('mCall',   m.needsCall      || 0);
  set('mRej',    Math.max(0,(m.totalProcessed||0)-(m.shortlisted||0)-(m.needsReview||0)-(m.needsCall||0)));
  set('mToday',  m.todayCount     || 0);

  // Quota
  if (d.quota) {
    var q = d.quota;
    document.getElementById('quotaBar').style.display = 'flex';
    setQ('qCvT','qCvF','CVs',    q.cvTotalUsed||0, 200);
    setQ('qEmT','qEmF','Emails', q.emailUsed||0,    90);
    setQ('qReT','qReF','Replies',q.replyUsed||0,    50);
  }

  // Dropdowns
  fillSel('fT', d.trades || []);
  fillSel('fN', d.nationalities || []);

  ALL = d.records || [];
  applyFilters();
}

function gotErr(e) {
  document.getElementById('btnRefresh').disabled = false;
  document.getElementById('ts').textContent = 'Error';
  showErr('Failed to load: ' + e);
  document.getElementById('tCon').innerHTML = '<div class="empty">Could not load data. <button onclick="loadData()" style="color:#0071e3;background:none;border:none;cursor:pointer;font-size:13px">Retry</button></div>';
}

function showErr(msg) {
  var b = document.getElementById('errBanner');
  b.textContent = '⚠ ' + msg;
  b.style.display = 'block';
}
function hidErr() { document.getElementById('errBanner').style.display = 'none'; }

function applyFilters() {
  var q  = (document.getElementById('fQ').value || '').toLowerCase().trim();
  var fv = document.getElementById('fV').value;
  var ff = document.getElementById('fF').value;
  var ft = document.getElementById('fT').value;
  var fn = document.getElementById('fN').value;
  var out = ALL.filter(function(r) {
    if (fv && r.verdict      !== fv) return false;
    if (ff && r.flags        !== ff) return false;
    if (ft && r.trade        !== ft) return false;
    if (fn && r.nationality  !== fn) return false;
    if (q) {
      var h = [r.name,r.email,r.mobile,r.trade,r.nationality,r.positionApplied].join(' ').toLowerCase();
      if (h.indexOf(q) < 0) return false;
    }
    return true;
  });
  document.getElementById('fCnt').textContent = out.length + ' of ' + ALL.length + ' candidates';
  render(out);
}

function clearF() {
  ['fQ','fV','fF','fT','fN'].forEach(function(id) {
    var el = document.getElementById(id);
    el.value = el.tagName === 'INPUT' ? '' : '';
  });
  applyFilters();
}

function render(rows) {
  if (!rows.length) {
    document.getElementById('tCon').innerHTML = '<div class="empty">No candidates match the current filters.</div>';
    return;
  }
  var h = '<div class="tscroll"><table><thead><tr>' +
    '<th>Stage</th><th>Date</th><th>Name</th><th>Nationality</th>' +
    '<th>Trade</th><th>Exp</th><th>Age</th><th>Score</th>' +
    '<th>Flag</th><th>Verdict</th><th>Mobile</th><th>Action</th><th>KAI</th><th>CV</th>' +
    '</tr></thead><tbody>';

  rows.forEach(function(r) {
    var sc = r.score || 0;
    var scCol = sc>=75?'#059669':sc>=60?'#d97706':sc>=45?'#ea580c':'#6b7280';
    var fb = badge(r.flags, r.flags);
    var vb = badge(r.flags, r.verdict || r.flags);
    var wa = r.whatsapp ? '<a href="'+esc(r.whatsapp)+'" target="_blank" title="WhatsApp">📱</a>' : (r.mobile ? esc(r.mobile) : '—');
    var cv = r.cvLink  ? '<a href="'+esc(r.cvLink)+'" target="_blank">📄</a>' : '—';
    var kai = r.kaiAssessment
      ? '<button class="kbtn" onclick="toggleK(this)">▾ KAI</button><div class="kpanel">'+esc(r.kaiAssessment)+'</div>'
      : '—';
    var act = r.recruiterAction
      ? '<span title="'+esc(r.recruiterAction)+'" style="font-size:11px;color:#6b7280;cursor:help">'+esc(r.recruiterAction.slice(0,35))+(r.recruiterAction.length>35?'…':'')+'</span>'
      : '—';
    h += '<tr>' +
      '<td><select class="stsel" data-row="'+r.rowIndex+'" onchange="saveStage(this)">'+stOpts(r.stage)+'</select></td>' +
      '<td class="sm">'+(r.applicationDate?r.applicationDate.slice(0,10):'—')+'</td>' +
      '<td class="nm" title="'+esc(r.name)+'">'+esc(r.name||'—')+'</td>' +
      '<td class="sm">'+esc(r.nationality||'—')+'</td>' +
      '<td>'+esc(r.trade||'—')+'</td>' +
      '<td class="sm">'+(r.experience||'—')+'yr</td>' +
      '<td class="sm">'+(r.age||'—')+'</td>' +
      '<td class="sc" style="color:'+scCol+'">'+sc+'</td>' +
      '<td>'+fb+'</td>' +
      '<td>'+vb+'</td>' +
      '<td>'+wa+'</td>' +
      '<td style="max-width:160px">'+act+'</td>' +
      '<td>'+kai+'</td>' +
      '<td>'+cv+'</td>' +
      '</tr>';
  });
  h += '</tbody></table></div>';
  document.getElementById('tCon').innerHTML = h;
}

function saveStage(sel) {
  var ri = parseInt(sel.getAttribute('data-row'), 10);
  var ns = sel.value;
  sel.style.opacity = '.5';
  sel.disabled = true;
  google.script.run
    .withSuccessHandler(function(r) {
      sel.style.opacity = '1'; sel.disabled = false;
      if (!r || !r.ok) { alert('Stage not saved: ' + (r?r.msg:'error')); loadData(); }
    })
    .withFailureHandler(function(e) {
      sel.style.opacity='1'; sel.disabled=false;
      alert('Error: '+e); loadData();
    })
    .updateCandidateStage(ri, ns);
}

function toggleK(btn) {
  var p = btn.nextElementSibling;
  var o = p.classList.toggle('open');
  btn.textContent = (o ? '▴ ' : '▾ ') + 'KAI';
}

// helpers
function set(id, v) { var e=document.getElementById(id); if(e) e.textContent=v; }
function setQ(tid,fid,lbl,used,cap) {
  document.getElementById(tid).textContent = lbl+' '+used+'/'+cap;
  document.getElementById(fid).style.width = Math.min(100,Math.round(used/cap*100))+'%';
}
function fillSel(id, arr) {
  var s = document.getElementById(id);
  var cv = s.value;
  while(s.options.length>1) s.remove(1);
  arr.forEach(function(v) { if(!v) return; var o=document.createElement('option'); o.value=o.textContent=v; s.appendChild(o); });
  if(cv) s.value = cv;
}
function stOpts(cur) {
  return STAGES.map(function(s){ return '<option value="'+esc(s)+'"'+(s===cur?' selected':'')+'>'+esc(s)+'</option>'; }).join('');
}
function badge(flag, label) {
  var cls = {GREEN:'G',YELLOW:'Y',ORANGE:'O',RED:'R'}[flag] || 'x';
  return '<span class="badge badge-'+cls+'">'+esc(label||flag||'')+'</span>';
}
function esc(s) {
  return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// boot
document.addEventListener('DOMContentLoaded', function() { loadData(); });
</script>
