<script>
/* ─────────────────────────────────────────────────────────────────
   KAI Login Page
   Calls GAS verifyLoginAndIssueToken(email, password)
   → {ok, token, user, role, redirectUrl}
   Stores session token in sessionStorage + cookie for web app auth
───────────────────────────────────────────────────────────────── */

var SESSION_KEY = 'kai_session_token';
var USER_KEY    = 'kai_active_user';

// ── Quick-fill for internal users ──
function fillUser(email) {
  document.getElementById('login-email').value = email;
  document.getElementById('login-pass').focus();
  // Extract hr code for display
  var match = email.match(/^(hr\d+|admin)/i);
  if (match) localStorage.setItem(USER_KEY, match[1].toLowerCase());
}

// ── Main login handler ──
function doLogin() {
  var email = (document.getElementById('login-email').value || '').trim().toLowerCase();
  var pass  = (document.getElementById('login-pass').value  || '').trim();

  clearAlerts();

  if (!email || email.indexOf('@') < 0) {
    showError('Enter a valid email address.'); return;
  }
  if (!pass) {
    showError('Enter your password.'); return;
  }

  // Show spinner
  setBusy(true);

  google.script.run
    .withSuccessHandler(function(res) {
      setBusy(false);
      if (!res) { showError('No response from server. Check GAS deployment.'); return; }
      if (!res.ok) {
        showError(res.msg || 'Invalid email or password.');
        document.getElementById('login-pass').value = '';
        document.getElementById('login-pass').classList.add('error');
        return;
      }

      // Store token
      sessionStorage.setItem(SESSION_KEY, res.token || '');
      if (res.user) localStorage.setItem(USER_KEY, res.user);

      // Show success
      document.getElementById('alert-success').classList.add('visible');
      document.getElementById('btn-login').disabled = true;

      // Redirect to dashboard
      setTimeout(function() {
        var url = res.redirectUrl || window.location.href.replace('/login', '');
        window.location.href = url;
      }, 1200);
    })
    .withFailureHandler(function(e) {
      setBusy(false);
      showError('System error: ' + (e.message || String(e)) +
                '. Ensure the GAS web app is deployed as "Anyone".');
    })
    .verifyLoginAndIssueToken(email, pass);
}

// ── UI helpers ──
function setBusy(busy) {
  var btn     = document.getElementById('btn-login');
  var txtEl   = document.getElementById('btn-login-text');
  var spinner = document.getElementById('login-spinner');
  btn.disabled            = busy;
  txtEl.textContent       = busy ? 'Signing in…' : '🔐 Sign In';
  spinner.style.display   = busy ? 'block' : 'none';
  // Clear input error states
  if (!busy) {
    document.getElementById('login-email').classList.remove('error');
    document.getElementById('login-pass').classList.remove('error');
  }
}

function showError(msg) {
  var el = document.getElementById('alert-error');
  el.textContent = '⚠️ ' + msg;
  el.classList.add('visible');
}

function clearAlerts() {
  document.getElementById('alert-error').classList.remove('visible');
  document.getElementById('alert-success').classList.remove('visible');
}

// ── Auto-focus on load ──
document.addEventListener('DOMContentLoaded', function() {
  // If already logged in (token in sessionStorage), validate and redirect to dashboard
  var token = sessionStorage.getItem(SESSION_KEY);
  if (token) {
    google.script.run
      .withSuccessHandler(function(valid) {
        if (valid) {
          // Must include ?token= in URL so doGet can validate server-side
          var baseUrl = window.location.href.split('?')[0];
          window.location.href = baseUrl + '?token=' + encodeURIComponent(token);
        } else {
          sessionStorage.removeItem(SESSION_KEY);
        }
      })
      .withFailureHandler(function() { /* silent — show login form normally */ })
      .validateSessionToken(token);
  }
  document.getElementById('login-email').focus();
});
</script>
