function loadData() {
  // If we already have data, keep it visible while loading — never go blank
  var hasExisting = ALL_RECORDS && ALL_RECORDS.length > 0;
  if (!hasExisting) showLoading(true);
  document.getElementById('btn-refresh').classList.add('loading');
  google.script.run
    .withSuccessHandler(function(d) { onData(d); })
    .withFailureHandler(function(e) { showError(e); })
    .getDashboardDataV2();
}

function refreshData() {
  showLoading(true);
  document.getElementById('btn-refresh').classList.add('loading');
  google.script.run
    .withSuccessHandler(function() {
      google.script.run
        .withSuccessHandler(onData)
        .withFailureHandler(showError)
        .getDashboardDataV2();
    })
    .withFailureHandler(function() { loadData(); })
    .clearDashboardCacheV2();
}

function onData(d) {
  showLoading(false);
  document.getElementById('btn-refresh').classList.remove('loading');
  // Guard: if server returned error or empty records, keep existing data if any
  if (!d || !d.records || !d.records.length) {
    var reason = (d && d.error) ? d.error : 'No data returned.';
    if (ALL_RECORDS && ALL_RECORDS.length > 0) {
      // Safe: keep current table, just notify recruiter
      showToast('⚠️ Data refresh failed — showing last loaded data. Will retry.');
      // Auto-retry silently after 20 seconds
      setTimeout(loadData, 20000);
    } else {
      showError({ message: reason });
    }
    return;
  }
  ALL_DATA    = d;
  ALL_RECORDS = d.records || [];
  OPEN_REQS   = d.openReqs || [];

  // Enrich records with client-side computed freshness (if backend omitted it)
  ALL_RECORDS.forEach(function(r) {
    if (!r.freshness && r.applicationDate) {
      r.freshness = clientFreshness_(r.applicationDate);
    }
  });

  renderMetrics(d.metrics || {});
  computeDeployReadyMetrics();
  populateFilterDropdowns(d);
  renderRequirementsStrip(OPEN_REQS);
  renderFlightBoard(OPEN_REQS);   // ← sidebar flight board
  PAGE = 1;
  applyFilters();
  var now = new Date();
  document.getElementById('last-updated').textContent =
    'Updated ' + now.getHours().toString().padStart(2,'0') + ':' +
    now.getMinutes().toString().padStart(2,'0');
  var qb = document.getElementById('quota-badge');
  if (d.quota && d.quota.processingPaused) {
    qb.textContent = '⏸ PAUSED'; qb.className = 'hdr-badge red';
  } else {
    qb.textContent = '● LIVE'; qb.className = 'hdr-badge green';
  }
}

// ── DEPLOY READY CLIENT-SIDE METRICS ─────────────────────────────────
function computeDeployReadyMetrics() {
  var yes = 0, possible = 0, hold = 0;
  ALL_RECORDS.forEach(function(r) {
    var dr = (r.deployReady || '').toUpperCase().trim();
    if (dr === 'YES')      yes++;
    else if (dr === 'POSSIBLE') possible++;
    else if (dr === 'HOLD')     hold++;
  });
  var setEl = function(id, v) { var el = document.getElementById(id); if (el) el.textContent = v; };
  setEl('m-drYes',      yes);
  setEl('m-drPossible', possible);
  setEl('m-drHold',     hold);
}

// ── MAIN FILTER LOGIC ─────────────────────────────────────────────────
function applyFilters() {
  var q    = (document.getElementById('search-input').value || '').toLowerCase().trim();
  var fSt  = document.getElementById('flt-state').value;
  var fMob = document.getElementById('flt-mobility').value;
  var fNat = document.getElementById('flt-nationality').value;
  var fTr  = document.getElementById('flt-trade').value;
  var fEdu = document.getElementById('flt-edu').value;
  var fTech= document.getElementById('flt-tech').value;
  var fPP  = document.getElementById('flt-passport') ? document.getElementById('flt-passport').value : '';
  var fFresh = document.getElementById('flt-freshness') ? document.getElementById('flt-freshness').value : '';
  var fDeploy= document.getElementById('flt-deployready') ? document.getElementById('flt-deployready').value : '';
  var fUrg = document.getElementById('flt-urgency').value;

  // Detect if state/mobility columns are empty — warn user
  var extOk = hasExtendedData();
  showExtendedNotice(!extOk && (fSt || fMob ||
    ['clientSent','deployed','deployReady','saudiTransfer','uaeVisit',
     'incompleteContact','unknownTrade','fresherPool','hold'].indexOf(ACTIVE_FILTER) >= 0));

  FILTERED = ALL_RECORDS.filter(function(r) {
    // Normalize fields for comparison — handles mixed-case sheet values
    var verdict = (r.verdict || '').toUpperCase().trim();
    var state   = (r.state   || '').toUpperCase().trim();
    var mob     = (r.mobilityStatus || '').toUpperCase().trim();
    var urg     = (r.nextActionUrgency || '').toUpperCase().trim();

    // Sidebar metric filter
    if (ACTIVE_FILTER !== 'all') {
      switch(ACTIVE_FILTER) {
        case 'needsReview':
          if (verdict !== 'NEEDS_REVIEW' && verdict !== 'NEEDS REVIEW') return false; break;
        case 'needsCall':
          if (verdict !== 'NEEDS_CALL' && verdict !== 'NEEDS CALL') return false; break;
        case 'shortlisted':
          if (verdict !== 'SHORTLISTED') return false; break;
        case 'clientSent':
          if (state !== 'CLIENT_SENT') return false; break;
        case 'deployed':
          if (state !== 'DEPLOYED') return false; break;
        case 'deployReady':
          if (state !== 'READY_TO_DEPLOY' && parseInt(r.deployScore||0) < 80) return false; break;
        case 'saudiTransfer':
          if (mob !== 'SAUDI_TRANSFER') return false; break;
        case 'uaeVisit':
          if (mob !== 'UAE_VISIT') return false; break;
        case 'incompleteContact':
          if (state !== 'INCOMPLETE_CONTACT') return false; break;
        case 'unknownTrade':
          if (state !== 'UNKNOWN_TRADE') return false; break;
        case 'fresherPool':
          if (state !== 'FRESHER_POOL') return false; break;
        case 'hold':
          if (state !== 'HOLD') return false; break;
        case 'drYes':
          if ((r.deployReady||'').toUpperCase() !== 'YES') return false; break;
        case 'drPossible':
          if ((r.deployReady||'').toUpperCase() !== 'POSSIBLE') return false; break;
        case 'drHold':
          if ((r.deployReady||'').toUpperCase() !== 'HOLD') return false; break;
        case 'today': {
          var td = new Date();
          var todayStr = td.getFullYear()+'-'+
            String(td.getMonth()+1).padStart(2,'0')+'-'+String(td.getDate()).padStart(2,'0');
          if (!r.applicationDate || r.applicationDate.slice(0,10) !== todayStr) return false; break;
        }
      }
    }

    // Dropdown filters — case-insensitive on all fields
    if (fSt   && state !== fSt.toUpperCase())  return false;
    if (fMob  && mob   !== fMob.toUpperCase())  return false;
    if (fNat  && (r.nationality||'').toLowerCase() !== fNat.toLowerCase()) return false;
    if (fTr   && (r.trade||'').toLowerCase()       !== fTr.toLowerCase())  return false;
    if (fEdu    && (r.educationEnum||'').toLowerCase()   !== fEdu.toLowerCase())    return false;
    if (fTech   && (r.techReview||'').toLowerCase()      !== fTech.toLowerCase())   return false;
    if (fPP     && (r.passportStatus||'').toLowerCase()  !== fPP.toLowerCase())     return false;
    if (fFresh) {
      if (fFresh === '__today__') {
        // Match only candidates whose CV was received TODAY
        var _todayStr = new Date().toISOString().slice(0,10);
        if (!r.applicationDate || r.applicationDate.slice(0,10) !== _todayStr) return false;
      } else if (fFresh === '__thisweek__') {
        // Match all candidates within the current calendar week (Mon–today)
        if (!isCurrentWeekCV_(r.applicationDate)) return false;
      } else {
        // Match against computed weekday-bucket label exactly
        var rowFresh = clientFreshness_(r.applicationDate) || r.freshness || '';
        if (rowFresh !== fFresh) return false;
      }
    }
    if (fDeploy && (r.deployReady||'').toLowerCase()     !== fDeploy.toLowerCase()) return false;
    if (fUrg    && urg !== fUrg.toUpperCase()) return false;
    // My Candidates filter
    if (!myCandsFilter_(r)) return false;

    // Full-text search — includes KAI assessment, notes, score breakdown
    // Covers: name, trade, location, AI comments, recruiter notes, approvals
    if (q) {
      var blob = [r.name, r.kaiNo, r.nationality, r.trade, r.positionApplied,
                  r.displayLocation, r.currentLocation, r.mobile, r.educationEnum,
                  r.state, r.mobilityStatus, r.verdict, r.flags,
                  (r.top3Positions||[]).join(' '),
                  r.kaiAssessment, r.notes, r.scoreBreakdown,
                  r.qualification, r.experience, r.gulfExp,
                  r.passportStatus, r.recruiterOwner, r.recruiterRemark]
                 .join(' ').toLowerCase();
      if (blob.indexOf(q) < 0) return false;
    }
    return true;
  });

  // Sort — primary: selected column; tiebreaker: rowIndex (sheet row order) for stability.
  // Shortlisted status has NO effect on sort position — candidates stay where their CV date puts them.
  FILTERED.sort(function(a,b) {
    // Freshness column → use numeric day-diff so newest CVs always rank first
    if (SORT_COL === 'freshness') {
      var ra = freshnessRank_(a.applicationDate);
      var rb = freshnessRank_(b.applicationDate);
      var d = (ra - rb) * SORT_DIR;  // SORT_DIR=1 → ascending day-diff = newest first
      return d !== 0 ? d : a.rowIndex - b.rowIndex; // stable: preserve sheet order on tie
    }
    var av = a[SORT_COL] || '', bv = b[SORT_COL] || '';
    if (SORT_COL === 'nextActionUrgency') {
      var ud = (URGENCY_ORD[av]||4) - (URGENCY_ORD[bv]||4);
      return ud !== 0 ? ud : a.rowIndex - b.rowIndex;
    }
    if (typeof av === 'number' && typeof bv === 'number') {
      var nd = (av - bv) * SORT_DIR;
      return nd !== 0 ? nd : a.rowIndex - b.rowIndex;
    }
    var sd = String(av).localeCompare(String(bv)) * SORT_DIR;
    return sd !== 0 ? sd : a.rowIndex - b.rowIndex;
  });

  document.getElementById('filter-count').textContent =
    FILTERED.length.toLocaleString() + ' of ' + ALL_RECORDS.length.toLocaleString() + ' candidates';
  renderTable();
  renderPagination();
}

function sortBy(col) {
  if (SORT_COL === col) { SORT_DIR *= -1; }
  else { SORT_COL = col; SORT_DIR = 1; }
  document.querySelectorAll('thead th').forEach(function(th) {
    th.classList.remove('sort-asc','sort-desc');
  });
  // Columns: chk|stage|name|nat|age|passport|loc|edu|qual|pos|trade|kai|phone|freshness|top3|pendingAction|cv
  var sortCols = ['','state','name','nationality','age','passportStatus','displayLocation',
                  'educationEnum','qualification','positionApplied','trade','techReview','','freshness',
                  'top3Positions','nextActionUrgency',''];
  var idx = sortCols.indexOf(col);
  if (idx >= 0) {
    var ths = document.querySelectorAll('thead th');
    if (ths[idx]) ths[idx].classList.add(SORT_DIR===1?'sort-asc':'sort-desc');
  }
  PAGE = 1;

}

// ── TABLE RENDER ──────────────────────────────────────────────────────
function renderTable() {
  var tbody = document.getElementById('cand-tbody');
  var start = (PAGE-1)*PAGE_SIZE;
  var end   = Math.min(start+PAGE_SIZE, FILTERED.length);
  var page  = FILTERED.slice(start, end);
  if (page.length === 0) {
    tbody.innerHTML = '';
    document.getElementById('empty-state').style.display = 'block';
    return;
  }
  document.getElementById('empty-state').style.display = 'none';
  var html = page.map(function(r) { return buildRow(r); }).join('');
  tbody.innerHTML = html;
  // Re-check selected state
  page.forEach(function(r) {
    if (SELECTED[r.rowIndex]) {
      var row = tbody.querySelector('tr[data-row="'+r.rowIndex+'"]');
      if (row) row.classList.add('selected-row');
    }
    if (PANEL_ROW === r.rowIndex) {
      var row = tbody.querySelector('tr[data-row="'+r.rowIndex+'"]');
      if (row) row.classList.add('panel-open');
    }
  });
  // Event delegation
  tbody.onclick = function(e) {
    var tr = e.target.closest('tr[data-row]');
    if (!tr) return;
    var ri  = parseInt(tr.getAttribute('data-row'));
    var chk = e.target.closest('.row-check');
    var ph  = e.target.closest('.phone-cell');
    var cv  = e.target.closest('a.cv-icon');
    if (chk) { e.stopPropagation(); toggleRowSelect(ri, tr, chk.checked); return; }
    if (ph)  { e.stopPropagation(); openContactPopup(ri, e); return; }
    if (cv)  { return; }
    openPanel(ri);
  };
}

// ── PAGINATION ────────────────────────────────────────────────────────
function renderPagination() {
  var total = FILTERED.length;
  var pages = Math.max(1, Math.ceil(total / PAGE_SIZE));
  var start = (PAGE-1)*PAGE_SIZE+1;
  var end   = Math.min(PAGE*PAGE_SIZE, total);
  document.getElementById('page-info').textContent =
    total === 0 ? 'No records' :
    'Showing ' + start + '–' + end + ' of ' + total.toLocaleString();
  var btns = document.getElementById('page-btns');
  var html = '<button class="page-btn" onclick="goPage(PAGE-1)" '+(PAGE<=1?'disabled':'')+'>‹ Prev</button>';
  var lo = Math.max(1, PAGE-2), hi = Math.min(pages, PAGE+2);
  if (lo > 1) html += '<button class="page-btn" onclick="goPage(1)">1</button>' + (lo>2?'<span style="padding:0 4px;color:var(--muted)">…</span>':'');
  for (var p = lo; p <= hi; p++) {
    html += '<button class="page-btn'+(p===PAGE?' active':'')+'" onclick="goPage('+p+')">'+p+'</button>';
  }
  if (hi < pages) html += (hi<pages-1?'<span style="padding:0 4px;color:var(--muted)">…</span>':'') +
    '<button class="page-btn" onclick="goPage('+pages+')">'+pages+'</button>';
  html += '<button class="page-btn" onclick="goPage(PAGE+1)" '+(PAGE>=pages?'disabled':'')+'>Next ›</button>';
  btns.innerHTML = html;
}

function goPage(p) {
  var total = FILTERED.length;
  var pages = Math.max(1, Math.ceil(total/PAGE_SIZE));
  PAGE = Math.max(1, Math.min(pages, p));
  renderTable();
  renderPagination();
  document.getElementById('table-wrap').scrollTop = 0;
}

// ── SELECTION ─────────────────────────────────────────────────────────
function toggleRowSelect(ri, tr, checked) {
  if (checked) { SELECTED[ri] = true; tr.classList.add('selected-row'); }
  else          { delete SELECTED[ri]; tr.classList.remove('selected-row'); }
  updateBulkToolbar();
}

function toggleSelectAll(chk) {
  var start = (PAGE-1)*PAGE_SIZE, end = Math.min(start+PAGE_SIZE, FILTERED.length);
  var page  = FILTERED.slice(start, end);
  page.forEach(function(r) {
    if (chk.checked) SELECTED[r.rowIndex] = true;
    else             delete SELECTED[r.rowIndex];
  });
  document.querySelectorAll('.row-check').forEach(function(c) { c.checked = chk.checked; });
  document.querySelectorAll('tbody tr[data-row]').forEach(function(tr) {
    tr.classList.toggle('selected-row', chk.checked);
  });
  updateBulkToolbar();
}

function clearSelection() {
  SELECTED = {};
  document.querySelectorAll('.row-check').forEach(function(c) { c.checked = false; });
  document.querySelectorAll('tbody tr.selected-row').forEach(function(tr) { tr.classList.remove('selected-row'); });
  document.getElementById('chk-all').checked = false;
  updateBulkToolbar();
}

function updateBulkToolbar() {
  var count = Object.keys(SELECTED).length;
  var tb = document.getElementById('bulk-toolbar');
  tb.classList.toggle('visible', count > 0);
  document.getElementById('bulk-count').textContent = count + ' selected';
  // Adjust main bottom padding
  document.getElementById('main').style.paddingBottom = count > 0 ? '58px' : '0';
}

function generateMissingDraft(draftType) {
  if (!PANEL_ROW) { showToast('No candidate selected'); return; }

  var keys = [];
  for (var i = 0; i < actionSel.options.length; i++) {

  resultDiv.innerHTML = '<div style="color:var(--muted);font-size:12px;padding:10px">Generating draft…</div>';

        return;

          waLink +

          '</div>';
      } else {
        resultDiv.innerHTML =
          '<div class="draft-result">' +
          '<div class="dr-hdr">Email Draft</div>' +
          '<div class="dr-subject" id="draft-subj">'+esc(res.subject||'')+'</div>' +
          '<div class="dr-body" id="draft-text-em">'+esc(res.body||'')+'</div>' +

    })
    .generateMissingInfoDraft(PANEL_ROW, keys, draftType);

function copyDraft(elId) {
  var el = document.getElementById(elId);
  if (!el) return;
  var text = el.textContent;
  if (navigator.clipboard) {
    navigator.clipboard.writeText(text).then(function() { showToast('✓ Copied to clipboard'); });
  } else {
    var ta = document.createElement('textarea');
    ta.value = text; ta.style.position = 'fixed'; ta.style.opacity = '0';
    document.body.appendChild(ta); ta.select(); document.execCommand('copy');
    document.body.removeChild(ta); showToast('✓ Copied to clipboard');
  }
}

function contactAction(type) {
  document.getElementById('contact-popup').classList.remove('visible');
  var rec = ALL_RECORDS.find(function(r) { return r.rowIndex === CONTACT_ROW; });
  if (!rec) return;
  switch(type) {
    case 'whatsapp':
      var phone = (rec.mobile||'').replace(/\D/g,'');
      if (!phone) { showToast('No phone number available'); return; }
      window.open('https://wa.me/'+phone, '_blank'); break;
    case 'call':
      var phone2 = (rec.mobile||'').replace(/\D/g,'');
      if (!phone2) { showToast('No phone number available'); return; }
      window.open('tel:'+phone2); break;
    case 'email':
      // v288: open direct email compose modal inside platform (no Gmail redirect)
      var _emailRec = ALL_RECORDS.find(function(r){ return r.rowIndex === CONTACT_ROW; });
      if (_emailRec) {
        // Load full intel to get email address, then open modal
        google.script.run
          .withSuccessHandler(function(intel) {
            if (intel && intel.email) {
              openEmailModal(CONTACT_ROW, intel.email, intel.name || rec.name, intel.positionApplied || rec.positionApplied);
            } else {
              showToast('⚠️ No email address on file for this candidate');
            }
          })
          .withFailureHandler(function() { showToast('Could not load candidate email'); })
          .getCandidateIntelligence(CONTACT_ROW);
      } else {
        showToast('Candidate not found');
      }
      break;
    case 'missing':
      openPanel(CONTACT_ROW);
      setTimeout(function() { switchTab('missing'); }, 300);
      break;
    case 'jd':
      REQ_MODAL_MODE = 'jd';
      openReqModal('Send JD to: ' + rec.name); break;
    case 'interview':
      showToast('📅 Interview scheduling — open candidate panel to update state to SHORTLISTED'); break;
  }
}

// ── BULK ACTIONS ──────────────────────────────────────────────────────
function bulkAction(type) {
  var rows = Object.keys(SELECTED).map(function(k) { return parseInt(k); });
  if (rows.length === 0) { showToast('No candidates selected'); return; }
  if (type === 'missing') {
    if (!confirm('Send missing info request to ' + rows.length + ' candidates?')) return;
    google.script.run
      .withSuccessHandler(function(res) {
        showToast('✓ Sent: ' + res.sent + ' | Failed: ' + res.failed);
        clearSelection(); refreshData();
      })
      .withFailureHandler(function(e) { showToast('Error: '+(e.message||e)); })
      .bulkSendMissingInfo(rows);
  } else if (type === 'state') {
    var newState = document.getElementById('bulk-state-sel').value;
    if (!newState) { showToast('Select a state first'); return; }
    if (!confirm('Update state to "'+newState+'" for '+rows.length+' candidates?')) return;
    google.script.run
      .withSuccessHandler(function(res) {
        showToast('✓ Updated ' + res.updated + ' candidates');
        clearSelection(); refreshData();
      })
      .withFailureHandler(function(e) { showToast('Error: '+(e.message||e)); })
      .bulkUpdateState(rows, newState);
  } else if (type === 'reject') {
    if (!confirm('Reject ' + rows.length + ' candidates? This cannot be undone easily.')) return;
    google.script.run
      .withSuccessHandler(function(res) {
        showToast('✓ Rejected ' + res.updated + ' candidates');
        clearSelection(); refreshData();
      })
      .withFailureHandler(function(e) { showToast('Error: '+(e.message||e)); })
      .bulkUpdateState(rows, 'REJECTED');
  } else if (type === 'shortlist') {
    REQ_MODAL_MODE = 'shortlist';
    openReqModal('Shortlist ' + rows.length + ' candidate(s) for:');
  } else if (type === 'whatsapp') {
    var phones = [];
    rows.forEach(function(ri) {
      var r = ALL_RECORDS.find(function(x) { return x.rowIndex === ri; });
      if (r && r.mobile) {
        var p = r.mobile.replace(/\D/g,'');
        if (p) phones.push(p);
      }
    });
    if (phones.length === 0) { showToast('No phone numbers in selection'); return; }
    showToast(phones.length + ' numbers copied — use WhatsApp Business for bulk');
  } else if (type === 'email') {
    showToast('Bulk email: use "Bulk Missing Info" for automated emails');
  }
}

// ── CLIENT-SIDE FRESHNESS COMPUTATION v288 ─────────────────────────────
// WEEKDAY-BASED SYSTEM:
//   Current calendar week (Mon–today) → weekday name (Monday, Tuesday, …)
//   Older → 15 Days / 30 Days / 45 Days / 60 Days / 3 Month / 4 Month /
//            5 Month / 6 Month / Legacy
//
// Labels MUST match PATCH_V288 backend getCvFreshnessV288_() exactly.
// applicationDate: 'yyyy-MM-dd' or 'yyyy-MM-dd HH:mm' string from backend.
function clientFreshness_(applicationDate) {
  if (!applicationDate) return '';
  var d = new Date(String(applicationDate).slice(0,10));
  if (isNaN(d.getTime())) return '';
  var now   = new Date();
  var today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  var appDay= new Date(d.getFullYear(),   d.getMonth(),   d.getDate());
  if (appDay > today) return '';
  var diffDays = Math.round((today - appDay) / 86400000);
  // Find Monday of current week
  var dow          = today.getDay();           // 0=Sun … 6=Sat
  var daysToMonday = (dow === 0) ? 6 : dow - 1;
  var weekStart    = new Date(today);
  weekStart.setDate(today.getDate() - daysToMonday);
  // Current calendar week → show day name
  if (appDay >= weekStart) {
    var DAYS = ['Sunday','Monday','Tuesday','Wednesday','Thursday','Friday','Saturday'];
    return DAYS[d.getDay()];
  }
  // Older buckets (labels match backend + filter values)
  if (diffDays <= 15)  return '15 Days';
  if (diffDays <= 30)  return '30 Days';
  if (diffDays <= 45)  return '45 Days';
  if (diffDays <= 60)  return '60 Days';
  if (diffDays <= 90)  return '3 Month';
  if (diffDays <= 120) return '4 Month';
  if (diffDays <= 150) return '5 Month';
  if (diffDays <= 180) return '6 Month';
  return 'Legacy';
}
// Numeric rank — lower = fresher. Drives newest-first sort regardless of label.
function freshnessRank_(applicationDate) {
  if (!applicationDate) return 99999;
  var d = new Date(String(applicationDate).slice(0,10));
  if (isNaN(d.getTime())) return 99999;
  var now   = new Date();
  var today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  return Math.round((today - d) / 86400000);
}
// Returns true if applicationDate falls within the current calendar week (Mon–today)
function isCurrentWeekCV_(applicationDate) {
  if (!applicationDate) return false;
  var d = new Date(String(applicationDate).slice(0,10));
  if (isNaN(d.getTime())) return false;
  var now   = new Date();
  var today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  var dow          = today.getDay();
  var daysToMonday = (dow === 0) ? 6 : dow - 1;
  var weekStart    = new Date(today);
  weekStart.setDate(today.getDate() - daysToMonday);
  return d >= weekStart && d <= today;
}

// Renders the airport-style board in the sidebar from OPEN_REQS array.
// Each row shows: ReqID · Trade · qty · Matches · Action needed · Status
function renderFlightBoard(reqs) {
  var board = document.getElementById('req-flight-board');
  var countEl = document.getElementById('req-sb-count');
  if (!board) return;
  var open = (reqs || []).filter(function(r) {
    var s = String(r.status||'').toUpperCase();
    return s === 'OPEN' || s === 'SHORTLISTING' || s === 'URGENT';
  });
  if (countEl) countEl.textContent = open.length;
  if (!open.length) {
    board.innerHTML = '<div class="fb-empty">No active requirements<br><span style="font-size:9.5px">Forward a JD email to create one</span></div>';
    return;
  }
  board.innerHTML = open.map(function(r) {
    var urgency  = String(r.urgency||'NORMAL').toUpperCase();

    var matchCnt = parseInt(r.shortlistCount||0);
    var qty      = parseInt(r.quantity||1);
    var gap      = Math.max(0, qty - matchCnt);
    return '<div class="'+rowCls+'" onclick="filterByReqFlight(\''+esc(r.reqId)+'\')" title="Click to find matching candidates">' +
      '<div class="fb-row-top">' +
        '<span class="fb-req-id">'+esc(r.reqId)+'</span>' +
        '<span class="fb-status '+stCls+'">'+stLbl+'</span>' +
      '</div>' +
      '<div class="fb-trade">'+esc(r.trade||'—')+'</div>' +
      '<div class="fb-meta">'+esc(r.clientName||'Client')+' · '+esc(r.deployCountry||'GCC')+' · '+qty+' nos</div>' +
      '<div class="fb-stats">' +
        '<div class="fb-stat"><span class="fb-stat-ico">🎯</span><span class="fb-stat-val">'+matchCnt+'</span><span class="fb-stat-lbl"> matched</span></div>' +
        (gap > 0 ? '<div class="fb-stat"><span class="fb-stat-ico">📞</span><span class="fb-stat-val" style="color:var(--red)">'+gap+'</span><span class="fb-stat-lbl"> needed</span></div>' : '') +

// ── RECRUITER SPRINT ENGINE ────────────────────────────────────────────
var SPRINT_CALLS = 0;
var SPRINT_TARGET = 5;
var SPRINT_MSGS = [
  '☕ Let\'s go — first call wins!',
  '🔥 You\'re warming up!',
  '💪 Halfway there — keep pushing!',
  '🚀 On fire! 3 more to go!',
  '⭐ Almost there — one last push!',
  '🏆 Target crushed! You\'re a legend!'
];

function initSprintWidget() {
  // Load progress from session storage (tab-scoped, resets each session)
  var saved = sessionStorage.getItem('kai_sprint');
  if (saved) {
    try {
      var s = JSON.parse(saved);
      // Reset if saved on a different day
      var today = new Date().toDateString();
      if (s.date && s.date !== today) {
        SPRINT_CALLS = 0;  // new day — fresh start
      } else {
        SPRINT_CALLS  = s.calls  || 0;
        SPRINT_TARGET = s.target || 5;
      }
    } catch(e) {}
  }
  // Sync recruiter name from localStorage (set by initUserSelector)
  var storedUser = localStorage.getItem('kai_active_user') || 'Recruiter';
  var el = document.getElementById('sprint-name');
  if (el) el.textContent = '👤 ' + storedUser;
  updateSprintDisplay();
}

function updateSprintDisplay() {
  var pct = Math.min(100, Math.round((SPRINT_CALLS / SPRINT_TARGET) * 100));
  var msgIdx = Math.min(SPRINT_MSGS.length-1, Math.floor((SPRINT_CALLS / SPRINT_TARGET) * (SPRINT_MSGS.length-1)));
  var el = function(id) { return document.getElementById(id); };
  if (el('sprint-bar'))    el('sprint-bar').style.width    = pct + '%';
  if (el('sprint-pct'))    el('sprint-pct').textContent    = pct + '%';
  if (el('sprint-done-lbl')) el('sprint-done-lbl').textContent = SPRINT_CALLS + ' done';
  if (el('sprint-target')) el('sprint-target').textContent = SPRINT_TARGET + ' calls';
  if (el('sprint-msg'))    el('sprint-msg').textContent    = SPRINT_MSGS[msgIdx];
  sessionStorage.setItem('kai_sprint', JSON.stringify({calls:SPRINT_CALLS, target:SPRINT_TARGET, date:new Date().toDateString()}));
}

// Called when recruiter logs a call action (hook into existing action tracking)
function sprintLogCall() {
  SPRINT_CALLS++;
  updateSprintDisplay();
  if (SPRINT_CALLS === SPRINT_TARGET) {
    showToast('🏆 Sprint target achieved! Outstanding work!');
  }
}

function esc(s) {
  return String(s||'')
    .replace(/&/g,'&amp;')
    .replace(/</g,'&lt;')
    .replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;')
    .replace(/'/g,'&#39;');
}

function debounce(fn, delay) {
  var timer;
  return function() {
    clearTimeout(timer);
    var args = arguments;
    timer = setTimeout(function() { fn.apply(null, args); }, delay);
  };
}

function processCsvFile(file) {
  IMP_STATE.csvFilename = file.name;
  var prog = document.getElementById('csv-progress');
  var fill = document.getElementById('csv-progress-fill');
  prog.style.display = 'block'; fill.style.width = '30%';
  document.getElementById('csv-file-name').textContent = '📄 ' + file.name;

  if (file.name.toLowerCase().endsWith('.csv')) {
    var reader = new FileReader();
    reader.onload = function(e) {
      fill.style.width = '60%';
      IMP_STATE.csvText = e.target.result;
      IMP_STATE.csvFile = file;
      // Get field mapping preview from backend
      google.script.run
        .withSuccessHandler(function(res) {
          fill.style.width = '100%';
          setTimeout(function() { prog.style.display = 'none'; fill.style.width = '0%'; }, 500);
          if (res && res.ok) showCsvMappingStep(res);
          else showToast('Mapping preview failed: ' + (res ? res.msg : 'Unknown error'));
        })
        .withFailureHandler(function(err) {
          prog.style.display = 'none';
          showToast('Error: ' + (err.message || err));
        })
        .previewFieldMapping(IMP_STATE.csvText);
    };
    reader.readAsText(file);
  } else if (file.name.toLowerCase().match(/\.xlsx?$/)) {
    // Excel: read as base64 and send to GAS for Drive conversion
    var reader2 = new FileReader();
    reader2.onload = function(e) {
      fill.style.width = '50%';
      var base64 = btoa(new Uint8Array(e.target.result).reduce(function(d,b){return d+String.fromCharCode(b);}, ''));
      document.getElementById('csv-file-name').textContent = '📊 ' + file.name + ' — Converting via Drive…';
      google.script.run
        .withSuccessHandler(function(res) {
          fill.style.width = '100%';
          setTimeout(function() { prog.style.display = 'none'; fill.style.width = '0%'; }, 500);
          if (res && res.ok) {
            showToast('✓ Excel converted (' + res.rowCount + ' rows). Submitting…');
            showCsvJobSubmitted(res.jobId, res.rowCount);
          } else {
            showToast('⚠️ ' + (res ? res.msg : 'Excel conversion failed'));
            document.getElementById('csv-file-name').textContent = res ? res.msg : 'Conversion failed';
          }
        })
        .withFailureHandler(function(err) { prog.style.display = 'none'; showToast('Error: ' + err.message); })
        .submitExcelImport(base64, file.name);
    };
    reader2.readAsArrayBuffer(file);
  }
}

// ── V285 LIVE MATCH ENGINE — client-side ────────────────────────────
/**
 * runLiveMatch(reqId, areaId, limit)
 * Called when recruiter clicks "⚡ Find Matches" in JD result card.
 * Shows loading spinner → calls matchCandidatesForReqPublic → renders ranked list.
 */
function runLiveMatch(reqId, areaId, limit) {
  var area = document.getElementById(areaId);
  if (!area) return;
  if (!reqId) { area.innerHTML = '<div style="color:var(--red);font-size:12px;padding:8px">No Req ID available — save the JD first.</div>'; return; }

  // Loading state
  area.innerHTML =
    '<div style="display:flex;align-items:center;gap:10px;padding:16px;background:var(--bg);border-radius:8px">' +
    '<div style="width:22px;height:22px;border:3px solid var(--border);border-top-color:var(--navy);border-radius:50%;animation:spin .7s linear infinite;flex-shrink:0"></div>' +
    '<div><div style="font-weight:600;font-size:12.5px">Running match engine…</div>' +
    '<div style="font-size:11px;color:var(--muted);margin-top:2px">Scoring all candidates across 7 dimensions via V285 taxonomy</div>' +
    '</div></div>';

  google.script.run
    .withSuccessHandler(function(res) {
      renderMatchResults(res, areaId);
    })
    .withFailureHandler(function(e) {
      var area2 = document.getElementById(areaId);
      if (area2) area2.innerHTML =
        '<div style="padding:10px;background:#fef2f2;border:1px solid #fecaca;border-radius:7px;color:#991b1b;font-size:12px">' +
        '⚠️ Match engine error: ' + esc(e.message || String(e)) + '</div>';
    })
    .matchCandidatesForReqPublic(reqId, limit || 20);
}

/** Helper: render one score dimension pill */
function renderScorePill_(label, val, max) {
  var pct  = max > 0 ? Math.round(val / max * 100) : 0;
  var col  = pct >= 80 ? '#10b981' : pct >= 50 ? '#3b82f6' : pct >= 25 ? '#f59e0b' : '#9ca3af';
  return '<span style="font-size:9.5px;padding:1px 5px;border-radius:5px;background:' + col + '22;color:' + col + ';border:1px solid ' + col + '44;white-space:nowrap" title="' + label + ': ' + val + '/' + max + '">' +
    label + ' ' + val + '</span>';
}

// ── REQUIREMENTS XLSX PROCESSOR ─────────────────────────────────────
function processReqXlsxFile(file) {
  var clientName = (document.getElementById('req-client-input').value || '').trim();
  if (!clientName) { showToast('Enter a client name first'); document.getElementById('req-client-input').focus(); return; }
  if (!file.name.toLowerCase().match(/\.xlsx?$/)) { showToast('Only .xlsx / .xls files accepted for requirements import'); return; }

  var country = document.getElementById('req-country-sel').value || 'GCC';
  var prog = document.getElementById('req-progress');
  var fill = document.getElementById('req-progress-fill');
  prog.style.display = 'block'; fill.style.width = '20%';
  document.getElementById('req-file-name').textContent = '📋 ' + file.name + ' — Reading…';
  document.getElementById('req-xlsx-result').innerHTML = '';

  var reader = new FileReader();
  reader.onload = function(e) {
    fill.style.width = '50%';
    var base64 = btoa(new Uint8Array(e.target.result).reduce(function(d,b){ return d + String.fromCharCode(b); }, ''));
    document.getElementById('req-file-name').textContent = '🤖 ' + file.name + ' — KAI Processing Requirements…';
    google.script.run
      .withSuccessHandler(function(res) {
        fill.style.width = '100%';
        setTimeout(function() { prog.style.display = 'none'; fill.style.width = '0%'; }, 500);
        document.getElementById('req-file-name').textContent = res && res.ok ? '✅ ' + file.name : '❌ ' + file.name;
        renderReqXlsxResult(res);
        if (res && res.ok) { loadImportStats(); var b = document.getElementById('imp-req-badge'); if (b) { b.style.display = 'inline'; b.textContent = res.saved; } }
      })
      .withFailureHandler(function(err) {
        prog.style.display = 'none';
        document.getElementById('req-file-name').textContent = '❌ Error';
        showToast('Error: ' + (err.message || err));
      })
      .processXlsxRequirementsPublic(base64, file.name, clientName, country);
  };
  reader.readAsArrayBuffer(file);
}

   Each ▾ icon opens an inline dropdown with unique values for that
   column. Selecting a value drives the corresponding filter bar
   select element and immediately calls applyFilters().
════════════════════════════════════════════════════════════════════ */

// Map column key → filter bar select element id
var HF_MAP = {
  'state':           'flt-state',
  'nationality':     'flt-nationality',
  'passportStatus':  'flt-passport',
  'displayLocation': null,          // custom — filter by location text search

  'nextActionUrgency':'flt-urgency'

var HF_ACTIVE_COL  = null;
var HF_ACTIVE_ICON = null;

// Inject dropdown HTML once
(function() {
  var d = document.createElement('div');
  d.id = 'hf-dropdown';
  d.innerHTML =
    '<input class="hfd-search" id="hfd-search" placeholder="Search…" oninput="hfSearch(this.value)">' +
    '<div class="hfd-list" id="hfd-list"></div>' +
    '<div class="hfd-clear" onclick="hfClear()">✕ Clear filter</div>';
  document.body.appendChild(d);
  document.addEventListener('click', function(e) {
    var dd = document.getElementById('hf-dropdown');
    if (dd && !e.target.closest('#hf-dropdown') && !e.target.closest('.hf-icon')) {
      dd.classList.remove('visible');
      HF_ACTIVE_ICON = null;
    }
  });
})();

function openHF(colKey, iconEl) {
  var dd = document.getElementById('hf-dropdown');
  // Toggle off if same column clicked again
  if (HF_ACTIVE_COL === colKey && dd.classList.contains('visible')) {
    dd.classList.remove('visible');
    HF_ACTIVE_COL = null; HF_ACTIVE_ICON = null;
    return;
  }
  HF_ACTIVE_COL  = colKey;
  HF_ACTIVE_ICON = iconEl;

  // Gather unique values from ALL_RECORDS for this column
  var seen = {}, vals = [];
  ALL_RECORDS.forEach(function(r) {
    var v = String(r[colKey] || '').trim();
    if (v && !seen[v]) { seen[v] = true; vals.push(v); }
  });
  vals.sort();

  // Current active value for this column
  var selId  = HF_MAP[colKey];
  var curVal = selId && document.getElementById(selId) ?
               document.getElementById(selId).value : '';

  // Build list
  var list = document.getElementById('hfd-list');
  list.innerHTML = vals.map(function(v) {
    var active = (v === curVal) ? ' selected' : '';
    return '<div class="hfd-item' + active + '" onclick="hfSelect(\'' + esc(v) + '\')">' +
           '<span class="hfd-tick">' + (v === curVal ? '✓' : '') + '</span>' +
           esc(v) + '</div>';
  }).join('') || '<div style="padding:12px;color:var(--muted);font-size:12px">No data available</div>';

  document.getElementById('hfd-search').value = '';

  // Position dropdown under the icon
  var rect = iconEl.getBoundingClientRect();
  dd.style.top  = (rect.bottom + 4) + 'px';
  dd.style.left = Math.min(rect.left, window.innerWidth - 250) + 'px';
  dd.classList.add('visible');
}

function hfSelect(value) {
  var selId = HF_MAP[HF_ACTIVE_COL];
  if (selId) {
    // Drive the filter bar select
    var sel = document.getElementById(selId);
    if (sel) {
      sel.value = value;
      // Mark icon as active
      if (HF_ACTIVE_ICON) HF_ACTIVE_ICON.classList.add('active');
    }
  } else {
    // For columns without a filter bar select, use search box
    var si = document.getElementById('search-input');
    if (si) { si.value = value; }
  }
  document.getElementById('hf-dropdown').classList.remove('visible');
  HF_ACTIVE_COL = null;
  applyFilters();
}

function hfClear() {
  var selId = HF_MAP[HF_ACTIVE_COL];
  if (selId) {
    var sel = document.getElementById(selId);
    if (sel) sel.value = '';
    if (HF_ACTIVE_ICON) HF_ACTIVE_ICON.classList.remove('active');
  }
  document.getElementById('hf-dropdown').classList.remove('visible');
  HF_ACTIVE_COL = null;
  applyFilters();
}

function hfSearch(q) {
  var items = document.querySelectorAll('#hfd-list .hfd-item');
  q = q.toLowerCase();
  items.forEach(function(item) {
    item.style.display = item.textContent.toLowerCase().indexOf(q) >= 0 ? '' : 'none';
  });
}

  } catch(e) {}

  if (_urlToken) {
    // URL has token — persist it to localStorage so refresh/reopen always works
    SESSION_TOKEN = _urlToken;
    try { localStorage.setItem('kai_session_token', _urlToken); } catch(e) {}
  } else {
    // No token in URL — check localStorage (covers: bookmark base URL, close/reopen tab)
    var _savedToken = '';
    try { _savedToken = localStorage.getItem('kai_session_token') || ''; } catch(e) {}
    if (_savedToken) {
      // Silently redirect to the same page with the saved token appended
      try {
        var _base = window.location.href.split('?')[0];
        window.location.replace(_base + '?token=' + encodeURIComponent(_savedToken));
        return; // stop here — page will reload with token in URL
      } catch(e) {}
    }
    SESSION_TOKEN = _savedToken; // fallback: use in-memory even without redirect
  }

  // Step 2 — Fast path: apply last known identity from localStorage
  var _stored = localStorage.getItem('kai_active_user') || 'hr5';
  ACTIVE_USER = _stored;
  var _sel = document.getElementById('user-role-sel');
  if (_sel) { _sel.value = _stored; }

  // Step 3 — Authoritative lock: resolve identity from server token
  if (SESSION_TOKEN && typeof google !== 'undefined' && google.script) {
    google.script.run
      .withSuccessHandler(function(user) {
        if (!user) return;
        ACTIVE_USER  = user.role || 'hr5';
        SESSION_USER = user;
        localStorage.setItem('kai_active_user', ACTIVE_USER);

        // Lock the selector — no switching allowed
        var sel = document.getElementById('user-role-sel');
        if (sel) {
          sel.value    = ACTIVE_USER;
          sel.disabled = true;
          sel.title    = '🔒 Identity locked to your login — ' + ACTIVE_USER;
          sel.style.opacity    = '0.75';
          sel.style.cursor     = 'not-allowed';
          sel.style.fontWeight = '700';
        }
        // Update sprint widget
        var sn = document.getElementById('sprint-name');
        if (sn) sn.textContent = '👤 ' + ACTIVE_USER;

        // Re-apply filters so "My Candidates" uses correct identity
        if (MY_CANDS_ACTIVE) applyFilters();
      })
      .withFailureHandler(function(e) {
        // Server unreachable — keep localStorage identity, log warning
        console.warn('[KAI] getSessionUser failed — using cached identity:', _stored);
      })
      .getSessionUser_v291_(SESSION_TOKEN);
  }
}

// EMAIL COMPOSE MODAL v288
// One-click email to candidate — no redirect to Gmail required.
// Opens inline modal inside the platform with pre-filled fields.

var EMAIL_MODAL_ROW   = null;
var EMAIL_MODAL_EMAIL = null;   // stores candidate email for Gmail compose

/**
 * Open the email compose modal for a candidate.
 * Called from the candidate panel action bar.
 */
function openEmailModal(rowIndex, toEmail, candName, posApplied) {
  EMAIL_MODAL_ROW   = rowIndex;
  EMAIL_MODAL_EMAIL = toEmail || '';
  var modal = document.getElementById('email-modal');
  if (!modal) return;

  // Pre-fill recipient display
  var toEl  = document.getElementById('em-to-display');
  var subjEl= document.getElementById('em-subject');
  var bodyEl= document.getElementById('em-body');
  if (toEl)   toEl.textContent  = (candName || 'Candidate') + ' <' + (toEmail || '') + '>';
  if (subjEl) subjEl.value      = 'Regarding Your Application' + (posApplied ? ' — ' + posApplied : '');
  var recruiterEmail = ACTIVE_USER ? (ACTIVE_USER.toLowerCase() + '@alyousufent.com') : 'ai@alyousufent.com';
  if (bodyEl) bodyEl.value      =
    'Dear ' + (candName ? candName.split(' ')[0] : 'Candidate') + ',\n\n' +
    'Thank you for your interest in the ' + (posApplied || 'position') + ' role.\n\n' +
    'We would like to discuss your application further.\n\n' +
    'Please reply to this email or contact us at your earliest convenience.\n\n' +
    'Warm regards,\n' +
    (ACTIVE_USER ? ACTIVE_USER.toUpperCase() : 'Recruitment Team') + '\n' +
    'Al Yousuf Enterprises LLP\n' +
    recruiterEmail;

  // Update footer "Sends as" label dynamically
  var sendsAsEl = document.getElementById('em-sends-as');
  if (sendsAsEl) sendsAsEl.textContent = recruiterEmail;

  // Reset send button
  var sendBtn = document.getElementById('em-send-btn');
  if (sendBtn) { sendBtn.disabled = false; sendBtn.textContent = '📧 Open in Gmail →'; }
  var statusEl = document.getElementById('em-status');
  if (statusEl) { statusEl.textContent = ''; statusEl.style.display = 'none'; }

  modal.style.display = 'flex';
}

function closeEmailModal() {
  var modal = document.getElementById('email-modal');
  if (modal) modal.style.display = 'none';
  EMAIL_MODAL_ROW = null;
}

/**
 * sendEmailFromModal — opens Gmail Compose in a new tab pre-filled with
 * To / Subject / Body. Email is sent FROM the recruiter's own logged-in
 * Gmail account (hr5@, hr7@, etc.) — not from ai@alyousufent.com.
 * No GAS quota consumed. Recruiter can personalise before sending.
 */
function sendEmailFromModal() {
  if (!EMAIL_MODAL_EMAIL) { showToast('No candidate email on file'); return; }
  var subjEl  = document.getElementById('em-subject');
  var bodyEl  = document.getElementById('em-body');
  var subject = subjEl ? subjEl.value.trim() : '';
  var body    = bodyEl ? bodyEl.value.trim() : '';
  if (!subject) { showToast('Subject is required'); return; }
  if (!body)    { showToast('Message body is required'); return; }

  var url = 'https://mail.google.com/mail/u/0/?view=cm' +
    '&to='   + encodeURIComponent(EMAIL_MODAL_EMAIL) +
    '&su='   + encodeURIComponent(subject) +
    '&body=' + encodeURIComponent(body);

  window.open(url, '_blank');
  closeEmailModal();

  var recruiterEmail = ACTIVE_USER ? (ACTIVE_USER.toLowerCase() + '@alyousufent.com') : 'Gmail';
  showToast('📧 Gmail opened — send from ' + recruiterEmail);
}

// Also expose as standalone utility (used by other callers)
function openGmailCompose(toEmail, subject, body) {
  if (!toEmail) { showToast('No email address on file'); return; }
  var url = 'https://mail.google.com/mail/u/0/?view=cm' +
    '&to='   + encodeURIComponent(toEmail) +
    '&su='   + encodeURIComponent(subject || '') +
    '&body=' + encodeURIComponent(body || '');
  window.open(url, '_blank');
}
