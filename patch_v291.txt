function callGemini_v291_(parts, circuit) {
  var apiKey    = validateApiKey_v291_();
  var model     = getEffectiveGeminiModel_();
  var url       = 'https://generativelanguage.googleapis.com/v1beta/models/' +
                  model + ':generateContent?key=' + encodeURIComponent(apiKey);
  var maxTokens = (typeof CONFIG !== 'undefined' && CONFIG.maxOutputTokens) || 4000;
  var retries   = (typeof CONFIG !== 'undefined' && CONFIG.geminiRetries)   || 5;
  var payload   = {
    contents: [{ role: 'user', parts: parts }],
    generationConfig: { responseMimeType: 'application/json', temperature: 0.1, topP: 0.1, maxOutputTokens: maxTokens }
  };
  var lastErr = null;
  for (var attempt = 1; attempt <= retries; attempt++) {
    try {
      Utilities.sleep(attempt > 1 ? Math.min(Math.pow(2, attempt - 1) * 1500, 30000) : 0);
      var resp = UrlFetchApp.fetch(url, { method: 'post', contentType: 'application/json', payload: JSON.stringify(payload), muteHttpExceptions: true });
      var code = resp.getResponseCode();
      var text = resp.getContentText();
      if (code === 200) { if (circuit) circuit.consecutive429 = 0; return text; }
      if (code === 429) {
        if (circuit) {
          circuit.consecutive429 = (circuit.consecutive429 || 0) + 1;
          var cbThreshold = (typeof CONFIG !== 'undefined' && CONFIG.geminiCircuitBreakerThreshold) || 8;
          if (circuit.consecutive429 >= cbThreshold) { circuit.broken = true; throw new Error('CIRCUIT_BROKEN after ' + circuit.consecutive429 + ' 429s'); }
        }
        var retryMs = parseInt((resp.getHeaders()['retry-after'] || '0'), 10) * 1000 || 15000;
        Utilities.sleep(Math.min(retryMs * attempt, 60000));
      }
      lastErr = new Error('Gemini HTTP ' + code + ': ' + text.slice(0, 300));
    } catch (e) {
      if (e.message && e.message.indexOf('CIRCUIT_BROKEN') === 0) throw e;
      lastErr = e;
    }
  }
  throw lastErr || new Error('Gemini failed after ' + retries + ' attempts');
}

function callGeminiString_v291_(prompt, circuit) {
  if (!prompt) return null;
  return callGemini_v291_([{ text: String(prompt) }], circuit);
}

function extractSingleJdBlock_v291_(text, subject) {
  var jd = {};
  try {
    var geminiPrompt =
      'Extract structured data from this job description. Return ONLY valid JSON, no markdown.\n\n' +
      'Fields: {"trade":"","quantity":1,"deployCountry":"","location":"","minExperience":0,"maxExperience":0,' +
      '"educationReq":"","salary":"","clientName":"","localTransfer":false,"visitVisaOK":false,' +
      '"gccPreference":false,"languageReq":"","certifications":[],"urgency":"NORMAL",' +
      '"specialRequirements":{"rotation":"","accommodation":"","transport":"","iqamaTransfer":false},' +
      '"externalRefNo":"","notes":""}\n\nJD Text:\n' +
      (typeof cleanOcrText_ === 'function' ? cleanOcrText_(text) : text).slice(0, 4000);
    var rawResp = callGeminiString_v291_(geminiPrompt, null);
    if (rawResp) {
      var gText = rawResp;
      try {
        var envelope = JSON.parse(rawResp);
        gText = ((envelope.candidates || [])[0] || {}).content
          ? ((envelope.candidates[0].content.parts || [])[0] || {}).text || rawResp : rawResp;
      } catch(pe) {}
      var s = gText.indexOf('{'), e2 = gText.lastIndexOf('}') + 1;
      if (s >= 0 && e2 > s) { var parsed = JSON.parse(gText.slice(s, e2)); if (parsed.trade) jd = parsed; }
    }
  } catch (gemErr) { Logger.log('S43: Gemini JD failed: ' + gemErr.message); }

  if (!jd.trade) {
    var cleaned = typeof cleanOcrText_ === 'function' ? cleanOcrText_(text) : text;
    var patterns = [/^position\s*:\s*([^\n]+)/im, /job\s+title\s*[:\s]+([^\n]+)/i, /role\s*:\s*([^\n]+)/i];
    for (var ti = 0; ti < patterns.length; ti++) {
      var tm = cleaned.match(patterns[ti]);
      if (tm && tm[1] && tm[1].trim().length > 2 && tm[1].trim().length < 80) { jd.trade = tm[1].trim(); break; }
    }
    if (!jd.trade && subject) jd.trade = subject.replace(/^(?:re:|fwd?:|fw:)\s*/i, '').trim();
    if (!jd.quantity) jd.quantity = 1;
    if (!jd.salary) jd.salary = 'Not Stated';
    if (!jd.urgency) jd.urgency = 'NORMAL';
    if (!jd.deployCountry) {
      var cl = cleaned.toLowerCase();
      if (/saudi|ksa/.test(cl)) jd.deployCountry = 'Saudi Arabia';
      else if (/\buae\b|dubai/.test(cl)) jd.deployCountry = 'UAE';
      else if (/qatar|doha/.test(cl)) jd.deployCountry = 'Qatar';
      else if (/oman|muscat/.test(cl)) jd.deployCountry = 'Oman';
      else if (/kuwait/.test(cl)) jd.deployCountry = 'Kuwait';
      else if (/bahrain|manama/.test(cl)) jd.deployCountry = 'Bahrain';
    }
  }
  return jd;
}

function buildGeminiPartsWithFallback_(attachment, emailBody, senderEmail, subject) {
  var maxBytes = (typeof CONFIG !== 'undefined' && CONFIG.maxAttachmentBytes) || 15728640;
  var size = attachment.getSize();
  var ext = (attachment.getName() || '').split('.').pop().toLowerCase();
  if (size <= maxBytes) { return (typeof buildGeminiParts_ === 'function') ? buildGeminiParts_(attachment, emailBody, senderEmail, subject) : null; }
  Logger.log('S45: Attachment oversized (' + size + ' bytes). Attempting text extraction.');
  var text = '';
  if (ext === 'docx' && typeof extractTextFromDocxSafe_ === 'function') text = extractTextFromDocxSafe_(attachment);
  if ((!text || text.length < 100) && typeof extractTextFromDocViaDrive_ === 'function') { try { text = extractTextFromDocViaDrive_(attachment); } catch (e) {} }
  if (!text || text.length < 100) {
    try {
      var raw = attachment.getBytes();
      text = String.fromCharCode.apply(null, raw).replace(/[^\x09\x0A\x0D\x20-\x7E]/g, ' ').replace(/\s+/g, ' ').trim().slice(0, 30000);
    } catch (e) {}
  }
  var promptFn = (typeof buildParsePrompt_ === 'function') ? buildParsePrompt_ : function(b, e, s) { return 'Parse this CV. Sender: ' + s; };
  if (text && text.length >= 100) {
    Logger.log('S45: Text extraction succeeded (' + text.length + ' chars).');
    return [{ text: promptFn(emailBody, senderEmail, subject) }, { text: 'CV text (extracted from large attachment):\n' + text.slice(0, 30000) }];
  }
  Logger.log('S45: All extraction attempts failed for ' + size + '-byte attachment.');
  return null;
}

    var runRejected = 0;
    var circuit = { consecutive429: 0, broken: false };
    var maxMs = (typeof CONFIG !== 'undefined' && CONFIG.maxExecutionMs) || 270000;
    for (var j = 0; j < threads.length; j++) {
      if (Date.now() - startMs > maxMs) break;
      var qs2 = (typeof getQuotaState_ === 'function') ? getQuotaState_() : { cvTotalRemaining: 999 };
      if (qs2.cvTotalRemaining <= 0) break;
      if (circuit.broken) break;
      try {
        var result = processThreadWithReply_(threads[j], circuit);
        if (result === 'REJECTED') runRejected++;
      } catch (e) {
        Logger.log('S46: Thread error — ' + e.message);
        try {
          if (typeof logError_ === 'function') logError_(threads[j], null, null, e.message, '');
          if (typeof moveLabel_ === 'function') { var el = (CONFIG.labels && CONFIG.labels.error) || 'karigar/error'; moveLabel_(threads[j], inputLabel, el); }
        } catch (e2) {}
      }
      if (j < threads.length - 1) Utilities.sleep(3000);
    }
    if (typeof logQuota_v291_ === 'function') logQuota_v291_(runRejected);
    else if (typeof logQuota_ === 'function') logQuota_(runRejected);
    Logger.log('S46: v291 run complete.');
  } finally { try { lock.releaseLock(); } catch (e) {} }
}

function processThread_v291_(thread, circuit) {

function processThreadWithReply_(thread, circuit) {
  var result = processThread_v291_(thread, circuit);

  // Mark as read in ai@alyousufent.com inbox after any processing
  try { thread.markRead(); } catch(e) {}

  // Re-engagement: known candidate replied with updated CV
  if (result === 'DUPLICATE') {
    try {
      var msgs = thread.getMessages();
      var msg  = msgs[msgs.length - 1];
      var att  = (typeof pickBestAttachment_ === 'function') ? pickBestAttachment_(msg) : null;
      if (att) {
        var senderEmail = extractEmailFromSender_(msg.getFrom());
        var rowNum      = senderEmail ? findCandidateByEmail_v291_(senderEmail) : 0;
        if (rowNum) {
          Logger.log('Re-engagement: ' + senderEmail + ' row ' + rowNum);
          return handleReEngagement_(thread, msg, att, senderEmail, rowNum, circuit);
        }
      }
    } catch(e) { Logger.log('processThreadWithReply_: re-engagement failed — ' + e.message); }
  }

  // Parse failure: queue resend-request reply
  if (result === 'ERROR_GEMINI' || result === 'ERROR_SIZE' || result === 'ERROR_NO_CV') {
    try {
      var msgs2 = thread.getMessages();
      var msg2  = msgs2[msgs2.length - 1];
      queueResendReply_(extractEmailFromSender_(msg2.getFrom()), msg2.getSubject() || '');
    } catch(e) { Logger.log('processThreadWithReply_: reply queue failed — ' + e.message); }
  }

  return result;
}

function backfillCVs() {
  var BATCH = 75, DELAY_MS = 300;
  var props = PropertiesService.getScriptProperties();
  var startIdx = parseInt(props.getProperty('bf_idx') || '0');
  props.setProperty('BACKFILL_MODE', 'true');
  var threads = GmailApp.search('to:ai@alyousufent.com after:2026/01/01', startIdx, BATCH);
  if (!threads || threads.length === 0) {
    Logger.log('✅ BACKFILL COMPLETE from index ' + startIdx);
    props.deleteProperty('bf_idx'); props.deleteProperty('BACKFILL_MODE'); return;
  }
  var processed = 0, skipped = 0, errors = 0;
  var startTime = new Date().getTime();
  for (var i = 0; i < threads.length; i++) {
    if (new Date().getTime() - startTime > 300000) { Logger.log('⏱ Time limit — saving progress.'); break; }
    try {
      var msgs = threads[i].getMessages();
      if (!msgs || !msgs.length) { skipped++; continue; }
      var from = msgs[0].getFrom();
      var em = from.match(/[\w.+\-]+@[\w\-]+\.[a-z.]+/i);
      var se = em ? em[0].toLowerCase() : '';
      if (se && isAlreadyScored_(se)) { skipped++; continue; }
      Logger.log('[' + (startIdx + i + 1) + '] ' + threads[i].getFirstMessageSubject().slice(0, 60));
      processThreadWithReply_(threads[i]);
      processed++;
      Utilities.sleep(DELAY_MS);
    } catch(e) { errors++; Logger.log('ERR [' + (startIdx + i) + ']: ' + e.message); Utilities.sleep(500); }
  }
  var nextIdx = startIdx + processed + skipped;
  props.setProperty('bf_idx', String(nextIdx));
  Logger.log('⏩ [' + startIdx + '→' + nextIdx + '] ✅ ' + processed + ' | ⏭ ' + skipped + ' | ❌ ' + errors);
}

function isAlreadyScored_(email) {
  try {
    var rowNum = findCandidateByEmail_v291_(email);
    if (!rowNum) return false;
    var score = getMasterSS_().getSheetByName(CONFIG.sheetName).getRange(rowNum, 17).getValue();
    return !!(score && parseFloat(score) > 0);
  } catch(e) { return false; }
}

function resetBackfill() {
  PropertiesService.getScriptProperties().deleteProperty('bf_idx');
  PropertiesService.getScriptProperties().deleteProperty('BACKFILL_MODE');
  Logger.log('Backfill reset — next run starts from Jan 2026.');
}

function autoBackfillTrigger() {
  ScriptApp.getProjectTriggers().forEach(function(t) { if (t.getHandlerFunction() === 'backfillCVs') ScriptApp.deleteTrigger(t); });
  ScriptApp.newTrigger('backfillCVs').timeBased().everyMinutes(5).create();
  Logger.log('Auto-trigger set — backfillCVs() runs every 5 min.');
}

  ScriptApp.getProjectTriggers().forEach(function(t) { if (t.getHandlerFunction() === 'backfillCVs') ScriptApp.deleteTrigger(t); });
  PropertiesService.getScriptProperties().deleteProperty('BACKFILL_MODE');
  Logger.log('Auto-trigger stopped. Quota limits restored.');
}

function backfillErrorReplies() {
  var ss = getMasterSS_();
  var sheet = ss.getSheetByName('_Errors');
  if (!sheet || sheet.getLastRow() < 2) { Logger.log('_Errors sheet empty.'); return; }
  var lastCol = sheet.getLastColumn();
  if (lastCol < 8) sheet.getRange(1, 8).setValue('Reply Status');
  else if (!sheet.getRange(1, 8).getValue()) sheet.getRange(1, 8).setValue('Reply Status');
  var totalRows = sheet.getLastRow() - 1;
  var data = sheet.getRange(2, 1, totalRows, 8).getValues();
  var queued = 0, skipped = 0;
  for (var i = 0; i < data.length; i++) {
    if (String(data[i][7]||'').toUpperCase().trim() === 'REPLIED') { skipped++; continue; }
    var toEmail = extractEmailFromSender_(String(data[i][2]||''));
    var subject = String(data[i][3]||'');
    if (!toEmail) { sheet.getRange(i+2,8).setValue('NO_EMAIL'); skipped++; continue; }
    if (typeof isInternalDomain_ === 'function' && isInternalDomain_(toEmail)) { sheet.getRange(i+2,8).setValue('INTERNAL_SKIP'); skipped++; continue; }
    queuePendingEmail_v291_(toEmail, subject ? 'Re: '+subject : 'Your CV Application — Al Yousuf Enterprises', RESEND_REPLY_BODY_, null, 'Al Yousuf Recruitment');
    sheet.getRange(i+2,8).setValue('REPLIED');
    queued++;
    if (queued % 50 === 0) flushPendingEmails_v291_();
  }
  flushPendingEmails_v291_();
  Logger.log('backfillErrorReplies DONE — Queued: ' + queued + ' | Skipped: ' + skipped + ' / ' + totalRows);
}

function sendRecruiterDailyReport() {
  var tz = 'Asia/Kolkata';
  var today      = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd');
  var todayLabel = Utilities.formatDate(new Date(), tz, 'd MMM yyyy');
  var ic = CONFIG.inputColumns;
  var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
  if (!sheet || sheet.getLastRow() < 2) return;
  var rows = sheet.getRange(2, 1, sheet.getLastRow()-1, CONFIG.headers.length).getValues();
  var total = 0, shortlisted = 0, needsReview = 0, rejected = 0;
  var trades = {};
  rows.forEach(function(row) {
    var ts = row[ic.applicationDate - 1];
    var ds = ts instanceof Date ? Utilities.formatDate(ts, tz, 'yyyy-MM-dd') : String(ts).slice(0,10);
    if (ds !== today) return;
    total++;
    var v = String(row[ic.verdict - 1] || '').toUpperCase();
    if (v === 'SHORTLISTED') shortlisted++;
    else if (v === 'NEEDS_REVIEW' || v === 'NEEDS_CALL') needsReview++;
    else if (v === 'REJECTED' || v === 'REJECT') rejected++;
    var t = String(row[ic.trade - 1] || '').trim();
    if (t) trades[t] = (trades[t] || 0) + 1;
  });
  var topTrades   = Object.keys(trades).sort(function(a,b){return trades[b]-trades[a];}).slice(0,3);
  var industry    = buildIndustrySuggestion_(topTrades, trades);
  var actions     = shortlisted + needsReview;
  var dailyTarget = 8;
  var recruiters  = getRecruitersForReport_();
  if (!recruiters.length) recruiters = [{ name:'Team', email: getOwnerEmail_() }];
  recruiters.forEach(function(rec) {
    var html = buildRecruiterReportHtml_(rec.name, todayLabel, total, shortlisted, needsReview, rejected, actions, dailyTarget, topTrades, trades, industry);
    try {
      MailApp.sendEmail({ to: rec.email, subject: 'Daily Update · ' + todayLabel + ' · ' + total + ' profile' + (total!==1?'s':'') + ' in', htmlBody: html, name: (CONFIG.brand && CONFIG.brand.companyName) || 'Al Yousuf Enterprises' });
      Logger.log('Recruiter report sent → ' + rec.email);
    } catch(e) { Logger.log('sendRecruiterDailyReport: failed for ' + rec.email + ' — ' + e.message); }
  });
}

function getRecruitersForReport_() {
  var result = [];
  try {
    var sheet = getMasterSS_().getSheetByName('_LoginSystem');
    if (!sheet || sheet.getLastRow() < 2) return result;
    sheet.getRange(2, 1, sheet.getLastRow()-1, 8).getValues().forEach(function(r) {
      var email = String(r[0]||'').trim().toLowerCase();
      var role  = String(r[2]||'').toLowerCase();

      result.push({ name: name, email: email });

  return result;
}

function buildIndustrySuggestion_(topTrades, trades) {
  var map = {
    'electrician':'construction & MEP','electrical':'construction & MEP',
    'plumber':'construction & MEP','hvac':'MEP & facilities','ac technician':'MEP & facilities',
    'welder':'oil & gas / fabrication','fitter':'oil & gas / fabrication','pipe fitter':'oil & gas / fabrication',
    'rigger':'oil & gas / offshore','scaffolder':'oil & gas / construction',
    'operator':'oil & gas / heavy industry','mason':'construction','steel fixer':'construction',
    'carpenter':'construction & fit-out','painter':'construction & fit-out','tiler':'construction & fit-out',
    'driver':'logistics & transport','heavy driver':'logistics & transport',
    'forklift':'warehouse & logistics','helper':'general contracting'
  };
  var industries = {};
  topTrades.forEach(function(t) { var ind = map[t.toLowerCase().trim()] || ''; if (ind) industries[ind] = (industries[ind]||0) + (trades[t]||1); });

}

function inferTradeIndustry_(trade) {
  var t = trade.toLowerCase();
  var rules = [
    [/oil|gas|petroleum|refin|offshore|pipeline|upstream|downstream|driller|wellhead|lng|lpg/, 'Oil & Gas'],
    [/weld|fabricat|piping.*fitter|structural.*steel/, 'Oil & Gas'],
    [/hse|safety officer|fire.*fight/, 'Oil & Gas'],
    [/\bqc\b|\bqa\b|quality.*control|ndt|radiograph|ultrasonic.*test|weld.*inspect/, 'Oil & Gas'],
    [/instrument|dcs|scada|\bplc\b|control.*system/, 'Oil & Gas'],
    [/mechanical.*tech|rotating.*equip/, 'Oil & Gas'],
    [/civil|concrete|mason|rebar|formwork|scaffold|surveyor|site.*engineer|construct/, 'Construction'],
    [/landscap|irrigation|horticulture/, 'Architecture & Construction'],
    [/hvac|plumb|mep|chiller|air.*condit|refrigerat/, 'Architecture & Construction'],
    [/electric/, 'Power & Utilities'],
    [/cement|mining|quarry|steel.*plant/, 'Cement Mining & Steel'],
    [/pharma|chemical|fertilizer|rubber|plastic/, 'Chemical / Fertilizers / Plastic / Rubber / Glass'],
    [/software|developer|programm|network.*admin|cloud|cyber|database/, 'IT & Technology'],
    [/nurse|doctor|physician|pharmacist|lab.*tech|radiolog/, 'Healthcare'],
    [/accountant|financ.*officer|audit|tax.*officer/, 'Banking & Finance'],
    [/logistic|supply chain|warehouse|fleet.*manage/, 'Logistics & Transportation'],
    [/sales|business dev|crm/, 'Sales & Marketing'],
    [/chef|cook|hospitality|housekeep|steward|catering/, 'Hospitality & Catering'],
    [/aircraft|aviation|pilot|cabin crew/, 'Aviation & Aerospace'],
    [/\bship\b|marine|deck officer|naval/, 'Shipping & Marine']
  ];
  for (var i = 0; i < rules.length; i++) { if (rules[i][0].test(t)) return rules[i][1]; }
  return 'Construction';
}

function inferTradeDepartment_(trade) {
  var t = trade.toLowerCase();
  if (/manager|director|head of|\bvp\b|ceo|coo/.test(t)) return 'Management';
  if (/safety|hse|ehs|\bqc\b|\bqa\b|quality|inspector|ndt/.test(t)) return 'Quality & Safety';
  if (/engineer/.test(t)) return 'Engineering';
  if (/supervisor|foreman|\blead\b/.test(t)) return 'Technical';
  if (/technician|mechanic/.test(t)) return 'Technical';
  if (/operator/.test(t)) return 'Operations';
  if (/admin|secretary|coordinator/.test(t)) return 'Administration';
  return 'Technical';
}

function growTaxonomyFromCandidates() {
  var ss = getMasterSS_(); if (!ss) return;
  var candSheet = ss.getSheetByName(CONFIG.sheetName || 'Candidates'); if (!candSheet) return;
  var taxSheet  = ss.getSheetByName('Taxonomy'); if (!taxSheet) return;

  // Build case-insensitive set of existing trades
  var taxRows = taxSheet.getLastRow() > 1
    ? taxSheet.getRange(2, 1, taxSheet.getLastRow()-1, 3).getValues() : [];
  var existing = {};
  taxRows.forEach(function(r) { if (r[2]) existing[String(r[2]).toLowerCase().trim()] = true; });

  // Count trades from candidates
  var lr = candSheet.getLastRow();
  if (lr < 2) { Logger.log('No candidates.'); return; }
  var tradeCol = candSheet.getRange(2, 9, lr-1, 1).getValues(); // col 9 = Trade
  var counts = {};
  tradeCol.forEach(function(r) {
    var t = String(r[0]||'').trim();
    if (t && t !== 'TEST TRADE' && t.length > 2 && !/^\d+$/.test(t))
      counts[t] = (counts[t]||0) + 1;
  });

  // Find and add missing trades, sorted by frequency
  var missing = Object.keys(counts).filter(function(t) { return !existing[t.toLowerCase()]; });
  missing.sort(function(a,b){ return counts[b]-counts[a]; });

  var newRows = missing.map(function(trade) {
    return [inferTradeIndustry_(trade), inferTradeDepartment_(trade), trade, ''];
  });
  if (newRows.length === 0) { Logger.log('Taxonomy is already up to date.'); return; }

  taxSheet.getRange(taxSheet.getLastRow()+1, 1, newRows.length, 4).setValues(newRows);
  Logger.log('growTaxonomyFromCandidates: added ' + newRows.length + ' trades. Top 10: ' +
    missing.slice(0,10).map(function(t){ return t+'('+counts[t]+')'; }).join(', '));
}

function deduplicateTaxonomy() {
  var ss = getMasterSS_(); if (!ss) return;
  var sheet = ss.getSheetByName('Taxonomy'); if (!sheet || sheet.getLastRow() < 2) return;
  var data = sheet.getRange(2, 1, sheet.getLastRow()-1, 4).getValues();
  var seen = {}, keep = [], removed = 0;
  data.forEach(function(r) {
    var key = (String(r[0]||'') + '|' + String(r[1]||'') + '|' + String(r[2]||'') + '|' + String(r[3]||'')).toLowerCase().trim();
    if (!seen[key] && (r[0]||r[2])) { seen[key]=true; keep.push(r); }
    else removed++;
  });
  if (removed === 0) { Logger.log('No duplicates found.'); return; }
  sheet.getRange(2, 1, sheet.getLastRow()-1, 4).clearContent();
  sheet.getRange(2, 1, keep.length, 4).setValues(keep);
  Logger.log('deduplicateTaxonomy: removed ' + removed + ' duplicate rows. ' + keep.length + ' clean rows remain.');
}

function approvePendingTaxonomySuggestions() {
  var ss = getMasterSS_(); if (!ss) return;
  var sugSheet = ss.getSheetByName('_Taxonomy_Suggestions'); if (!sugSheet || sugSheet.getLastRow() < 2) return;
  var taxSheet = ss.getSheetByName('Taxonomy'); if (!taxSheet) return;
  var data = sugSheet.getRange(2, 1, sugSheet.getLastRow()-1, 8).getValues();
  var approved = 0;
  data.forEach(function(r, i) {
    if (String(r[5]||'').toUpperCase() !== 'PENDING') return;
    var trade = String(r[3]||'').trim(); if (!trade) return;
    var industry = inferTradeIndustry_(trade);
    var dept     = inferTradeDepartment_(trade);
    taxSheet.appendRow([industry, dept, trade, '']);
    sugSheet.getRange(i+2, 6).setValue('APPROVED');
    sugSheet.getRange(i+2, 7).setValue('KAI-AUTO');
    approved++;
  });
  Logger.log('approvePendingTaxonomySuggestions: approved ' + approved + ' suggestions.');
}

function retryUnrepliedErrors() {
  var ss = getMasterSS_(); if (!ss) return;
  var errSheet = ss.getSheetByName('_Errors'); if (!errSheet || errSheet.getLastRow() < 2) return;
  var data = errSheet.getRange(2, 1, errSheet.getLastRow()-1, 8).getValues();
  var queued = 0;
  data.forEach(function(r, i) {
    var replyStatus = String(r[7]||'').toUpperCase();
    if (replyStatus && replyStatus !== 'INTERNAL_SKIP') return; // already handled
    var sender = String(r[2]||'');
    var toEmail = (typeof extractEmailFromSender_ === 'function') ? extractEmailFromSender_(sender) : '';
    if (!toEmail || (typeof isInternalDomain_ === 'function' && isInternalDomain_(toEmail))) {
      errSheet.getRange(i+2, 8).setValue('INTERNAL_SKIP');
      return;
    }
    var errType = String(r[4]||'');
    // Only queue resend reply for no-CV errors — Gemini nulls already got one at parse time
    if (/No valid CV attachment|No CV attachment|body.*too short/i.test(errType)) {
      var subj = String(r[3]||'');
      if (typeof queueResendReply_ === 'function') queueResendReply_(toEmail, subj);
      errSheet.getRange(i+2, 8).setValue('REPLIED');
      queued++;
    }
  });
  if (typeof flushPendingEmails_v291_ === 'function') flushPendingEmails_v291_();
  Logger.log('retryUnrepliedErrors: queued replies for ' + queued + ' errors.');
}

function printDailyQuotaSummary() {
  var ss = getMasterSS_(); if (!ss) return;
  var qSheet = ss.getSheetByName('_Quota'); if (!qSheet || qSheet.getLastRow() < 2) { Logger.log('No quota data.'); return; }

  // Gemini 2.5-flash-lite Tier 1 paid: 1500 RPM, no daily cap → effectively unlimited per day
  // Gmail Workspace send limit: 1500 emails/day
  // GAS: 6 min/execution, ~90 trigger executions/day
  var GEMINI_RPM_LIMIT = 1500; // per minute
  var GMAIL_DAILY_LIMIT = 1500;
  var SAFE_DAILY_GEMINI = 1000; // conservative daily target to stay well under burst

  var lr = qSheet.getLastRow();
  var rows = qSheet.getRange(2, 1, lr-1, 6).getValues();
  var last7 = rows.slice(-7);

  var lines = ['═══ KAI DAILY QUOTA SUMMARY ═══', ''];
  lines.push('Limits:');
  lines.push('  Gemini 2.5-flash-lite (Tier 1 paid) — 1500 RPM, no daily cap');
  lines.push('  Gmail Workspace — 1500 emails/day');
  lines.push('  Safe daily Gemini target — ≤ ' + SAFE_DAILY_GEMINI + ' calls/day');
  lines.push('');
  lines.push('Last 7 days:');
  lines.push('  Date         CV-Fresh  CV-BL  JDs  Replies  Emails  Total-API  Gmail%');
  last7.forEach(function(r) {
    var dateStr = r[0] instanceof Date
      ? Utilities.formatDate(r[0], 'Asia/Kolkata', 'dd MMM')
      : String(r[0]).slice(5,10);
    var cvF  = r[1]||0, cvB = r[2]||0, jd = r[3]||0, rep = r[4]||0, em = r[5]||0;
    var api  = cvF + cvB + jd + rep;
    var gmailPct = Math.round((em / GMAIL_DAILY_LIMIT) * 100);
    var flag = api > SAFE_DAILY_GEMINI ? ' ⚠ HIGH' : '';
    lines.push('  ' + dateStr + '        ' + pad_(cvF,6) + '  ' + pad_(cvB,5) + '  ' + pad_(jd,3) + '  ' + pad_(rep,7) + '  ' + pad_(em,6) + '  ' + pad_(api,9) + '  ' + gmailPct + '%' + flag);
  });

  // Today totals
  var today = last7[last7.length-1];
  var todayApi = (today[1]||0) + (today[2]||0) + (today[3]||0) + (today[4]||0);
  lines.push('');
  lines.push('Today: ' + todayApi + ' Gemini API calls  |  ' + (today[5]||0) + ' emails sent');
  lines.push('Gemini headroom: ' + (SAFE_DAILY_GEMINI - todayApi) + ' calls before soft limit');
  lines.push('Gmail headroom:  ' + (GMAIL_DAILY_LIMIT - (today[5]||0)) + ' emails remaining today');

  // Pending emails
  var pendSheet = ss.getSheetByName('_PendingEmails');
  if (pendSheet && pendSheet.getLastRow() > 1) {
    var pendData = pendSheet.getRange(2, 1, pendSheet.getLastRow()-1, 6).getValues();
    var queued = pendData.filter(function(r){ return String(r[5]).toUpperCase()==='QUEUED'; }).length;
    lines.push('');
    lines.push('_PendingEmails QUEUED: ' + queued + ' (flush runs automatically with pipeline)');
    if (queued > 0) {
      var batchesNeeded = Math.ceil(queued / 50);
      lines.push('  Estimated flushes needed: ' + batchesNeeded + ' trigger runs (~' + batchesNeeded + ' min)');
    }
  }

  // Errors without reply
  var errSheet = ss.getSheetByName('_Errors');
  if (errSheet && errSheet.getLastRow() > 1) {
    var errData = errSheet.getRange(2, 1, errSheet.getLastRow()-1, 8).getValues();
    var noReply = errData.filter(function(r){ return !r[7]; }).length;
    if (noReply > 0) lines.push('\n⚠ Unreplied errors: ' + noReply + ' — run retryUnrepliedErrors() to fix');
  }

  lines.push('');
  lines.push('═══════════════════════════════');
  Logger.log(lines.join('\n'));
}

function pad_(n, w) {
  var s = String(n); while (s.length < w) s = ' ' + s; return s;
}
