// S42.F03 — upgraded callGemini_v291_ using override model
// Accepts parts as an ARRAY of {text:...} objects (the correct API format).
function callGemini_v291_(parts, circuit) {
  var apiKey    = validateApiKey_v291_();
  var model     = getEffectiveGeminiModel_();
  var url       = 'https://generativelanguage.googleapis.com/v1beta/models/' +
                  model + ':generateContent?key=' + encodeURIComponent(apiKey);
  var maxTokens = (typeof CONFIG !== 'undefined' && CONFIG.maxOutputTokens) || 4000;
  var retries   = (typeof CONFIG !== 'undefined' && CONFIG.geminiRetries)   || 5;
  var payload   = {
    contents: [{ role: 'user', parts: parts }],
    generationConfig: {
      responseMimeType: 'application/json',
      temperature: 0.1, topP: 0.1,
      maxOutputTokens: maxTokens
    }
  };
  var lastErr = null;
  for (var attempt = 1; attempt <= retries; attempt++) {
    try {
      Utilities.sleep(attempt > 1 ? Math.min(Math.pow(2, attempt - 1) * 1500, 30000) : 0);
      var resp = UrlFetchApp.fetch(url, {
        method: 'post', contentType: 'application/json',
        payload: JSON.stringify(payload), muteHttpExceptions: true
      });
      var code = resp.getResponseCode();
      var text = resp.getContentText();
      if (code === 200) {
        if (circuit) circuit.consecutive429 = 0;
        return text;
      }
      if (code === 429) {
        if (circuit) {
          circuit.consecutive429 = (circuit.consecutive429 || 0) + 1;
          var cbThreshold = (typeof CONFIG !== 'undefined' && CONFIG.geminiCircuitBreakerThreshold) || 8;
          if (circuit.consecutive429 >= cbThreshold) {
            circuit.broken = true;
            throw new Error('CIRCUIT_BROKEN after ' + circuit.consecutive429 + ' 429s');
          }
        }
        var retryMs = parseInt((resp.getHeaders()['retry-after'] || '0'), 10) * 1000 || 15000;
        Utilities.sleep(Math.min(retryMs * attempt, 60000));
      }
      lastErr = new Error('Gemini HTTP ' + code + ': ' + text.slice(0, 300));
    } catch (e) {
      if (e.message && e.message.indexOf('CIRCUIT_BROKEN') === 0) throw e;
      lastErr = e;
    }
  }
  throw lastErr || new Error('Gemini failed after ' + retries + ' attempts');
}

// S42.F04 — STRING-to-PARTS compatibility wrapper (FIX for H1)
// v284's extractSingleJdBlock_ calls callGemini_(prompt) where prompt is a STRING.
// Code.gs's callGemini_ expects (parts:ARRAY, circuit). This wrapper fixes the mismatch.
// extractSingleJdBlock_v291_ below calls this instead of callGemini_.
function callGeminiString_v291_(prompt, circuit) {
  if (!prompt) return null;
  var parts = [{ text: String(prompt) }];
  return callGemini_v291_(parts, circuit);
}

// S43.F01 — Fixed single JD block extractor (replaces v284 S73.F01 logic)
function extractSingleJdBlock_v291_(text, subject) {
  var jd = {};

  // Step 1: Gemini with CORRECT parts array format (fixes the string/array bug)
  try {
    var geminiPrompt =
      'Extract structured data from this job description. Return ONLY valid JSON, no markdown.\n\n' +
      'Fields to extract:\n' +
      '{"trade":"","quantity":1,"deployCountry":"","location":"","minExperience":0,"maxExperience":0,' +
      '"educationReq":"","salary":"","clientName":"","localTransfer":false,"visitVisaOK":false,' +
      '"gccPreference":false,"languageReq":"","certifications":[],"urgency":"NORMAL",' +
      '"specialRequirements":{"rotation":"","accommodation":"","transport":"","iqamaTransfer":false},' +
      '"externalRefNo":"","notes":""}\n\n' +
      'JD Text:\n' +
      (typeof cleanOcrText_ === 'function' ? cleanOcrText_(text) : text).slice(0, 4000);

    // Correct call: wrap prompt in array (callGeminiString_v291_ does this)
    var rawResp = callGeminiString_v291_(geminiPrompt, null);
    if (rawResp) {
      // Extract JSON from Gemini response envelope
      var gText = rawResp;
      try {
        var envelope = JSON.parse(rawResp);
        gText = ((envelope.candidates || [])[0] || {}).content
          ? ((envelope.candidates[0].content.parts || [])[0] || {}).text || rawResp
          : rawResp;
      } catch(pe) {}
      var s = gText.indexOf('{'), e2 = gText.lastIndexOf('}') + 1;
      if (s >= 0 && e2 > s) {
        var parsed = JSON.parse(gText.slice(s, e2));
        if (parsed.trade) jd = parsed;
      }
    }
  } catch (gemErr) {
    Logger.log('S43: Gemini JD extraction failed, using fuzzy fallback: ' + gemErr.message);
  }

  // Step 2: Delegate regex fallback to v284 helpers (additive)
  // These helpers exist in v284 and v285 — call them safely
  if (!jd.trade && typeof extractSingleJdBlock_ === 'function') {
    // v284 extractSingleJdBlock_ has the regex fallback but wrong Gemini call.
    // Since Gemini already failed above, the regex fallback inside v284 is safe
    // to use as long as we catch any Gemini error inside it.
    // We skip calling it to avoid the string/array bug — regex fallback only:
  }

  // Minimal regex trade extraction if Gemini returned nothing
  if (!jd.trade) {
    var cleaned = typeof cleanOcrText_ === 'function' ? cleanOcrText_(text) : text;
    var tradePatterns = [
      /^position\s*:\s*([^\n]+)/im,
      /job\s+title\s*[:\s]+([^\n]+)/i,
      /role\s*:\s*([^\n]+)/i,
      /seeking\s+(?:a\s+)?([^\n]+?)(?:\s+to\s+join|\s+for|\s*$)/i
    ];
    for (var ti = 0; ti < tradePatterns.length; ti++) {
      var tm = cleaned.match(tradePatterns[ti]);
      if (tm && tm[1] && tm[1].trim().length > 2 && tm[1].trim().length < 80) {
        jd.trade = tm[1].trim();
        break;
      }
    }
    if (!jd.trade && subject) jd.trade = subject.replace(/^(?:re:|fwd?:|fw:)\s*/i, '').trim();
    if (!jd.quantity)       jd.quantity = 1;
    if (!jd.salary)         jd.salary   = 'Not Stated';
    if (!jd.urgency)        jd.urgency  = 'NORMAL';
    if (!jd.deployCountry) {
      var cl = cleaned.toLowerCase();
      if (/saudi|ksa/.test(cl))            jd.deployCountry = 'Saudi Arabia';
      else if (/\buae\b|dubai/.test(cl))   jd.deployCountry = 'UAE';
      else if (/qatar|doha/.test(cl))      jd.deployCountry = 'Qatar';
      else if (/oman|muscat/.test(cl))     jd.deployCountry = 'Oman';
      else if (/kuwait/.test(cl))          jd.deployCountry = 'Kuwait';
      else if (/bahrain|manama/.test(cl))  jd.deployCountry = 'Bahrain';
    }
  }

  return jd;
}

// S43.F02 — Fixed full JD extraction chain (replaces v284 S73.F02 / extractJdV284_)
function extractJdV284_v291_(text, subject) {
  var result = { ok: false, jds: [], count: 0, errors: [] };
  var blocks = typeof splitMultipleJds_ === 'function' ? splitMultipleJds_(text) : [text];

  blocks.forEach(function(block, idx) {
    try {
      var jd = extractSingleJdBlock_v291_(block, subject);
      if (jd.trade || jd.externalRefNo) result.jds.push(jd);
    } catch (e) {
      result.errors.push('Block ' + (idx + 1) + ': ' + e.message);
    }
  });

  result.ok    = result.jds.length > 0;
  result.count = result.jds.length;
  return result;
}

// S43.F03 — Fixed public JD file upload entry point
// Dashboard: change google.script.run.extractJdPublicV2(...) →
//            google.script.run.extractJdPublicV2Fixed(...)
function extractJdPublicV2Fixed(base64Data, filename, mimeType) {
  var result = { ok:false, count:0, jds:[], isMulti:false, msg:'', errors:[] };
  var tempFileId = null, docFileId = null;
  try {
    var mime = mimeType || 'application/pdf';
    var blob = Utilities.newBlob(
      Utilities.base64Decode(base64Data), mime, filename || 'jd_upload');

    var text = '';
    try {
      var tempFile = DriveApp.createFile(blob);
      tempFileId   = tempFile.getId();
      var docMeta  = Drive.Files.copy(
        { title: 'kai_tmp_jd_v291_' + Date.now(),
          mimeType: 'application/vnd.google-apps.document' },
        tempFileId);
      docFileId = docMeta.id;
      text = DocumentApp.openById(docFileId).getBody().getText();
    } catch(convErr) {
      result.errors.push('Text extraction failed: ' + convErr.message);
    }

    if (!text || text.trim().length < 20) {
      result.msg = 'Could not extract readable text. Please paste the JD text manually.';
      return result;
    }

    var extracted = extractJdV284_v291_(text, filename);
    if (!extracted.ok) {
      result.msg = 'Could not identify a valid JD.';
      result.errors = extracted.errors;
      return result;
    }

    var jds = extracted.jds;
    if (jds.length === 1) {
      var reqId = typeof saveRequirementV284_ === 'function' ? saveRequirementV284_(jds[0]) : 'REQ-UNSAVED';
      jds[0].reqId = reqId;
      result.ok = true; result.count = 1; result.jds = jds; result.isMulti = false;
      result.msg = '1 position extracted and saved';
      return result;
    }

    // Multi-JD: cache and return previews (same pattern as v285 S86)
    var batchId = 'B' + Date.now() + '_' + Math.floor(Math.random() * 9999);
    try {
      CacheService.getScriptCache().put('JD_BATCH_' + batchId, JSON.stringify(jds), 1800);
    } catch(ce) {
      var saved = [];
      jds.forEach(function(jd) {
        try { var rId = typeof saveRequirementV284_ === 'function' ? saveRequirementV284_(jd) : 'REQ-UNSAVED'; jd.reqId = rId; saved.push(jd); } catch(e) {}
      });
      result.ok = true; result.count = saved.length; result.jds = saved; result.isMulti = true;
      result.msg = saved.length + ' positions saved (cache unavailable)';
      return result;
    }
    var previews = jds.map(function(jd, idx) {
      return { idx:idx, trade:jd.trade||'', clientName:jd.clientName||'',
               deployCountry:jd.deployCountry||'', quantity:jd.quantity||1,
               salary:jd.salary||'', urgency:jd.urgency||'' };
    });
    result.ok = true; result.isMulti = true; result.pendingSave = true;
    result.batchId = batchId; result.count = jds.length; result.jds = previews;
    result.msg = jds.length + ' positions found — please confirm which to save';
    return result;

  } catch(err) {
    result.msg = 'Extraction error: ' + err.message;
    return result;
  } finally {
    try { if (tempFileId) DriveApp.getFileById(tempFileId).setTrashed(true); } catch(e3) {}
    try { if (docFileId)  DriveApp.getFileById(docFileId).setTrashed(true);  } catch(e4) {}
  }
}

      jds.forEach(function(jd) {
        try { var r = typeof saveRequirementV284_ === 'function' ? saveRequirementV284_(jd) : ''; jd.reqId = r; saved.push(jd); } catch(e) {}

    }
    var previews = jds.map(function(jd, idx) {
      return { idx:idx, trade:jd.trade||'', clientName:jd.clientName||'',
               deployCountry:jd.deployCountry||'', quantity:jd.quantity||1 };

    result.batchId = batchId; result.count = jds.length; result.jds = previews;
    result.msg = jds.length + ' positions found — please confirm which to save';
    return result;
  } catch(e) {
    result.msg = 'Parse error: ' + e.message;
    return result;
  }
}

  try {
    if (!email) return null;
    var ss    = getMasterSS_();
    var sheet = ss.getSheetByName(CONFIG.sheetName);
    if (!sheet || sheet.getLastRow() < 2) return null;
    var emailCol = CONFIG.inputColumns.email;

  return null;
}

// S44.F02 — Correct findCandidateByPhone (uses CONFIG.inputColumns.mobile)
function findCandidateByPhone_v291_(phone) {
  try {
    if (!phone) return null;
    var ss    = getMasterSS_();
    var sheet = ss.getSheetByName(CONFIG.sheetName);
    if (!sheet || sheet.getLastRow() < 2) return null;
    var mobileCol = CONFIG.inputColumns.mobile;
    var data      = sheet.getRange(2, mobileCol, sheet.getLastRow() - 1, 1).getValues();
    var suffix    = phone.replace(/\D/g, '').slice(-9);
    for (var i = 0; i < data.length; i++) {
      var rowPhone = String(data[i][0]).replace(/\D/g, '');
      if (rowPhone && rowPhone.endsWith(suffix)) return i + 2;
    }

function processReply_v291_(message) {
  var result = { ok:false, kaiNo:null, updated:[], notFound:false };
  try {
    var from    = message.getFrom();
    var subject = message.getSubject();
    var body    = message.getPlainBody();
    var allText = subject + ' ' + body;

    // Step 1: Find candidate row

    }

    // Check sender email (FIXED: uses CONFIG column via v291 helper)

      if (senderEmail) rowIndex = findCandidateByEmail_v291_(senderEmail);

                       body.match(/(?:\+?971|00971)[0-9]{8,9}/) ||
                       body.match(/(?:\+?966|00966)[0-9]{8,9}/);

    }

    if (!rowIndex) {
      result.notFound = true;
      Logger.log('S44: Reply not matched to any candidate. From:' + from + ' Sub:' + subject);
      return result;
    }

    // Step 2: Extract fields from reply
    var updates = typeof extractFieldsFromReply_ === 'function'
      ? extractFieldsFromReply_(body)
      : {};

    if (Object.keys(updates).length === 0) {
      result.ok = true; result.kaiNo = kaiNo;
      return result;
    }

    // Step 3: Apply with CORRECT column mapping (FIXED)
    var applied = applyReplyUpdates_v291_(rowIndex, updates);
    result.ok      = true;
    result.kaiNo   = kaiNo;

    try {

        appendTimeline_(rowIndex, 'CANDIDATE_REPLY',
          'Reply received — fields updated: ' + applied.join(', '), from);

// Adds Drive text extraction before hard-rejecting >15MB CVs.
// ═══════════════════════════════════════════════════════════════════

function buildGeminiPartsWithFallback_(attachment, emailBody, senderEmail, subject) {
  var maxBytes = (typeof CONFIG !== 'undefined' && CONFIG.maxAttachmentBytes) || 15728640;
  var size     = attachment.getSize();
  var ext      = (attachment.getName() || '').split('.').pop().toLowerCase();

  if (size <= maxBytes) {
    return (typeof buildGeminiParts_ === 'function')
      ? buildGeminiParts_(attachment, emailBody, senderEmail, subject)
      : null;
  }

  Logger.log('S45: Attachment oversized (' + size + ' bytes). Attempting text extraction.');
  var text = '';

  if (ext === 'docx' && typeof extractTextFromDocxSafe_ === 'function') {
    text = extractTextFromDocxSafe_(attachment);
  }
  if ((!text || text.length < 100) && typeof extractTextFromDocViaDrive_ === 'function') {
    try { text = extractTextFromDocViaDrive_(attachment); } catch (e) {}
  }
  if (!text || text.length < 100) {
    try {
      var raw = attachment.getBytes();
      text = String.fromCharCode.apply(null, raw)
               .replace(/[^\x09\x0A\x0D\x20-\x7E]/g, ' ')
               .replace(/\s+/g, ' ').trim().slice(0, 30000);
    } catch (e) {}
  }

  var promptFn = (typeof buildParsePrompt_ === 'function')
    ? buildParsePrompt_
    : function(b, e, s) { return 'Parse this CV. Sender: ' + s; };

  if (text && text.length >= 100) {
    Logger.log('S45: Text extraction succeeded (' + text.length + ' chars).');
    return [
      { text: promptFn(emailBody, senderEmail, subject) },
      { text: 'CV text (extracted from large attachment):\n' + text.slice(0, 30000) }
    ];
  }

  Logger.log('S45: All extraction attempts failed for ' + size + '-byte attachment.');
  return null;
}

      if (threads.length >= maxRun) break;
    }
    if (!threads.length) { Logger.log('S46: No ' + mode + ' threads found.'); return; }

    var runRejected = 0;
    var circuit = { consecutive429: 0, broken: false };
    var maxMs   = (typeof CONFIG !== 'undefined' && CONFIG.maxExecutionMs) || 270000;

    for (var j = 0; j < threads.length; j++) {

      var qs2 = (typeof getQuotaState_ === 'function') ? getQuotaState_() : { cvTotalRemaining: 999 };
      if (qs2.cvTotalRemaining <= 0) break;

  }

  }

  if (typeof moveLabel_ === 'function') moveLabel_(thread, inputLabel, doneLabel);
  return 'PROCESSED';
}

function resolveTenantSheet_(toAddress) {
  if (!toAddress) return null;
  try {
    var match = String(toAddress).toLowerCase().match(/^intake\+([a-z0-9_]+)@karigarai\.com$/);
    var slug  = match ? match[1] : null;

    var masterSS   = (typeof getMasterSS_ === 'function') ? getMasterSS_() : null;
    if (!masterSS) return null;
    var routeSheet = masterSS.getSheetByName('_ClientRouting');
    if (!routeSheet || routeSheet.getLastRow() < 2) return null;
    var rows = routeSheet.getRange(2, 1, routeSheet.getLastRow() - 1, 4).getValues();

    for (var i = 0; i < rows.length; i++) {
      var forwarder  = String(rows[i][0]).toLowerCase().trim();
      var clientSsId = String(rows[i][1]).trim();
      var clientName = String(rows[i][2]).trim();
      var active     = String(rows[i][3]).toUpperCase().trim();

      if (active !== 'TRUE' && active !== '1') continue;

      var matchesSlug     = slug && forwarder.indexOf(slug) >= 0;
      var matchesForwarder = toAddress.toLowerCase() === forwarder;

      if (matchesSlug || matchesForwarder) {
        return {
          ss:         SpreadsheetApp.openById(clientSsId),
          clientName: clientName,
          sheetId:    clientSsId
        };
      }
    }
    Logger.log('S51: No active routing found for: ' + toAddress);
    return null;
  } catch (e) {
    Logger.log('S51: resolveTenantSheet_ error: ' + e.message);
    return null;
  }
}

function getTenantBrand_(ss) {
  var defaults = {
    companyName:       'Al Yousuf Enterprises LLP',
    senderDisplayName: 'Al Yousuf Enterprises LLP',
    signatureLine1:    'Recruitment Team',
    signatureLine2:    'Al Yousuf Enterprises LLP'
  };
  if (!ss) return defaults;
  try {
    var cfg = ss.getSheetByName('_Config');
    if (!cfg || cfg.getLastRow() < 2) return defaults;
    var rows = cfg.getRange(2, 1, cfg.getLastRow() - 1, 2).getValues();
    var map  = {};
    rows.forEach(function(r) { if (r[0]) map[String(r[0]).trim()] = String(r[1]).trim(); });
    return {
      companyName:       map['brand_company'] || defaults.companyName,
      senderDisplayName: map['brand_sender']  || defaults.senderDisplayName,
      signatureLine1:    map['brand_sig1']    || defaults.signatureLine1,
      signatureLine2:    map['brand_sig2']    || defaults.signatureLine2
    };
  } catch (e) { return defaults; }
}
