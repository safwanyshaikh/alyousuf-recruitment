// S32.F01 — Triggered every 30 min: scan inbox for forwarded JDs
function scanForJDs() {
  if (isPipelinePaused_()) return;
  var lock = LockService.getScriptLock();
  if (!lock.tryLock(20000)) return;
  try {
    var label = GmailApp.getUserLabelByName(CONFIG.labels.input);
    if (!label) return;
    var threads  = label.getThreads(0, 20);
    var processed = 0;
    threads.forEach(function(thread) {
      if (processed >= 5) return;
      var msgs   = thread.getMessages();
      var msg    = msgs[msgs.length - 1];
      var sender = msg.getFrom();
      if (!isInternalDomain_(sender)) return;         // internal staff only
      var subject = msg.getSubject() || '';
      var body    = msg.getPlainBody() || '';
      if (!isJdEmail_(subject, body)) return;
      if (processJdEmail_(thread, msg, sender, subject, body)) processed++;
    });
    if (processed > 0) Logger.log('S32: Processed ' + processed + ' JD(s)');
  } catch(e) {
    Logger.log('S32: scanForJDs error: ' + e.message);
  } finally {
    lock.releaseLock();
  }
}

// S32.F02 — Heuristic: is this email a job requirement?
function isJdEmail_(subject, body) {
  var text = (subject + ' ' + body).toLowerCase();
  var hasKeyword = CONFIG_V2.jdKeywords.some(function(kw) {
    return text.indexOf(kw) >= 0;
  });
  if (!hasKeyword) return false;
  var hasQuantity = /\b\d+\s*(nos|pax|position|candidate|vacancy|vacancies|profile)\b/i.test(text);
  var hasRole = /\b(engineer|technician|supervisor|foreman|inspector|operator|welder|fitter|electrician|mechanic|driver|helper|manager|officer|coordinator|mason|painter|rigger|scaffolder|fabricator)\b/i.test(text);
  return hasKeyword && (hasQuantity || hasRole);
}

// S32.F03 — Main JD processor
function processJdEmail_(thread, msg, sender, subject, body) {
  try {
    var jdData = extractJdWithGemini_(body, subject);
    if (!jdData || !jdData.trade) {
      Logger.log('S32: JD extraction failed for: ' + subject); return false;
    }
    var reqId        = generateReqId_();
    jdData.reqId     = reqId;
    jdData.receivedDate = new Date();
    jdData.sourcedBy = sender;
    jdData.status    = 'OPEN';
    jdData.rawJd     = body.slice(0, 2000);
    jdData.shortlistCount = 0;
    jdData.selectedCount  = 0;
    saveRequirement_(jdData);
    var matches = matchCandidatesToRequirement_(jdData, 20);
    sendShortlistToRecruiter_(sender, jdData, matches);
    // Re-label thread
    try {
      var jdLabel   = GmailApp.getUserLabelByName('karigar/jd-received') ||
                      GmailApp.createLabel('karigar/jd-received');
      var inputLabel = GmailApp.getUserLabelByName(CONFIG.labels.input);
      if (inputLabel) thread.removeLabel(inputLabel);
      thread.addLabel(jdLabel);
    } catch(e) {}
    appendLog_({ status:'JD_PROCESSED',
      notes: reqId+' | '+jdData.trade+' | '+(jdData.deployCountry||'')+
             ' | '+(jdData.quantity||1)+' nos | '+matches.length+' matches' });
    Logger.log('S32: ' + reqId + ' — ' + jdData.trade + ' / ' + (jdData.deployCountry||''));
    return true;
  } catch(e) {
    Logger.log('S32: processJdEmail_ error: ' + e.message); return false;
  }
}

// S32.F04 — Gemini-based JD extraction
function extractJdWithGemini_(body, subject) {
  var prompt =
    'You are a GCC recruitment intelligence system.\n' +
    'Extract the job requirement from this email.\n\n' +
    'Email Subject: ' + subject + '\n' +
    'Email Body:\n' + body.slice(0, 4000) + '\n\n' +
    'Return ONLY valid JSON (no markdown, no explanation):\n' +
    '{\n' +
    '  "clientName": "company name or Unknown",\n' +
    '  "deployCountry": "Saudi Arabia|UAE|Qatar|Bahrain|Oman|Kuwait|India|Unknown",\n' +
    '  "trade": "specific trade or position title",\n' +
    '  "quantity": <integer>,\n' +
    '  "minExperience": <years as number>,\n' +
    '  "minAge": <number or null>,\n' +
    '  "maxAge": <number or null>,\n' +
    '  "gccPreference": <true|false>,\n' +
    '  "localTransfer": <true if Saudi iqama transfer required>,\n' +
    '  "visitVisaOK": <true if UAE visit visa candidates acceptable>,\n' +
    '  "certifications": ["cert1","cert2"],\n' +
    '  "urgency": "URGENT|NORMAL|FUTURE",\n' +
    '  "specialRequirements": {\n' +
    '    "height": null, "weight": null, "drivingLicense": false,\n' +
    '    "ecnr": false, "englishLevel": null, "aramcoApproved": false,\n' +
    '    "sapNumber": false, "foodAccommodationProvided": false, "other": []\n' +
    '  },\n' +
    '  "salary": "salary details if mentioned or null",\n' +
    '  "notes": "other important details"\n' +
    '}';
  try {
    var raw   = callGemini_(prompt);
    if (!raw) return null;
    var clean = raw.replace(/```json?\s*/gi,'').replace(/```\s*/g,'').trim();
    var parsed = JSON.parse(clean);
    // Normalise country
    var country = String(parsed.deployCountry||'').toLowerCase();
    for (var k in CONFIG_V2.countryMap) {
      if (country.indexOf(k) >= 0) { parsed.deployCountry = CONFIG_V2.countryMap[k]; break; }
    }
    return parsed;
  } catch(e) {
    Logger.log('S32: extractJdWithGemini_ error: ' + e.message); return null;
  }
}

// S32.F05 — Generate unique Req ID (AYE-REQ-2026-0001)
function generateReqId_() {
  var props   = PropertiesService.getScriptProperties();
  var counter = parseInt(props.getProperty(CONFIG_V2.reqIdCounterKey)||'0') + 1;
  props.setProperty(CONFIG_V2.reqIdCounterKey, String(counter));
  return CONFIG_V2.reqIdPrefix + '-' + new Date().getFullYear() + '-' +
         String(counter).padStart(4,'0');
}

// S32.F07 — Save requirement to _Requirements sheet
function saveRequirement_(jdData) {
  var ss    = getMasterSS_();
  var sheet = ss.getSheetByName(CONFIG_V2.requirementsSheet) ||
              ensureSheet_(ss, CONFIG_V2.requirementsSheet, CONFIG_V2.reqHeaders, '#1e3a5f');
  var certsStr  = (jdData.certifications||[]).join(', ');
  var specStr   = safeStringify_(jdData.specialRequirements||{});
  sheet.appendRow([
    jdData.reqId,          jdData.receivedDate,
    jdData.clientName||'Unknown', jdData.deployCountry||'Unknown',
    jdData.trade||'',      jdData.quantity||1,
    jdData.minExperience||0, jdData.minAge||21, jdData.maxAge||50,
    jdData.gccPreference  ? 'YES':'NO',
    jdData.localTransfer  ? 'YES':'NO',
    jdData.visitVisaOK    ? 'YES':'NO',
    certsStr, jdData.urgency||'NORMAL', jdData.status||'OPEN',
    jdData.sourcedBy||'', specStr,
    jdData.shortlistCount||0, jdData.selectedCount||0,
    jdData.notes||'',     jdData.rawJd||''
  ]);
}

// S32.F08 — Load requirement object by Req ID
function loadRequirement_(reqId) {
  var ss    = getMasterSS_();
  var sheet = ss.getSheetByName(CONFIG_V2.requirementsSheet);
  if (!sheet || sheet.getLastRow() < 2) return null;
  var data  = sheet.getRange(2,1,sheet.getLastRow()-1,CONFIG_V2.reqHeaders.length).getValues();
  for (var i=0; i<data.length; i++) {
    if (String(data[i][0]).trim() === reqId) {
      var r = data[i];
      return {
        reqId:r[0], receivedDate:r[1], clientName:r[2], deployCountry:r[3],
        trade:r[4], quantity:r[5],     minExperience:r[6], minAge:r[7], maxAge:r[8],
        gccPreference: String(r[9]).toUpperCase()==='YES',
        localTransfer: String(r[10]).toUpperCase()==='YES',
        visitVisaOK:   String(r[11]).toUpperCase()==='YES',
        certifications:String(r[12]).split(',').map(function(s){return s.trim();}),
        urgency:r[13], status:r[14], sourcedBy:r[15],
        specialRequirements: safeParseJson_(r[16],{}),
        row: i+2
      };
    }
  }
  return null;
}

// S33.F01 — Match all active candidates to a requirement
function matchCandidatesToRequirement_(req, maxResults) {
  maxResults = maxResults || 20;
  var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
  if (!sheet || sheet.getLastRow() < 2) return [];
  var ic   = CONFIG.inputColumns;
  var ec   = CONFIG_V2.extCol;
  var cols = Math.max(CONFIG.headers.length, ec.timeline);
  var data = sheet.getRange(2, 1, sheet.getLastRow()-1, cols).getValues();
  var scored = [];
  data.forEach(function(row, idx) {
    var av = String(row[ic.active-1]||'').toUpperCase().trim();
    if (av==='SUPERSEDED' || av==='ARCHIVED') return;
    var name  = String(row[ic.name-1]||'').trim();
    var email = String(row[ic.email-1]||'').trim();
    if (!name && !email) return;
    var state = String(row[ec.state-1]||'').trim();
    if (state==='REJECTED' || state==='DEPLOYED') return;
    var cand   = buildCandidateObj_(row, idx+2);
    var result = contextScore_(cand, req);
    if (result.score <= 0) return;
    scored.push({
      rowIndex:       idx+2,
      kaiNo:          cand.kaiNo,
      name:           name,
      trade:          cand.trade,
      nationality:    cand.nationality,
      experience:     cand.experience,
      gulfExp:        cand.gulfExp,
      currentLocation:cand.currentLocation,
      mobilityStatus: cand.mobilityStatus,
      deployScore:    cand.deployScore,
      missingFields:  cand.missingFields,
      email:          email,
      mobile:         cand.mobile,
      matchScore:     result.score,
      matchReasons:   result.reasons,
      matchWarnings:  result.warnings
    });
  });
  scored.sort(function(a,b){ return b.matchScore - a.matchScore; });
  return scored.slice(0, maxResults);
}

// S33.F02 — Context-aware scoring (NON-FLAT, requirement-driven)
function contextScore_(candidate, req) {
  var score = 0;
  var reasons = [], warnings = [];

  // ── Trade match (mandatory gate) ─────────────────────────────
  var ts = tradeSimilarityScore_(candidate.trade, req.trade);
  if (ts === 0) return { score:0, reasons:['Trade mismatch'], warnings:[] };
  score += ts;
  reasons.push(ts>=20 ? 'Trade exact match ✓' : 'Trade family match ('+ts+'/25)');

  // ── Context: Saudi local transfer (dominant signal) ──────────
  if (req.deployCountry==='Saudi Arabia' && req.localTransfer) {
    if (candidate.mobilityStatus==='SAUDI_TRANSFER') {
      score += 45; reasons.push('Saudi transferable iqama ✓ — critical fit');
    } else {
      score -= 35; warnings.push('No Saudi transfer — req mandates iqama transfer');
    }
  }

  // ── Context: UAE visit visa ───────────────────────────────────
  if (req.deployCountry==='UAE' && req.visitVisaOK) {
    if (candidate.mobilityStatus==='UAE_VISIT')     { score+=35; reasons.push('UAE visit visa available ✓ — immediate deploy'); }
    else if (candidate.mobilityStatus==='UAE_EMPLOYED') { score+=18; reasons.push('UAE employed — notice period applies'); }
  }

  // ── Context: Qatar local ──────────────────────────────────────
  if (req.deployCountry==='Qatar' && candidate.mobilityStatus==='QATAR_LOCAL') {
    score += 35; reasons.push('Qatar local transfer candidate ✓');
  }

  // ── Context: India deployment — prefer GCC-returned ──────────
  if (req.deployCountry==='India') {
    if (candidate.mobilityStatus==='GCC_RETURN' || candidate.mobilityStatus==='INDIA_READY') {
      score += 30; reasons.push('Gulf-returned, India-based ✓');
    }
  }

  // ── GCC experience ────────────────────────────────────────────
  var gulf = String(candidate.gulfExp||'').trim();
  if (req.gccPreference) {
    if (gulf && gulf.length>5)  { score+=15; reasons.push('GCC experienced ✓'); }
    else                        { score-=5;  warnings.push('No GCC experience (req prefers)'); }
  } else if (gulf && gulf.length>3) { score+=8; reasons.push('Has GCC experience (bonus)'); }

  // ── Experience ────────────────────────────────────────────────
  var exp    = parseFloat(candidate.experience) || 0;
  var reqExp = parseFloat(req.minExperience)    || 0;
  if (reqExp>0) {
    if      (exp>=reqExp*1.5) { score+=10; reasons.push('Exceeds exp req ('+exp+'yr) ✓'); }
    else if (exp>=reqExp)     { score+=8;  reasons.push('Meets exp req ('+exp+'yr) ✓'); }
    else if (exp>=reqExp*0.7) { score+=3;  warnings.push('Slightly below exp ('+exp+'yr vs '+reqExp+'yr req)'); }
    else                      { score-=5;  warnings.push('Below exp req ('+exp+'yr vs '+reqExp+'yr)'); }
  } else if (exp>0) { score+=5; reasons.push('Has experience: '+exp+'yr'); }

  // ── Age compliance ────────────────────────────────────────────
  var age    = parseInt(candidate.age)  || 0;
  var minAge = parseInt(req.minAge)     || 21;
  var maxAge = parseInt(req.maxAge)     || 50;
  if (age>0) {
    if (age>=minAge && age<=maxAge) { score+=5; reasons.push('Age compliant: '+age+' ✓'); }
    else { score-=20; warnings.push('Age non-compliant: '+age+' (req '+minAge+'–'+maxAge+')'); }
  } else { warnings.push('Age unknown — DOB missing'); }

  // ── Deployability score ───────────────────────────────────────
  var ds = parseInt(candidate.deployScore)||0;
  if      (ds>=80) { score+=10; reasons.push('High deployability ('+ds+'/100) ✓'); }
  else if (ds>=60) { score+=5;  reasons.push('Moderate deployability ('+ds+')'); }
  else if (ds>0)   { warnings.push('Low deployability ('+ds+'/100)'); }

  // ── ECR/ECNR ─────────────────────────────────────────────────
  if (req.specialRequirements && req.specialRequirements.ecnr) {
    if (String(candidate.ecrStatus||'').toUpperCase()==='ECNR') { score+=5; reasons.push('ECNR passport ✓'); }
    else warnings.push('ECR status unknown — verify before visa');
  }

  // ── Certification match ───────────────────────────────────────
  if (req.certifications && req.certifications.length>0) {
    var candText = (String(candidate.education||'')+' '+String(candidate.gulfExp||'')).toLowerCase();
    var certHits = req.certifications.filter(function(c){ return c && candText.indexOf(c.toLowerCase())>=0; });
    if (certHits.length>0) { score+=certHits.length*3; reasons.push('Certs: '+certHits.join(', ')+' ✓'); }
  }

  return { score: Math.max(0,Math.min(100,Math.round(score))), reasons:reasons, warnings:warnings };
}

// S33.F03 — Trade similarity scoring (fuzzy, family-aware)
function tradeSimilarityScore_(trade1, trade2) {
  if (!trade1 || !trade2) return 0;
  var t1 = String(trade1).toLowerCase().trim();
  var t2 = String(trade2).toLowerCase().trim();
  if (t1===t2) return 25;
  if (t1.indexOf(t2)>=0 || t2.indexOf(t1)>=0) return 20;
  var w1 = t1.split(/\W+/).filter(function(w){return w.length>3;});
  var w2 = t2.split(/\W+/).filter(function(w){return w.length>3;});
  var ov = w1.filter(function(w){return w2.indexOf(w)>=0;});
  if (ov.length>=2) return 15;
  if (ov.length===1) return 8;
  // Trade family check
  for (var fam in CONFIG_V2.tradeFamilies) {
    var kws = CONFIG_V2.tradeFamilies[fam];
    var in1 = kws.some(function(k){return t1.indexOf(k)>=0;});
    var in2 = kws.some(function(k){return t2.indexOf(k)>=0;});
    if (in1 && in2) return 12;
  }
  return 0;
}

// S34.F01 — Determine GCC mobility status from candidate profile
function computeMobilityStatus_(candidate) {
  var loc   = String(candidate.currentLocation||'').toLowerCase();
  var gulf  = String(candidate.gulfExp||'').toLowerCase();
  var state = String(candidate.empStatus||'').toLowerCase();
  var notes = (String(candidate.kaiAssessment||'')+' '+gulf).toLowerCase();

  if (/transferable|iqama transfer|local transfer/.test(notes) &&
      /saudi|riyadh|jeddah|jubail|ksa/.test(notes+loc))    return 'SAUDI_TRANSFER';
  if (/visit visa/.test(notes) &&
      (/uae|dubai|abu dhabi/.test(loc)||/uae|dubai/.test(notes))) return 'UAE_VISIT';
  if (/uae|dubai|abu dhabi|sharjah/.test(loc) &&
      /employed|working|job/.test(state+notes))             return 'UAE_EMPLOYED';
  if (/qatar|doha/.test(loc))                               return 'QATAR_LOCAL';
  if (/overstay|illegal|expired visa/.test(notes))          return 'OVERSTAY_RISK';
  if (/exit permit|final exit|absconding/.test(notes))      return 'EXIT_REQUIRED';
  if (/india|mumbai|delhi|hyderabad|kerala|bangalore|chennai/.test(loc) &&
      gulf && gulf.length>5)                                return 'GCC_RETURN';
  if (/india|mumbai|delhi|hyderabad|kerala|bangalore|chennai/.test(loc)) return 'INDIA_READY';
  return 'UNKNOWN';
}

// S34.F02 — Compute deployability score 0–100
function computeDeployabilityScore_(candidate) {
  var score = 0;
  // Passport validity (20 pts)
  var ppExp = candidate.passportExpiry;
  if (ppExp) {
    var pp = (ppExp instanceof Date) ? ppExp : new Date(String(ppExp));
    if (!isNaN(pp)) {
      var mo = (pp - new Date()) / (1000*60*60*24*30);
      if      (mo>=18) score+=20;
      else if (mo>=12) score+=15;
      else if (mo>=6)  score+=8;
    }
  }
  // Mobility (25 pts)
  var MOB_PTS = {INDIA_READY:18,GCC_RETURN:22,UAE_VISIT:25,SAUDI_TRANSFER:25,
                 QATAR_LOCAL:25,UAE_EMPLOYED:15,OVERSTAY_RISK:0,EXIT_REQUIRED:0,UNKNOWN:5};
  score += (MOB_PTS[String(candidate.mobilityStatus||'').toUpperCase()] || 5);
  // Employment/notice (20 pts)
  var emp    = String(candidate.empStatus||'').toLowerCase();
  var notice = parseInt(candidate.noticeDays)||99;
  if (/unemployed|not working|immediately|available/.test(emp)||notice===0) score+=20;
  else if (notice<=15) score+=15;
  else if (notice<=30) score+=10;
  else if (notice<=60) score+=5;
  // Medical (15 pts)
  var med = String(candidate.medicalStatus||'').toLowerCase();
  if (/fit|cleared|passed/.test(med))  score+=15;
  else if (/pending|scheduled/.test(med)) score+=8;
  else score+=5;
  // Gulf exp (10 pts)
  if (String(candidate.gulfExp||'').length>5) score+=10;
  // ECR/ECNR (10 pts)
  var ecr = String(candidate.ecrStatus||'').toUpperCase();
  if (ecr==='ECNR') score+=10; else if (ecr==='ECR') score+=6; else score+=3;
  return Math.min(100, Math.round(score));
}

// S34.F03 — Get list of missing critical field keys
function getMissingFields_(candidate, req) {
  var missing = [];
  if (!candidate.age && !candidate.dob)    missing.push('dob');
  if (!candidate.mobile)                    missing.push('mobile');
  if (!candidate.currentLocation)           missing.push('currentLoc');
  if (!candidate.empStatus)                 missing.push('empStatus');
  if (!candidate.passportExpiry)            missing.push('passportExp');
  if (!candidate.ecrStatus)                 missing.push('ecrStatus');
  if (!candidate.gulfExp||String(candidate.gulfExp).length<3) missing.push('gulfExp');
  if (!candidate.nationality)               missing.push('nationality');
  if (req) {
    if (req.localTransfer && candidate.mobilityStatus!=='SAUDI_TRANSFER') {
      if (!/iqama|transferable/.test(String(candidate.kaiAssessment||'').toLowerCase()))
        missing.push('iqamaStatus');
    }
    if (req.specialRequirements && req.specialRequirements.drivingLicense) missing.push('drivingLicense');
    if (req.specialRequirements && req.specialRequirements.ecnr && !candidate.ecrStatus) missing.push('ecrStatus');
  }
  return missing;
}

// S34.F04 — Batch refresh mobility + deployability for all candidates
function refreshMobilityAndDeployability() {
  var ss    = getMasterSS_();
  var sheet = ss.getSheetByName(CONFIG.sheetName);
  if (!sheet || sheet.getLastRow()<2) return;
  ensureExtendedColumns_(sheet);
  var ic   = CONFIG.inputColumns;
  var ec   = CONFIG_V2.extCol;
  var rows = sheet.getLastRow()-1;
  var cols = Math.max(CONFIG.headers.length, ec.timeline);
  var data = sheet.getRange(2,1,rows,cols).getValues();
  var bKai=[],bMob=[],bDs=[],bMiss=[],bState=[];
  data.forEach(function(row) {
    var av = String(row[ic.active-1]||'').toUpperCase().trim();
    var name=String(row[ic.name-1]||'').trim(), email=String(row[ic.email-1]||'').trim();
    if (av==='SUPERSEDED'||av==='ARCHIVED'||(!name&&!email)) {
      bKai.push([row[ec.kaiNo-1]||'']); bMob.push(['']); bDs.push([0]);
      bMiss.push(['']); bState.push([row[ec.state-1]||'']); return;
    }
    var cand = buildCandidateObj_(row, 0);
    var kaiNo = cand.kaiNo || generateKaiNo_();
    bKai.push([kaiNo]);
    var mob = computeMobilityStatus_(cand);
    bMob.push([mob]);
    cand.mobilityStatus = mob;
    bDs.push([computeDeployabilityScore_(cand)]);
    bMiss.push([safeStringify_(getMissingFields_(cand,null))]);
    var state = cand.state || deriveCandidateState_(cand);
    bState.push([state]);
  });
  sheet.getRange(2,ec.kaiNo,      rows,1).setValues(bKai);
  sheet.getRange(2,ec.mobility,   rows,1).setValues(bMob);
  sheet.getRange(2,ec.deployScore,rows,1).setValues(bDs);
  sheet.getRange(2,ec.missingFields,rows,1).setValues(bMiss);
  sheet.getRange(2,ec.state,      rows,1).setValues(bState);
  SpreadsheetApp.flush();
  Logger.log('S34: Mobility/deployability refreshed for '+rows+' rows');
  try { SpreadsheetApp.getUi().alert('✓ Mobility & deployability refreshed for '+rows+' candidates.'); } catch(e) {}
}

// S34.F05 — Derive candidate state from parsed fields
function deriveCandidateState_(candidate) {
  if (!candidate.mobile && !candidate.email)  return 'INCOMPLETE_CONTACT';
  if (!candidate.trade||/unknown|n\/a/i.test(candidate.trade)) return 'UNKNOWN_TRADE';
  if (!candidate.experience && (!candidate.age||candidate.age<21)) return 'FRESHER_POOL';
  var v = String(candidate.verdict||'').toUpperCase();
  if (v==='SHORTLISTED')  return 'SHORTLISTED';
  if (v==='REJECT')       return 'REJECTED';
  if (v==='NEEDS_REVIEW'||v==='NEEDS_CALL') return 'UNDER_REVIEW';
  return 'PARSED';
}

  sheet.getRange(1, newStart, 1, newCount)

    .setFontWeight('bold').setHorizontalAlignment('center');
  Logger.log('S34: Extended columns added (cols 25–38)');
}

// S35.F02 — Bulk send missing info to INCOMPLETE_CONTACT / UNKNOWN_TRADE candidates
function sendBulkMissingInfo() {
  var ss    = getMasterSS_();
  var sheet = ss.getSheetByName(CONFIG.sheetName);
  if (!sheet||sheet.getLastRow()<2) return;
  var ic   = CONFIG.inputColumns;
  var ec   = CONFIG_V2.extCol;
  var cols = Math.max(CONFIG.headers.length, ec.timeline);
  var data = sheet.getRange(2,1,sheet.getLastRow()-1,cols).getValues();
  var sent=0;
  data.forEach(function(row,i) {
    if (sent>=20) return;
    var state = String(row[ec.state-1]||'').trim();
    if (state!=='INCOMPLETE_CONTACT'&&state!=='UNKNOWN_TRADE') return;
    var email = String(row[ic.email-1]||'').trim();
    if (!email) return;
    // Throttle — don't resend within 3 days
    var lc = row[ec.lastContact-1];
    if (lc) {
      var lcDate = (lc instanceof Date)?lc:new Date(String(lc));
      if (!isNaN(lcDate) && (new Date()-lcDate)<3*24*60*60*1000) return;
    }
    var cand    = buildCandidateObj_(row,i+2);
    var missing = getMissingFields_(cand,null);
    if (missing.length===0) return;
    var res = sendMissingInfoRequest(i+2, missing, null, row[ic.trade-1]||row[ic.positionApplied-1]);
    if (res.ok) sent++;
  });
  Logger.log('S35: Bulk missing info sent: '+sent);
  try { SpreadsheetApp.getUi().alert('✓ Missing info requests sent to '+sent+' candidates.'); } catch(e) {}
}

// S37.F01 — Compute next recommended action for a candidate
function getNextAction_(candidate) {
  var state   = String(candidate.state||'').toUpperCase();
  var mob     = String(candidate.mobilityStatus||'').toUpperCase();
  var ds      = parseInt(candidate.deployScore)||0;
  var missing = Array.isArray(candidate.missingFields)?candidate.missingFields:[];
  var verdict = String(candidate.verdict||'').toUpperCase();

  if (missing.length>0 && state!=='DEPLOYED') {
    return {action:'REQUEST_MISSING_INFO',label:'Request Missing Info',icon:'📋',urgency:'HIGH',
      detail:'Missing: '+missing.slice(0,3).map(function(k){
        return CONFIG_V2.missingInfoDefs[k]?CONFIG_V2.missingInfoDefs[k].label:k;
      }).join(', ')};
  }
  switch(state) {
    case 'CLIENT_SENT':      return {action:'FOLLOW_UP_CLIENT',label:'Follow Up with Client',icon:'📞',urgency:'MEDIUM',detail:'Awaiting client decision'};
    case 'CLIENT_SELECTED':  return {action:'INITIATE_OFFER',label:'Issue Offer Letter',icon:'📄',urgency:'HIGH',detail:'Client selected — prepare offer'};
    case 'OFFER_ISSUED':     return {action:'COLLECT_DOCS',label:'Collect Documents',icon:'📁',urgency:'HIGH',detail:'Passport, certs, photos needed'};
    case 'VISA_PROCESS':     return {action:'TRACK_VISA',label:'Track Visa Status',icon:'🛂',urgency:'MEDIUM',detail:'Visa processing in progress'};
    case 'MEDICAL_PENDING':  return {action:'SCHEDULE_MEDICAL',label:'Schedule Medical',icon:'🏥',urgency:'HIGH',detail:'Medical clearance required'};
    case 'ECR_PENDING':      return {action:'TRACK_ECR',label:'Track ECR Clearance',icon:'📜',urgency:'HIGH',detail:'ECR clearance before visa stamping'};
    case 'READY_TO_DEPLOY':  return {action:'BOOK_TRAVEL',label:'Book Travel & Deploy',icon:'✈️',urgency:'HIGH',detail:'All clear — arrange travel'};
    case 'INCOMPLETE_CONTACT':return {action:'REQUEST_MISSING_INFO',label:'Get Contact Details',icon:'📋',urgency:'HIGH',detail:'No mobile/email on file'};
    case 'UNKNOWN_TRADE':    return {action:'CLARIFY_TRADE',label:'Clarify Trade / Skill',icon:'🔍',urgency:'MEDIUM',detail:'Trade classification missing'};
    case 'FRESHER_POOL':     return {action:'POOL_REVIEW',label:'Pool Review',icon:'🎓',urgency:'LOW',detail:'Fresher — no active requirements'};
  }
  if (mob==='SAUDI_TRANSFER'&&verdict==='SHORTLISTED')
    return {action:'SUBMIT_CLIENT',label:'Submit — Saudi Transfer Ready',icon:'🇸🇦',urgency:'HIGH',detail:'Transferable iqama — high priority'};
  if (mob==='UAE_VISIT')
    return {action:'URGENT_SHORTLIST',label:'Urgent: UAE Visit Visa Available',icon:'🇦🇪',urgency:'HIGH',detail:'Visit visa — immediate deploy possible'};
  if (ds>=80)
    return {action:'SHORTLIST_NOW',label:'High Deploy Score — Shortlist',icon:'⭐',urgency:'MEDIUM',detail:'Deploy score '+ds+'/100'};
  if (verdict==='NEEDS_CALL')
    return {action:'CALL_CANDIDATE',label:'Call Candidate',icon:'📞',urgency:'NORMAL',detail:'Needs phone verification'};
  if (verdict==='NEEDS_REVIEW')
    return {action:'REVIEW_PROFILE',label:'Review Profile',icon:'👁️',urgency:'NORMAL',detail:'Shortlisting decision needed'};
  return {action:'REVIEW_PROFILE',label:'Review Profile',icon:'👁️',urgency:'LOW',detail:'No specific action flagged'};
}

// S37.F02 — Full intelligence panel data for a candidate (dashboard side panel)
function getCandidateIntelligence(rowIndex) {
  try {
    var sheet  = getMasterSS_().getSheetByName(CONFIG.sheetName);
    var ic     = CONFIG.inputColumns;
    var ec     = CONFIG_V2.extCol;
    // Include V282 extended cols 39-42 + recruiter ownership cols 43-44
    var maxCol = Math.max(CONFIG.headers.length, ec.timeline, 44);
    var cols   = Math.min(maxCol, sheet.getLastColumn());
    var row    = sheet.getRange(rowIndex, 1, 1, cols).getValues()[0];
    var cand   = buildCandidateObj_(row, rowIndex);
    var missing = Array.isArray(cand.missingFields)?cand.missingFields:safeParseJson_(cand.missingFields,[]);
    var missingLabels = missing.map(function(k){
      return CONFIG_V2.missingInfoDefs[k]?CONFIG_V2.missingInfoDefs[k].label:k;
    });
    var timeline = safeParseJson_(row[ec.timeline-1],[]);
    // V282 extended fields (cols 39-42)
    var ec2     = (typeof PATCH_V282 !== 'undefined') ? PATCH_V282.extCol2 : {educationEnum:39,techReview:40,reviewedBy:41,top3Positions:42};
    var rawEdu  = cand.education;
    var eduEnum = (row.length >= ec2.educationEnum) ? String(row[ec2.educationEnum-1]||'') : '';

    var top3       = top3Raw ? top3Raw.split(' | ').filter(Boolean).slice(0,3) : [];
    if (!top3.length && typeof computeTop3Positions_ === 'function') {
      top3 = computeTop3Positions_({
        trade:cand.trade, educationEnum:eduEnum,
        experience:cand.experience, gulfExp:cand.gulfExp
      });
    }
    return {
      kaiNo:          cand.kaiNo,
      name:           cand.name,
      state:          cand.state,
      mobilityStatus: cand.mobilityStatus,
      deployScore:    cand.deployScore,

      noticeDays:     cand.noticeDays,
      medicalStatus:  cand.medicalStatus,
      empStatus:      cand.empStatus,
      currentLocation:cand.currentLocation,
      missingFields:  missingLabels,
      missingKeys:    missing,
      nextAction:     getNextAction_(cand),
      timeline:       timeline.slice(-10).reverse(),
      kaiAssessment:  row[ic.kaiAssessment-1],
      scoreBreakdown: row[ic.scoreBreakdown-1],
      reqMatch:       cand.reqMatch,
      gulfExp:        cand.gulfExp,
      cvLink:         cand.cvLink,
      mobile:         cand.mobile,
      email:          cand.email,
      verdict:        cand.verdict,
      score:          cand.score,
      nationality:    cand.nationality,
      education:      rawEdu,
      educationEnum:  eduEnum,
      trade:          cand.trade,
      experience:     cand.experience,
      positionApplied:cand.positionApplied,
      techReview:      techReview,
      reviewedBy:      reviewedBy,
      top3Positions:   top3,
      // Recruiter ownership (cols 43/44)
      recruiterOwner:  row.length >= 43 ? String(row[42]||'').trim() : '',
      recruiterRemark: row.length >= 44 ? String(row[43]||'').trim() : ''
    };
  } catch(e) {
    Logger.log('S37: getCandidateIntelligence error: '+e.message);

// S37.F03 — getDashboardDataV2 (OS dashboard data feed)
function getDashboardDataV2() {
  try {
    // Only serve from cache if KAI Nos are already populated (avoid stale data after first run)
    var cache  = CacheService.getScriptCache();
    var cached = cache.get('kai_dashboard_v2');
    if (cached) {
      try {
        var p = JSON.parse(cached);
        // Cache is valid only if records exist AND first record has a KAI No
        if (p && p.records && p.records.length > 0 && p.records[0].kaiNo) return p;
      } catch(e) {}
    }
    var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
    if (!sheet) return {metrics:{},records:[],version:CONFIG.version+'-OS'};
    var lr = sheet.getLastRow();
    if (lr<2) return {metrics:{total:0},records:[],version:CONFIG.version+'-OS'};
    var ic   = CONFIG.inputColumns;
    var ec   = CONFIG_V2.extCol;
    // Ensure extended columns exist before reading
    if (sheet.getLastColumn() < ec.kaiNo) ensureExtendedColumns_(sheet);
    // Read up to col 44 (includes recruiterOwner col 43, recruiterRemark col 44)
    var cols = Math.max(CONFIG.headers.length, ec.timeline, 44);
    var actualCols = Math.min(cols, sheet.getLastColumn());
    var data = sheet.getRange(2,1,lr-1,actualCols).getValues();
    var tz   = 'Asia/Kolkata';
    var now  = new Date();
    var today = Utilities.formatDate(now,tz,'yyyy-MM-dd');
    var metrics = {
      total:0,shortlisted:0,clientSent:0,deployed:0,
      needsReview:0,needsCall:0,incompleteContact:0,unknownTrade:0,
      todayCount:0,deployReady:0,saudiTransfer:0,uaeVisit:0,
      fresherPool:0,hold:0
    };
    var kaiNosToWrite = {};  // rowIndex → kaiNo for batch write
    var records=[], trades={}, nationalities={}, statesMap={}, mobMap={};
    data.forEach(function(row,idx) {
      var av = String(row[ic.active-1]||'').toUpperCase().trim();
      if (av==='SUPERSEDED'||av==='ARCHIVED') return;
      var name=String(row[ic.name-1]||'').trim(), email=String(row[ic.email-1]||'').trim();
      if (!name&&!email) return;
      metrics.total++;
      var verdict = String(row[ic.verdict-1]||'').toUpperCase();
      var state   = String(row[ec.state-1]||'').trim();
      var mob     = String(row[ec.mobility-1]||'').trim();
      var ds      = parseInt(row[ec.deployScore-1])||0;
      var missRaw = safeParseJson_(row[ec.missingFields-1],[]);
      if (verdict==='SHORTLISTED')              metrics.shortlisted++;
      if (state==='CLIENT_SENT')                metrics.clientSent++;
      if (state==='DEPLOYED')                   metrics.deployed++;
      if (verdict==='NEEDS_REVIEW')             metrics.needsReview++;
      if (verdict==='NEEDS_CALL')               metrics.needsCall++;
      if (state==='INCOMPLETE_CONTACT')         metrics.incompleteContact++;
      if (state==='UNKNOWN_TRADE')              metrics.unknownTrade++;
      if (state==='READY_TO_DEPLOY'||ds>=80)    metrics.deployReady++;
      if (state==='FRESHER_POOL')               metrics.fresherPool++;
      if (state==='HOLD')                       metrics.hold++;
      if (mob==='SAUDI_TRANSFER')               metrics.saudiTransfer++;
      if (mob==='UAE_VISIT')                    metrics.uaeVisit++;
      var appDt = row[ic.applicationDate-1];
      if (appDt instanceof Date && Utilities.formatDate(appDt,tz,'yyyy-MM-dd')===today) metrics.todayCount++;
      var t=String(row[ic.trade-1]||''), n=String(row[ic.nationality-1]||'');
      if (t) trades[t]=true; if (n) nationalities[n]=true;
      if (state) statesMap[state]=true; if (mob) mobMap[mob]=true;
      var cand   = buildCandidateObj_(row, idx+2);
      var na     = getNextAction_(cand);
      // KAI No: use existing or generate new — batch write at end for performance
      var kaiNo = String(row[ec.kaiNo-1]||'').trim();
      if (!kaiNo) {
        kaiNo = generateKaiNo_();
        kaiNosToWrite[idx+2] = kaiNo;  // collected for batch write below
      }
      records.push({
        rowIndex:        idx+2,
        kaiNo:           kaiNo,
        name:            name,
        nationality:     n,
        age:             parseInt(row[ic.age-1])||0,
        currentLocation: String(row[ec.currentLocation-1]||''),
        education:       String(row[ic.education-1]||''),
        positionApplied: String(row[ic.positionApplied-1]||''),
        trade:           t,
        mobile:          String(row[ic.mobile-1]||'').replace(/^'/,''),
        cvLink:          String(row[ic.cvLink-1]||''),
        state:           state,
        mobilityStatus:  mob,
        deployScore:     ds,
        verdict:         String(row[ic.verdict-1]||'').toUpperCase().trim(),
        flags:           String(row[ic.flags-1]||''),
        score:           parseInt(row[ic.score-1])||0,
        missingCount:    missRaw.length,
        nextActionIcon:  na.icon,
        nextActionLabel: na.label,
        nextActionUrgency:na.urgency,
        applicationDate: appDt instanceof Date ?
          Utilities.formatDate(appDt,tz,'yyyy-MM-dd') : '',
        // ── Core fields needed for search & modal ──────────────────
        email:           String(row[ic.email-1]||'').trim(),
        experience:      parseFloat(row[ic.experience-1])||0,
        gulfExp:         String(row[ic.gulf-1]||'').trim(),
        kaiAssessment:   String(row[ic.kaiAssessment-1]||'').trim(),
        notes:           String(row[ic.notes-1]||'').trim(),
        scoreBreakdown:  String(row[ic.scoreBreakdown-1]||'').trim(),
        qualification:   String(row[ic.education-1]||'').trim(),
        displayLocation: String(row[ec.currentLocation-1]||'').trim(),
        passportStatus:  String(row[ec.ecrStatus-1]||'').trim(),
        // ── Extended v282 columns (39-42) — filters + Top3 display ─
        // educationEnum: read col 39; classify on-the-fly if empty
        educationEnum: (function(){
          var en = row.length >= 39 ? String(row[38]||'').trim() : '';
          if (!en && typeof classifyEducation_ === 'function') {
            en = classifyEducation_(String(row[ic.education-1]||''));
          }
          return en;
        }()),
        techReview:      row.length >= 40 ? String(row[39]||'').trim() : '',
        top3Positions:   row.length >= 42
          ? String(row[41]||'').trim().split(' | ').filter(Boolean)
          : [],
        // ── Recruiter ownership (cols 43/44) ───────────────────────
        recruiterOwner:  row.length >= 43 ? String(row[42]||'').trim() : '',
        recruiterRemark: row.length >= 44 ? String(row[43]||'').trim() : ''
      });
    });

    // Batch-write any newly generated KAI Nos to the sheet (permanent, survives all script changes)
    var newKaiKeys = Object.keys(kaiNosToWrite);
    if (newKaiKeys.length > 0) {
      newKaiKeys.forEach(function(rowIdx) {
        try { sheet.getRange(parseInt(rowIdx), ec.kaiNo).setValue(kaiNosToWrite[rowIdx]); } catch(e) {}
      });
      try { SpreadsheetApp.flush(); } catch(e) {}
      Logger.log('S37: Generated ' + newKaiKeys.length + ' new KAI Nos');
    }

    records.sort(function(a,b){
      // Primary: newest CV date first. Tiebreaker: row order in sheet (stable, no shortlist boost).
      var d = String(b.applicationDate).localeCompare(String(a.applicationDate));
      return d !== 0 ? d : a.rowIndex - b.rowIndex;
    });
    var result = {
      metrics:       metrics,
      records:       records,
      trades:        Object.keys(trades).sort(),
      nationalities: Object.keys(nationalities).sort(),
      states:        Object.keys(statesMap).sort(),
      mobilityStates:Object.keys(mobMap).sort(),
      quota:         getQuotaState_(),
      openReqs:      getOpenRequirements(),
      version:       CONFIG.version+'-OS'
    };
    try { cache.put('kai_dashboard_v2', JSON.stringify(result), 90); } catch(e) {}
    return result;
  } catch(err) {
    Logger.log('S37: getDashboardDataV2 error: '+err.message);
    return {metrics:{},records:[],error:err.message,version:CONFIG.version+'-OS'};
  }
}

  if (!recruiter) return { ok:false, msg:'Recruiter name required' };

}

// S38.F01 — Bulk send missing info to selected row indices
function bulkSendMissingInfo(rowIndices) {
  if (!Array.isArray(rowIndices)||rowIndices.length===0) return {ok:false,msg:'No rows'};
  var sent=0,failed=0;
  var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
  var ic = CONFIG.inputColumns, ec = CONFIG_V2.extCol;
  var cols = Math.max(CONFIG.headers.length, ec.timeline);
  rowIndices.forEach(function(ri) {
    try {
      var row  = sheet.getRange(ri,1,1,cols).getValues()[0];
      var cand = buildCandidateObj_(row, ri);
      var miss = getMissingFields_(cand, null);
      if (miss.length===0) return;
      var res = sendMissingInfoRequest(ri, miss, null, row[ic.trade-1]);
      if (res.ok) sent++; else failed++;
    } catch(e) { failed++; }
  });
  return {ok:true, sent:sent, failed:failed, msg:'Sent: '+sent+', Failed: '+failed};
}

// S38.F02 — Bulk update candidate state
function bulkUpdateState(rowIndices, newState) {
  if (!Array.isArray(rowIndices)||!newState) return {ok:false};
  var updated=0;
  rowIndices.forEach(function(ri) {
    if (updateCandidateState(ri, newState, 'recruiter-bulk').ok) updated++;
  });
  return {ok:true, updated:updated};
}

// S38.F03 — Bulk shortlist for a requirement
function bulkShortlistForReq(rowIndices, reqId) {
  if (!rowIndices||!reqId) return {ok:false};
  var sheet = getMasterSS_().getSheetByName(CONFIG.sheetName);
  var ec    = CONFIG_V2.extCol;
  var updated=0;
  rowIndices.forEach(function(ri) {
    try {
      sheet.getRange(ri, ec.state).setValue('SHORTLISTED');
      var ex = String(sheet.getRange(ri, ec.reqMatch).getValue()||'');
      if (ex.indexOf(reqId)<0) sheet.getRange(ri, ec.reqMatch).setValue(ex?ex+','+reqId:reqId);
      appendTimeline_(ri,'SHORTLISTED_FOR_REQ', reqId,'recruiter');
      updated++;
    } catch(e) {}
  });
  // Increment shortlist count in requirements
  try {
    var rs = getMasterSS_().getSheetByName(CONFIG_V2.requirementsSheet);
    if (rs&&rs.getLastRow()>1) {
      var rd = rs.getRange(2,1,rs.getLastRow()-1,1).getValues();
      for (var i=0;i<rd.length;i++) {
        if (String(rd[i][0])===reqId) {
          var cc = rs.getRange(i+2,18);
          cc.setValue((parseInt(cc.getValue())||0)+updated); break;
        }
      }
    }
  } catch(e) {}
  return {ok:true, updated:updated};
}
